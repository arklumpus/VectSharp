148 /// <summary>
149 /// The minimum distance that is enforced between consecutive gradient stops.

172 /// <summary>
173 /// Creates a new <see cref="GradientStops"/> instance containing the specified gradient stops.
174 /// </summary>
175 /// <param name="gradientStops">The gradient stops that will be contained in the <see cref="GradientStops"/> object.</param>
176 public GradientStops(IEnumerable<GradientStop> gradientStops)
177 {
178 List<GradientStop> stops = (from el in gradientStops orderby el.Offset ascending select el).ToList();
179
180 if (stops.Count == 0)
181 {
182 stops.Add(new GradientStop(Colour.FromRgba(0, 0, 0, 0), 0));
183 }
184
185 if (stops[0].Offset > 0)
186 {
187 stops.Insert(0, new GradientStop(stops[0].Colour, 0));
188 }
189
190 if (stops[stops.Count - 1].Offset < 1)
191 {
192 stops.Add(new GradientStop(stops[stops.Count - 1].Colour, 1));
193 }
194
195 for (int i = 1; i < stops.Count - 1; i++)
196 {
197 bool closeToPrevious = (stops[i].Offset - stops[i - 1].Offset < StopTolerance);
198 bool closeToNext = (stops[i + 1].Offset - stops[i].Offset < StopTolerance);
199
200 if (closeToPrevious && !closeToNext)
201 {
202 stops[i] = new GradientStop(stops[i].Colour, stops[i - 1].Offset + StopTolerance);
203 }
204 else if (!closeToPrevious && closeToNext)
205 {
206 stops[i] = new GradientStop(stops[i].Colour, stops[i + 1].Offset - StopTolerance);
207 }
208 else if (closeToPrevious && closeToNext)
209 {
210 stops.RemoveAt(i);
211 i--;
212 }
213 }
214
215 this.gradientStops = ImmutableList.Create(stops.ToArray());
216 }

227 /// <summary>
228 /// Gets the colour at a certain position on the gradient.
229 /// </summary>
230 /// <param name="position">The position in the gradient (ranging from 0 to 1).</param>
231 /// <returns>The colour of the gradient at the specified position.</returns>
232 public Colour GetColourAt(double position)
233 {
234 if (position <= 0)
235 {
236 return this.gradientStops[0].Colour;
237 }
238 else if (position >= 1)
239 {
240 return this.gradientStops[this.gradientStops.Count - 1].Colour;
241 }
242 else
243 {
244 for (int i = 1; i < this.gradientStops.Count; i++)
245 {
246 if (this.gradientStops[i - 1].Offset < position && this.gradientStops[i].Offset >= position)
247 {
248 double factor = (position - this.gradientStops[i - 1].Offset) / (this.gradientStops[i].Offset - this.gradientStops[i - 1].Offset);
249
250 return Colour.FromRgba(this.gradientStops[i - 1].Colour.R * (1 - factor) + this.gradientStops[i].Colour.R * factor,
251 this.gradientStops[i - 1].Colour.G * (1 - factor) + this.gradientStops[i].Colour.G * factor,
252 this.gradientStops[i - 1].Colour.B * (1 - factor) + this.gradientStops[i].Colour.B * factor,
253 this.gradientStops[i - 1].Colour.A * (1 - factor) + this.gradientStops[i].Colour.A * factor);
254 }
255 }
256 return Colour.FromRgba(0, 0, 0, 0);
257 }
258 }

263 /// <param name="stops">The <see cref="GradientStops"/> to convert.</param>
264 public static implicit operator Func<double, Colour>(GradientStops stops) => stops.GetColourAt;

339 /// <summary>
340 /// Returns a <see cref="LinearGradientBrush"/> with the same gradient stops as the current instance, whose start and end point correspond to the points of the current instance in the
341 /// original reference frame of the <paramref name="referenceGraphics"/>. This involves computing the current transform matrix of the <paramref name="referenceGraphics" />, inverting it,
342 /// and applying the inverse matrix to the <see cref="StartPoint"/> and <see cref="EndPoint"/> of the current instance.
343 /// </summary>
344 /// <param name="referenceGraphics">The <see cref="Graphics"/> whose original reference frame is to be used.</param>
345 /// <returns>A <see cref="LinearGradientBrush"/> with the same gradient stops as the current instance, whose start and end point correspond to the points of the current instance in the
346 /// original reference frame of the <paramref name="referenceGraphics"/>.</returns>
347 public LinearGradientBrush RelativeTo(Graphics referenceGraphics)
348 {
349 Stack<double[,]> transformMatrix = new Stack<double[,]>();
350 double[,] currMatrix = new double[3, 3] { { 1, 0, 0 }, { 0, 1, 0 }, { 0, 0, 1 } };
351
352 for (int i = 0; i < referenceGraphics.Actions.Count; i++)
353 {
354 if (referenceGraphics.Actions[i] is TransformAction)
355 {
356 TransformAction trf = referenceGraphics.Actions[i] as TransformAction;
357
358 if (trf.Delta != null)
359 {
360 currMatrix = Graphics.Multiply(currMatrix, Graphics.TranslationMatrix(trf.Delta.Value.X, trf.Delta.Value.Y));
361 }
362 else if (trf.Angle != null)
363 {
364 currMatrix = Graphics.Multiply(currMatrix, Graphics.RotationMatrix(trf.Angle.Value));
365 }
366 else if (trf.Scale != null)
367 {
368 currMatrix = Graphics.Multiply(currMatrix, Graphics.ScaleMatrix(trf.Scale.Value.Width, trf.Scale.Value.Height));
369 }
370 else if (trf.Matrix != null)
371 {
372 currMatrix = Graphics.Multiply(currMatrix, trf.Matrix);
373 }
374 }
375 else if (referenceGraphics.Actions[i] is StateAction)
376 {
377 if (((StateAction)referenceGraphics.Actions[i]).StateActionType == StateAction.StateActionTypes.Save)
378 {
379 transformMatrix.Push(currMatrix);
380 }
381 else
382 {
383 currMatrix = transformMatrix.Pop();
384 }
385 }
386 }
387
388 currMatrix = Graphics.Invert(currMatrix);
389
390 Point p1 = Graphics.Multiply(currMatrix, this.StartPoint);
391 Point p2 = Graphics.Multiply(currMatrix, this.EndPoint);
392
393 return new LinearGradientBrush(p1, p2, this.GradientStops);
394 }

398 {
399 return new LinearGradientBrush(this.StartPoint, this.EndPoint, from el in this.GradientStops select el.MultiplyOpacity(opacity));

439 this.Centre = centre;
440 this.Radius = radius;
