18using System;
19using System.Diagnostics.Contracts;
20using System.Threading.Tasks;

22namespace VectSharp.Filters
23{
24 /// <summary>
25 /// Represents a filter that applies a matrix convolution to the image.
26 /// </summary>
27 public class ConvolutionFilter : ILocationInvariantFilter
28 {
29 /// <inheritdoc/>
30 public Point TopLeftMargin { get; protected set; }
31 /// <inheritdoc/>
32 public Point BottomRightMargin { get; protected set; }
33
34 /// <summary>
35 /// The kernel of the <see cref="ConvolutionFilter"/>. The dimensions of this matrix should all be odd numbers. The larger the kernel, the worse the performance.
36 /// </summary>
37 public virtual double[,] Kernel { get; protected set; }
38
39 /// <summary>
40 /// The normalisation value that is applies to the kernel.
41 /// </summary>
42 public virtual double Normalisation { get; protected set; } = 1;
43
44 /// <summary>
45 /// The bias value that is added to every colour component when the filter is applied.
46 /// </summary>
47 public virtual double Bias { get; protected set; } = 0;
48
49 /// <summary>
50 /// The scale relating the size of the kernel to graphics units.
51 /// </summary>
52 public virtual double Scale { get; protected set; }
53
54 /// <summary>
55 /// If this is <see langword="true"/>, the alpha value of the input pixels is preserved. Otherwise, the alpha channel is subject to the same convolution process as the other colour components.
56 /// </summary>
57 public virtual bool PreserveAlpha { get; protected set; } = true;
58
59 /// <summary>
60 /// Creates a new <see cref="ConvolutionFilter"/> with the specified parameters.
61 /// </summary>
62 /// <param name="kernel">The kernel of the <see cref="ConvolutionFilter"/>. The dimensions of this matrix should all be odd numbers. The larger the kernel, the worse the performance.</param>
63 /// <param name="scale">The scale relating the size of the kernel to graphics units.</param>
64 /// <param name="preserveAlpha">If this is <see langword="true"/>, the alpha value of the input pixels is preserved. Otherwise, the alpha channel is subject to the same convolution process as the other colour components.</param>
65 /// <param name="normalisation">The normalisation value that is applies to the kernel.</param>
66 /// <param name="bias">The bias value that is added to every colour component when the filter is applied.</param>
67 /// <exception cref="ArgumentException">This exception is thrown when the kernel dimensions are not odd numbers.</exception>
68 public ConvolutionFilter(double[,] kernel, double scale, bool preserveAlpha = true, double normalisation = 1, double bias = 0)
69 {
70 if (kernel.GetLength(0) % 2 != 1 || kernel.GetLength(1) % 2 != 1)
71 {
72 throw new ArgumentException("The kernel must have an odd number of rows and columns!", nameof(kernel));
73 }
74
75 this.Kernel = kernel;
76
77 int kernelWidth = (this.Kernel.GetLength(0) - 1) / 2;
78 int kernelHeight = (this.Kernel.GetLength(1) - 1) / 2;
79
80 this.TopLeftMargin = new Point(kernelWidth * scale, kernelHeight * scale);
81 this.BottomRightMargin = new Point(kernelWidth * scale, kernelHeight * scale);
82 this.Scale = scale;
83 this.PreserveAlpha = preserveAlpha;
84 this.Normalisation = normalisation;
85 this.Bias = bias;
86 }
87
88 /// <inheritdoc/>
89 [Pure]
90 public virtual RasterImage Filter(RasterImage image, double scale)
91 {
92 IntPtr tbrData = System.Runtime.InteropServices.Marshal.AllocHGlobal(image.Width * image.Height * (image.HasAlpha ? 4 : 3));
93
94 int width = image.Width;
95 int height = image.Height;
96
97 int kernelWidth = (this.Kernel.GetLength(0) - 1) / 2;
98 int kernelHeight = (this.Kernel.GetLength(1) - 1) / 2;
99
100 int actualKernelWidth = (int)Math.Round(kernelWidth * scale * this.Scale);
101 int actualKernelHeight = (int)Math.Round(kernelHeight * scale * this.Scale);
102
103 actualKernelWidth = Math.Max(actualKernelWidth, 1);
104 actualKernelHeight = Math.Max(actualKernelHeight, 1);
105
106 int[] kernelX = new int[actualKernelWidth * 2 + 1];
107
108 for (int x = 0; x < actualKernelWidth * 2 + 1; x++)
109 {
110 kernelX[x] = (int)Math.Round((double)(x - actualKernelWidth) / actualKernelWidth * kernelWidth + kernelWidth);
111 }
112
113 int[] kernelY = new int[actualKernelHeight * 2 + 1];
114
115 for (int y = 0; y < actualKernelHeight * 2 + 1; y++)
116 {
117 kernelY[y] = (int)Math.Round((double)(y - actualKernelHeight) / actualKernelHeight * kernelHeight + kernelHeight);
118 }
119
120 int[] countsX = new int[2 * kernelWidth + 1];
121
122 for (int i = 0; i < 2 * actualKernelWidth + 1; i++)
123 {
124 countsX[kernelX[i]]++;
125 }
126
127 int[] countsY = new int[2 * kernelHeight + 1];
128
129 for (int i = 0; i < 2 * actualKernelHeight + 1; i++)
130 {
131 countsY[kernelY[i]]++;
132 }
133
134
135 double[] weightsX = new double[2 * actualKernelWidth + 1];
136
137 for (int i = 0; i < 2 * actualKernelWidth + 1; i++)
138 {
139 weightsX[i] = 1.0 / countsX[kernelX[i]];
140 }
141
142 double[] weightsY = new double[2 * actualKernelHeight + 1];
143
144 for (int i = 0; i < 2 * actualKernelHeight + 1; i++)
145 {
146 weightsY[i] = 1.0 / countsY[kernelY[i]];
147 }
148
149 kernelWidth = actualKernelWidth;
150 kernelHeight = actualKernelHeight;
151
152 double normalisation = this.Normalisation;
153
154 if (double.IsNaN(normalisation) || normalisation == 0)
155 {
156 normalisation = 1;
157 }
158
159 double totalWeight = 0;
160 for (int i = 0; i < kernelWidth * 2 + 1; i++)
161 {
162 for (int j = 0; j < kernelHeight * 2 + 1; j++)
163 {
164 totalWeight += Kernel[kernelX[i], kernelY[j]] * weightsX[i] * weightsY[j];
165 }
166 }
167
168 if (Math.Abs(totalWeight) < 1e-5)
169 {
170 totalWeight = 1;
171 }
172
173
174 double bias = this.Bias * 255;
175 int pixelSize = image.HasAlpha ? 4 : 3;
176 int stride = image.Width * pixelSize;
177
178 int threads = Math.Min(8, Environment.ProcessorCount);
179
180 unsafe
181 {
182 byte* input = (byte*)image.ImageDataAddress;
183 byte* output = (byte*)tbrData;
184
185 Action<int> yLoop;
186
187 if (image.HasAlpha && !PreserveAlpha)
188 {
189 yLoop = (y) =>
190 {
191 for (int x = 0; x < width; x++)
192 {
193 double R = 0;
194 double G = 0;
195 double B = 0;
196
197 double weight = 0;
198
199 for (int targetX = 0; targetX <= kernelWidth * 2; targetX++)
200 {
201 for (int targetY = 0; targetY <= kernelHeight * 2; targetY++)
202 {
203 int tX = Math.Min(Math.Max(0, x + targetX - kernelWidth), width - 1);
204 int tY = Math.Min(Math.Max(0, y + targetY - kernelHeight), height - 1);
205
206 double a = input[tY * stride + tX * 4 + 3] / 255.0 * weightsX[targetX] * weightsY[targetY];
207
208 int projectedX = kernelX[targetX];
209 int projectedY = kernelY[targetY];
210
211 weight += Kernel[projectedX, projectedY] * a;
212
213 R += Kernel[projectedX, projectedY] * input[tY * stride + tX * 4] * a;
214 G += Kernel[projectedX, projectedY] * input[tY * stride + tX * 4 + 1] * a;
215 B += Kernel[projectedX, projectedY] * input[tY * stride + tX * 4 + 2] * a;
216 }
217 }
218
219 if (weight != 0)
220 {
221 output[y * stride + x * 4] = (byte)Math.Min(255, Math.Max(0, R / (normalisation * weight) + bias));
222 output[y * stride + x * 4 + 1] = (byte)Math.Min(255, Math.Max(0, G / (normalisation * weight) + bias));
223 output[y * stride + x * 4 + 2] = (byte)Math.Min(255, Math.Max(0, B / (normalisation * weight) + bias));
224 output[y * stride + x * 4 + 3] = (byte)Math.Min(255, Math.Max(0, (weight / (normalisation * totalWeight) * 255 + bias)));
225 }
226 else
227 {
228 output[y * stride + x * 4] = 0;
229 output[y * stride + x * 4 + 1] = 0;
230 output[y * stride + x * 4 + 2] = 0;
231 output[y * stride + x * 4 + 3] = 0;
232 }
233 }
234 };
235 }
236 else if (image.HasAlpha && PreserveAlpha)
237 {
238 yLoop = (y) =>
239 {
240 for (int x = 0; x < width; x++)
241 {
242 double R = 0;
243 double G = 0;
244 double B = 0;
245
246 for (int targetX = 0; targetX <= kernelWidth * 2; targetX++)
247 {
248 for (int targetY = 0; targetY <= kernelHeight * 2; targetY++)
249 {
250 int tX = Math.Min(Math.Max(0, x + targetX - kernelWidth), width - 1);
251 int tY = Math.Min(Math.Max(0, y + targetY - kernelHeight), height - 1);
252
253 int projectedX = kernelX[targetX];
254 int projectedY = kernelY[targetY];
255
256 R += Kernel[projectedX, projectedY] * input[tY * stride + tX * 4] * weightsX[targetX] * weightsY[targetY];
257 G += Kernel[projectedX, projectedY] * input[tY * stride + tX * 4 + 1] * weightsX[targetX] * weightsY[targetY];
258 B += Kernel[projectedX, projectedY] * input[tY * stride + tX * 4 + 2] * weightsX[targetX] * weightsY[targetY];
259 }
260 }
261
262 output[y * stride + x * 4] = (byte)Math.Min(255, Math.Max(0, R / (normalisation * totalWeight) + bias));
263 output[y * stride + x * 4 + 1] = (byte)Math.Min(255, Math.Max(0, G / (normalisation * totalWeight) + bias));
264 output[y * stride + x * 4 + 2] = (byte)Math.Min(255, Math.Max(0, B / (normalisation * totalWeight) + bias));
265 output[y * stride + x * 4 + 3] = input[y * stride + x * 4 + 3];
266 }
267 };
268 }
269 else
270 {
271 yLoop = (y) =>
272 {
273 for (int x = 0; x < width; x++)
274 {
275 double R = 0;
276 double G = 0;
277 double B = 0;
278
279 for (int targetX = 0; targetX <= kernelWidth * 2; targetX++)
280 {
281 for (int targetY = 0; targetY <= kernelHeight * 2; targetY++)
282 {
283 int tX = Math.Min(Math.Max(0, x + targetX - kernelWidth), width - 1);
284 int tY = Math.Min(Math.Max(0, y + targetY - kernelHeight), height - 1);
285
286 int projectedX = kernelX[targetX];
287 int projectedY = kernelY[targetY];
288
289 R += Kernel[projectedX, projectedY] * input[tY * stride + tX * 3] * weightsX[targetX] * weightsY[targetY];
290 G += Kernel[projectedX, projectedY] * input[tY * stride + tX * 3 + 1] * weightsX[targetX] * weightsY[targetY];
291 B += Kernel[projectedX, projectedY] * input[tY * stride + tX * 3 + 2] * weightsX[targetX] * weightsY[targetY];
292 }
293 }
294
295 output[y * stride + x * 3] = (byte)Math.Min(255, Math.Max(0, R / (normalisation * totalWeight) + bias));
296 output[y * stride + x * 3 + 1] = (byte)Math.Min(255, Math.Max(0, G / (normalisation * totalWeight) + bias));
297 output[y * stride + x * 3 + 2] = (byte)Math.Min(255, Math.Max(0, B / (normalisation * totalWeight) + bias));
298 }
299 };
300 }
301
302 if (threads == 1)
303 {
304 for (int y = 0; y < height; y++)
305 {
306 yLoop(y);
307 }
308 }
309 else
310 {
311 ParallelOptions options = new ParallelOptions() { MaxDegreeOfParallelism = threads };
312
313 Parallel.For(0, height, options, yLoop);
314 }
315 }
316
317 return new RasterImage(tbrData, image.Width, image.Height, image.HasAlpha, image.Interpolate);
318 }
319 }
