110 /// <inheritdoc/>
111 public void Plot(Graphics target)
112 {
113 double total = Data.Sum();
114
115 double currAngle = StartAngle;
116
117 double minAngle = Data.Where(x => x > 0).Min() / total * 2 * Math.PI;
118
119 minAngle = Math.Min(minAngle, 1);
120 minAngle = Math.Max(minAngle, 0.1);
121
122 for (int i = 0; i < Data.Count; i++)
123 {
124 double endAngle = currAngle + Data[i] / total * 2 * Math.PI;
125
126 GraphicsPath pth = new GraphicsPath();
127
128 if (InnerRadius[0] != 0 || InnerRadius[1] != 0)
129 {
130 Point[] innerCircle = new Point[(int)((endAngle - currAngle) / minAngle * 10) + 1];
131
132 Point[] outerCircle = new Point[(int)((endAngle - currAngle) / minAngle * 10) + 1];
133
134 double step = (endAngle - currAngle) / (innerCircle.Length - 1);
135
136 for (int j = 0; j < innerCircle.Length; j++)
137 {
138 double currAng = currAngle + step * j;
139
140 if (Clockwise)
141 {
142 currAng *= -1;
143 }
144
145 innerCircle[j] = CoordinateSystem.ToPlotCoordinates(new double[] { Centre[0] + InnerRadius[0] * Math.Cos(currAng), Centre[1] + InnerRadius[1] * Math.Sin(currAng) });
146 outerCircle[outerCircle.Length - 1 - j] = CoordinateSystem.ToPlotCoordinates(new double[] { Centre[0] + OuterRadius[0] * Math.Cos(currAng), Centre[1] + OuterRadius[1] * Math.Sin(currAng) });
147 }
148
149 currAngle = endAngle;
150
151 pth.AddSmoothSpline(innerCircle);
152
153 pth.LineTo(outerCircle[0]);
154 pth.AddSmoothSpline(outerCircle);
155
156 pth.Close();
157 }
158 else
159 {
160 Point[] outerCircle = new Point[(int)((endAngle - currAngle) / minAngle * 10) + 1];
161
162 double step = (endAngle - currAngle) / (outerCircle.Length - 1);
163
164 for (int j = 0; j < outerCircle.Length; j++)
165 {
166 double currAng = currAngle + step * j;
167
168 if (Clockwise)
169 {
170 currAng *= -1;
171 }
172
173 outerCircle[outerCircle.Length - 1 - j] = CoordinateSystem.ToPlotCoordinates(new double[] { Centre[0] + OuterRadius[0] * Math.Cos(currAng), Centre[1] + OuterRadius[1] * Math.Sin(currAng) });
174 }
175
176 currAngle = endAngle;
177
178 pth.MoveTo(CoordinateSystem.ToPlotCoordinates(Centre));
179
180 pth.LineTo(outerCircle[0]);
181 pth.AddSmoothSpline(outerCircle);
182
183 pth.Close();
184 }
185
186 PlotElementPresentationAttributes attributes = PresentationAttributes[i % PresentationAttributes.Count];
187
188 string tag = Tag;
189 string strokeTag = tag;
190
191 if (target.UseUniqueTags && !string.IsNullOrEmpty(tag))
192 {
193 tag += "@" + i.ToString();
194 strokeTag += "@" + i.ToString() + "_stroke";
195 }
196
197 if (attributes.Fill != null)
198 {
199 target.FillPath(pth, attributes.Fill, tag);
200 }
201
202 if (attributes.Stroke != null)
203 {
204 target.StrokePath(pth, attributes.Stroke, attributes.LineWidth, attributes.LineCap, attributes.LineJoin, attributes.LineDash, strokeTag);
205 }
206 }
207 }
208 }
