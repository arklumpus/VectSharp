238 /// <summary>
239 /// Creates a link destination that sets the state of one or more <see cref="OptionalContentGroup"/>s.
240 /// </summary>
241 /// <param name="on"><see cref="OptionalContentGroup"/>s to turn on.</param>
242 /// <param name="off"><see cref="OptionalContentGroup"/>s to turn off.</param>
243 /// <param name="toggle"><see cref="OptionalContentGroup"/>s to toggle.</param>
244 /// <returns>A string representation of an action that sets the specified optional content group states. Use this within the <c>linkDestionations</c> parameter of the <see cref="PDFContextInterpreter.CreatePDFDocument"/> method.</returns>
245 /// <exception cref="ArgumentException">Thrown if no <see cref="OptionalContentGroup"/>s are provided in any of the arguments.</exception>
246 public static string CreateSetOCGLink(IReadOnlyList<OptionalContentGroup> on = null, IReadOnlyList<OptionalContentGroup> off = null, IReadOnlyList<OptionalContentGroup> toggle = null)
247 {
248 StringBuilder tbr = new StringBuilder("#@OCG:");
249
250 bool any = false;
251
252 if (on != null)
253 {
254 for (int i = 0; i < on.Count; i++)
255 {
256 any = true;
257 tbr.Append(on[i].ToString());
258 tbr.Append(",on");
259
260 if (i < on.Count - 1 || off?.Count > 0 || toggle?.Count > 0)
261 {
262 tbr.Append(";");
263 }
264 }
265 }
266
267 if (off != null)
268 {
269 for (int i = 0; i < off.Count; i++)
270 {
271 any = true;
272 tbr.Append(off[i].ToString());
273 tbr.Append(",off");
274
275 if (i < off.Count - 1 || toggle?.Count > 0)
276 {
277 tbr.Append(";");
278 }
279 }
280 }
281
282 if (toggle != null)
283 {
284 for (int i = 0; i < toggle.Count; i++)
285 {
286 any = true;
287 tbr.Append(toggle[i].ToString());
288 tbr.Append(",toggle");
289
290 if (i < toggle.Count - 1)
291 {
292 tbr.Append(";");
293 }
294 }
295 }
296
297 if (!any)
298 {
299 throw new ArgumentException("No optional content groups provided!");
300 }
301
302 return tbr.ToString();
303 }
