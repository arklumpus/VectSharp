93 /// <summary>
94 /// Create a new <see cref="TextFragment"/>.
95 /// </summary>
96 /// <param name="position">The position of the text.</param>
97 /// <param name="text">The text contained in the fragment.</param>
98 /// <param name="font">The font used to draw the text.</param>
99 /// <param name="colour">The colour used to draw the text.</param>
100 /// <param name="tag">Graphics action tag for the text.</param>
101 public TextFragment(Point position, string text, Font font, Colour colour, string tag)
102 {
103 this.Position = position;
104 this.Text = text;
105 this.Font = font;
106 this.Colour = colour;
107 this.Tag = tag;
108 }
109
110 /// <inheritdoc/>
111 public override void Translate(double deltaX, double deltaY)
112 {
113 this.Position = new Point(this.Position.X + deltaX, this.Position.Y + deltaY);
114 }
115
116 /// <inheritdoc/>
117 public override double GetAscent(double lineAscent)
118 {
119 return this.Font.Ascent;
120 }
121
122 /// <inheritdoc/>
123 public override double GetMaxY()
124 {
125 return this.Position.Y - this.Font.Descent;
126 }
127
128 /// <inheritdoc/>
129 public override double GetMaxX()
130 {
131 Size size = this.Font.MeasureText(this.Text);
132 return size.Width + this.Position.X;
133 }
134
135 /// <inheritdoc/>
136 public override Point? Render(Graphics graphics, double deltaY)
137 {
138 Size size = this.Font.MeasureText(this.Text);
139 graphics.FillText(this.Position.X, this.Position.Y + deltaY, this.Text, this.Font, this.Colour, TextBaselines.Baseline, tag: this.Tag);
140 return new Point(size.Width + this.Position.X, this.Position.Y + size.Height + deltaY);
141 }

351 /// <summary>
352 /// Represents a text line.
353 /// </summary>
354 public class Line
355 {
356 /// <summary>
357 /// The fragments contained in the line.
358 /// </summary>
359 public List<LineFragment> Fragments { get; }
360
361 /// <summary>
362 /// The initial ascent for text on the <see cref="Line"/>.
363 /// </summary>
364 public double InitialAscent { get; }
365
366 /// <summary>
367 /// Create a new <see cref="Line"/>.
368 /// </summary>
369 /// <param name="initialAscent">The initial ascent for text on the <see cref="Line"/>.</param>
370 public Line(double initialAscent)
371 {
372 this.InitialAscent = initialAscent;
373 this.Fragments = new List<LineFragment>();
374 }
375
376 internal void Render(ref Graphics graphics, ref MarkdownContext context, MarkdownRenderer.NewPageAction newPageAction, double pageMaxY, double pageMaxX, MarkdownRenderer renderer)
377 {
378 double deltaY = 0;
379 double maxY = 0;
380 double maxX = 0;
381
382 foreach (LineFragment fragment in this.Fragments)
383 {
384 deltaY = Math.Max(deltaY, fragment.GetAscent(this.InitialAscent) - InitialAscent);
385 maxY = Math.Max(maxY, fragment.GetMaxY());
386 maxX = Math.Max(maxX, fragment.GetMaxX());
387 }
388
389 maxY += deltaY;
390
391 double minY = context.Cursor.Y -(this.InitialAscent + deltaY);
392
393 Line line = this;
394
395 renderer.RaiseLineMeasured(ref context, ref graphics, ref line, pageMaxX, pageMaxY, maxX, minY, maxY, context.Cursor.Y);
396
397 if (line != null && line.Fragments.Count > 0)
398 {
399 deltaY = 0;
400 maxY = 0;
401 maxX = 0;
402
403 foreach (LineFragment fragment in line.Fragments)
404 {
405 deltaY = Math.Max(deltaY, fragment.GetAscent(line.InitialAscent) - InitialAscent);
406 maxY = Math.Max(maxY, fragment.GetMaxY());
407 maxX = Math.Max(maxX, fragment.GetMaxX());
408 }
409
410 maxY += deltaY;
411
412 if (maxY > pageMaxY)
413 {
414 double currCursY = context.Cursor.Y;
415
416 newPageAction(ref context, ref graphics);
417
418 double currDelta = deltaY;
419
420 context.Cursor = new Point(context.Cursor.X, context.Cursor.Y + line.InitialAscent + currDelta);
421
422 deltaY -= currCursY - context.Cursor.Y + currDelta;
423
424 context.Cursor = new Point(context.Cursor.X, context.Cursor.Y);
425 }
426 else
427 {
428 context.Cursor = new Point(context.Cursor.X, context.Cursor.Y + deltaY);
429 }
430
431 renderer.RaiseLineRendering(ref context, ref graphics, line, pageMaxX, pageMaxY, maxX, minY, maxY, context.Cursor.Y);
432
433 for (int i = 0; i < line.Fragments.Count; i++)
434 {
435 Point? pt = line.Fragments[i].Render(graphics, deltaY);
436
437 if (pt != null)
438 {
439 context.BottomRight = new Point(Math.Max(context.BottomRight.X, pt.Value.X + context.Translation.X), Math.Max(context.BottomRight.Y, pt.Value.Y + context.Translation.Y));
440 }
441 }
442
443 renderer.RaiseLineRendered(ref context, ref graphics, line, pageMaxX, pageMaxY, maxX, minY, maxY, context.Cursor.Y);
444 }
445 }
446 }
