93 /// <summary>
94 /// Checks whether this <see cref="Point"/> is equal to another <see cref="Point"/>, up to a specified tolerance.
95 /// </summary>
96 /// <param name="p2">The <see cref="Point"/> to compare.</param>
97 /// <param name="tolerance">The tolerance threshold.</param>
98 /// <returns><see langword="true"/> if both coordinates of the <see cref="Point"/>s are closer than <paramref name="tolerance"/> or if their relative difference (i.e. <c>(a - b) / (a + b) * 2</c>) is smaller than <paramref name="tolerance"/>. <see langword="false"/> otherwise.</returns>
99 public bool IsEqual(Point p2, double tolerance)
100 {
101 return (Math.Abs(p2.X - this.X) <= tolerance || Math.Abs((p2.X - this.X) / (p2.X + this.X)) <= tolerance * 0.5) && (Math.Abs(p2.Y - this.Y) <= tolerance || Math.Abs((p2.Y - this.Y) / (p2.Y + this.Y)) <= tolerance * 0.5);
102 }

104 /// <summary>
105 /// Computes the top-left corner of the <see cref="Rectangle"/> identified by two <see cref="Point"/>s.

126 /// <summary>
127 /// Computes the smallest <see cref="Rectangle"/> that contains all the specified points.
128 /// </summary>
129 /// <param name="points">The points whose bounds are being computed.</param>
130 /// <returns>The smallest <see cref="Rectangle"/> that contains all the specified points.</returns>
131 public static Rectangle Bounds(IEnumerable<Point> points)
132 {
133 bool initialised = false;
134 Point min = new Point(double.NaN, double.NaN);
135 Point max = new Point(double.NaN, double.NaN);
136
137 foreach (Point pt in points)
138 {
139 if (!initialised)
140 {
141 min = pt;
142 max = pt;
143 initialised = true;
144 }
145 else
146 {
147 min = Point.Min(min, pt);
148 max = Point.Max(max, pt);
149 }
150 }
151
152 return new Rectangle(min, max);
153 }

159 /// <returns>The smallest <see cref="Rectangle"/> that contains all the specified points.</returns>
160 public static Rectangle Bounds(params Point[] points)
161 {
162 return Bounds((IEnumerable<Point>)points);

361 /// <summary>
362 /// Computes the rectangular bounds of the union of two <see cref="Rectangle"/>s.
363 /// </summary>
364 /// <param name="rectangle1">The first <see cref="Rectangle"/>.</param>
365 /// <param name="rectangle2">The second <see cref="Rectangle"/>.</param>
366 /// <returns>The smallest <see cref="Rectangle"/> containing both <paramref name="rectangle1"/> and <paramref name="rectangle2"/>.</returns>
367 public static Rectangle Union(Rectangle rectangle1, Rectangle rectangle2)
368 {
369 double minX = rectangle1.Location.X;
370 double minY = rectangle1.Location.Y;
371 double maxX = rectangle1.Location.X;
372 double maxY = rectangle1.Location.Y;
373
374 minX = Math.Min(minX, rectangle1.Location.X + rectangle1.Size.Width);
375 minX = Math.Min(minX, rectangle2.Location.X);
376 minX = Math.Min(minX, rectangle2.Location.X + rectangle2.Size.Width);
377
378 minY = Math.Min(minY, rectangle1.Location.Y + rectangle1.Size.Height);
379 minY = Math.Min(minY, rectangle2.Location.Y);
380 minY = Math.Min(minY, rectangle2.Location.Y + rectangle2.Size.Height);
381
382
383 maxX = Math.Max(maxX, rectangle1.Location.X + rectangle1.Size.Width);
384 maxX = Math.Max(maxX, rectangle2.Location.X);
385 maxX = Math.Max(maxX, rectangle2.Location.X + rectangle2.Size.Width);
386
387 maxY = Math.Max(maxY, rectangle1.Location.Y + rectangle1.Size.Height);
388 maxY = Math.Max(maxY, rectangle2.Location.Y);
389 maxY = Math.Max(maxY, rectangle2.Location.Y + rectangle2.Size.Height);
390
391 return new Rectangle(minX, minY, maxX - minX, maxY - minY);
392 }

394 /// <summary>
395 /// Computes the intersection of two <see cref="Rectangle"/>s.
396 /// </summary>
397 /// <param name="rectangle1">The first <see cref="Rectangle"/>.</param>
398 /// <param name="rectangle2">The second <see cref="Rectangle"/>.</param>
399 /// <returns>The rectangle corresponding to the intersection of <paramref name="rectangle1"/> and <paramref name="rectangle2"/>, or <see cref="Rectangle.NaN"/> if the intersection is empty.</returns>
400 public static Rectangle Intersection(Rectangle rectangle1, Rectangle rectangle2)
401 {
402 double x0 = Math.Max(rectangle1.Location.X, rectangle2.Location.X);
403 double x1 = Math.Min(rectangle1.Location.X + rectangle1.Size.Width, rectangle2.Location.X + rectangle2.Size.Width);
404
405 double y0 = Math.Max(rectangle1.Location.Y, rectangle2.Location.Y);
406 double y1 = Math.Min(rectangle1.Location.Y + rectangle1.Size.Height, rectangle2.Location.Y + rectangle2.Size.Height);
407
408 if (x1 >= x0 && y1 >= y0)
409 {
410 return new Rectangle(x0, y0, x1 - x0, y1 - y0);
411 }
412 else
413 {
414 return Rectangle.NaN;
415 }
416
417 }

419 /// <summary>
420 /// Computes the rectangular bounds of the union of multiple <see cref="Rectangle"/>s.
421 /// </summary>
422 /// <param name="rectangles">The <see cref="Rectangle"/>s whose union will be computed.</param>
423 /// <returns>The smallest <see cref="Rectangle"/> containing all the <paramref name="rectangles"/>.</returns>
424 public static Rectangle Union(IEnumerable<Rectangle> rectangles)
425 {
426 if (rectangles.Any())
427 {
428 bool initialised = false;
429
430 Rectangle tbr = new Rectangle();
431
432 foreach (Rectangle rect in rectangles)
433 {
434 if (!initialised)
435 {
436 tbr = rect;
437 }
438 else
439 {
440 tbr = Union(rect, tbr);
441 }
442 }
443
444 return tbr;
445 }
446 else
447 {
448 return Rectangle.NaN;
449 }
450 }

456 /// <returns>The smallest <see cref="Rectangle"/> containing all the <paramref name="rectangles"/>.</returns>
457 public static Rectangle Union(params Rectangle[] rectangles)
458 {
459 return Union((IEnumerable<Rectangle>)rectangles);
