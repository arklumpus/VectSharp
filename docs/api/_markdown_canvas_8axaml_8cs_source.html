142 /// <inheritdoc/>
143 protected override void OnPropertyChanged(AvaloniaPropertyChangedEventArgs change)
144 {
145 if (change.Property != ContentProperty || !initialized)
146 {
147 base.OnPropertyChanged(change);
148
149 if (this.initialized)
150 {
151 if (change.Property == MarkdownCanvasControl.BoundsProperty || change.Property == MarkdownCanvasControl.MinRenderWidthProperty || change.Property == MarkdownCanvasControl.MaxRenderWidthProperty || change.Property == MarkdownCanvasControl.MinVariationProperty)
152 {
153 Render();
154 }
155 else if (change.Property == MarkdownCanvasControl.DocumentProperty)
156 {
157 forcedRerender = true;
158 Render();
159 }
160 else if (change.Property == MarkdownCanvasControl.FontSizeProperty)
161 {
162 this.Renderer.BaseFontSize = this.FontSize;
163 forcedRerender = true;
164 Render();
165 }
166 else if (change.Property == MarkdownCanvasControl.DocumentSourceProperty)
167 {
168 MarkdownDocument document = Markdig.Markdown.Parse(change.GetNewValue<string>(), Renderer.MarkdownPipelineBuilder.Build());
169 this.Document = document;
170 }
171 else if (change.Property == MarkdownCanvasControl.VerticalContentAlignmentProperty)
172 {
173 this.FindControl<ScrollViewer>("ScrollViewer").VerticalContentAlignment = this.VerticalContentAlignment;
174 }
175 else if (change.Property == MarkdownCanvasControl.HorizontalContentAlignmentProperty)
176 {
177 this.FindControl<ScrollViewer>("ScrollViewer").HorizontalContentAlignment = this.HorizontalContentAlignment;
178 }
179 }
180 }
181 else
182 {
183 initialized = false;
184 this.Content = change.OldValue;
185 initialized = true;
186 }

191 private void Render()
192 {
193 if (Document != null)
194 {
195 double width = Math.Min(MaxRenderWidth, Math.Max(MinRenderWidth, this.Bounds.Width - MinVariation - 13));
196
197 if (forcedRerender || double.IsNaN(lastRenderedWidth) || width != lastRenderedWidth && width < lastRenderedWidth - MinVariation || width > lastRenderedWidth + MinVariation)
198 {
199 Page pag;
200 Dictionary<string, string> linkDestinations;
201
202 try
203 {
204 pag = Renderer.RenderSinglePage(this.Document, width, out linkDestinations, out _);
205 }
206 catch
207 {
208 pag = new Page(width, 0);
209 linkDestinations = new Dictionary<string, string>();
210 }
211
212 Dictionary<string, Delegate> taggedActions = new Dictionary<string, Delegate>();
213 Dictionary<string, Avalonia.Point> linkDestinationPoints = new Dictionary<string, Avalonia.Point>();
214
215 foreach (KeyValuePair<string, string> linkDestination in linkDestinations)
216 {
217 string url = linkDestination.Value;
218
219 taggedActions.Add(linkDestination.Key, (Func<RenderAction, IEnumerable<RenderAction>>)(act =>
220 {
221 act.PointerEntered += (s, e) =>
222 {
223 act.Parent.Cursor = new Avalonia.Input.Cursor(Avalonia.Input.StandardCursorType.Hand);
224 };
225
226 act.PointerExited += (s, e) =>
227 {
228 act.Parent.Cursor = new Avalonia.Input.Cursor(Avalonia.Input.StandardCursorType.Arrow);
229 };
230
231 act.PointerPressed += (s, e) =>
232 {
233 if (url.StartsWith("#"))
234 {
235 if (linkDestinationPoints.TryGetValue(url.Substring(1), out Avalonia.Point target))
236 {
237 ScrollViewer scrollViewer = this.FindControl<ScrollViewer>("ScrollViewer");
238
239 scrollViewer.Offset = new Vector(Math.Max(Math.Min(scrollViewer.Offset.X, target.X), target.X - scrollViewer.Viewport.Width), target.Y);
240 }
241 }
242 else
243 {
244 System.Diagnostics.Process.Start(new System.Diagnostics.ProcessStartInfo() { FileName = url, UseShellExecute = true });
245 }
246 };
247
248 if (act.ActionType == RenderAction.ActionTypes.Path)
249 {
250 linkDestinationPoints[linkDestination.Key] = act.Geometry.Bounds.TopLeft.Transform(act.Transform);
251 }
252 else if (act.ActionType == RenderAction.ActionTypes.Text)
253 {
254 linkDestinationPoints[linkDestination.Key] = new Avalonia.Point(0, 0).Transform(act.Transform);
255 }
256 else if (act.ActionType == RenderAction.ActionTypes.RasterImage)
257 {
258 linkDestinationPoints[linkDestination.Key] = act.ImageDestination.Value.TopLeft.Transform(act.Transform);
259 }
260
261 return new RenderAction[] { act };
262 }));
263
264 if (url.StartsWith("#"))
265 {
266 if (!taggedActions.ContainsKey(url.Substring(1)))
267 {
268 taggedActions.Add(url.Substring(1), (Func<RenderAction, IEnumerable<RenderAction>>)(act =>
269 {
270 if (act.ActionType == RenderAction.ActionTypes.Path)
271 {
272 linkDestinationPoints[url.Substring(1)] = act.Geometry.Bounds.TopLeft.Transform(act.Transform);
273 }
274 else if (act.ActionType == RenderAction.ActionTypes.Text)
275 {
276 linkDestinationPoints[url.Substring(1)] = new Avalonia.Point(0, 0).Transform(act.Transform);
277 }
278 else if (act.ActionType == RenderAction.ActionTypes.RasterImage)
279 {
280 linkDestinationPoints[url.Substring(1)] = act.ImageDestination.Value.TopLeft.Transform(act.Transform);
281 }
282
283 return new RenderAction[] { act };
284 }));
285 }
286 }
287 }
288
289 Control can = pag.PaintToCanvas(false, taggedActions, false, this.TextConversionOption);
290
291 this.FindControl<ScrollViewer>("ScrollViewer").Content = can;
292 this.FindControl<ScrollViewer>("ScrollViewer").Padding = new Thickness(0, 0, 0, 0);
293 lastRenderedWidth = width;
294 forcedRerender = false;
295 }
296 else
297 {
298 this.FindControl<ScrollViewer>("ScrollViewer").Padding = new Thickness(0, 0, width - lastRenderedWidth, 0);
299 }
300 }
