94 /// <summary>
95 /// Translate and resize the <see cref="Page"/> so that it displays the rectangle defined by <paramref name="topLeft"/> and <paramref name="size"/>.
96 /// </summary>
97 /// <param name="topLeft">The top left corner of the area to include in the page.</param>
98 /// <param name="size">The size of the area to include in the page.</param>
99 /// <param name="removeClippedGraphics">If this is <see langword="true"/>, graphics actions that fall outside of the specified region are completely removed from the plot, otherwise they are just hidden.</param>
100 /// <param name="tag">A tag to identify the transform.</param>
101 public void Crop(Point topLeft, Size size, bool removeClippedGraphics = false, string tag = null)
102 {
103 if (removeClippedGraphics)
104 {
105 this.Graphics.Crop(new Rectangle(topLeft, size));
106 }
107
108 if (this.Graphics.Actions[0] is TransformAction transf)
109 {
110 double[,] currMatrix = transf.GetMatrix();
111
112 double[,] newMatrix = Graphics.Multiply(Graphics.TranslationMatrix(-topLeft.X, -topLeft.Y), currMatrix);
113
114 this.Graphics.Actions[0] = new TransformAction(newMatrix, tag);
115 }
116 else
117 {
118 this.Graphics.Actions.Insert(0, new TransformAction(new Point(-topLeft.X, -topLeft.Y), tag));
119 }
120
121 this.Width = size.Width;
122 this.Height = size.Height;
123 }
124
125 /// <summary>
126 /// Translate and resize the <see cref="Page"/> so that it displays the rectangle corresponding to the bounding box of its contents.
127 /// </summary>
128 /// <param name="tag">A tag to identify the transform.</param>
129 public void Crop(string tag = null)
130 {
131 Rectangle bounds = this.Graphics.GetBounds();
132
133 if (this.Graphics.Actions[0] is TransformAction transf)
134 {
135 double[,] currMatrix = transf.GetMatrix();
136
137 double[,] newMatrix = Graphics.Multiply(Graphics.TranslationMatrix(-bounds.Location.X, -bounds.Location.Y), currMatrix);
138
139 this.Graphics.Actions[0] = new TransformAction(newMatrix, tag);
140 }
141 else
142 {
143 this.Graphics.Actions.Insert(0, new TransformAction(new Point(-bounds.Location.X, -bounds.Location.Y), tag));
144 }
145
146 this.Width = bounds.Size.Width;
147 this.Height = bounds.Size.Height;
148 }
149 }
