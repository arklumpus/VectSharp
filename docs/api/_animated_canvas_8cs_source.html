34 {
35 AnimatedCanvas owner = (AnimatedCanvas)ownerObject;
36

39 if (val < owner.maxFrameIndex)
40 {

162 internal AnimatedCanvas(VectSharp.Animation animation, double durationScaling, double frameRate, AvaloniaContextInterpreter.TextOptions textOption, FilterOption filterOption)
163 {
164 Colour backgroundColour = animation.Background;
165
166 this.BackgroundBrush = new SolidColorBrush(Color.FromArgb((byte)(backgroundColour.A * 255), (byte)(backgroundColour.R * 255), (byte)(backgroundColour.G * 255), (byte)(backgroundColour.B * 255)));
167
168 this.Width = animation.Width;
169 this.Height = animation.Height;
170 this.Images = new Dictionary<string, (IImage, bool)>();
171
172 int frames = (int)Math.Ceiling(animation.Duration * frameRate * durationScaling / 1000);
173
174 this.RenderActions = new List<RenderAction>[frames];
175
176 for (int i = 0; i < frames; i++)
177 {
178 double frameTime = i / frameRate / durationScaling * 1000;
179
180 Page pag = animation.GetFrameAtAbsolute(frameTime);
181
182 AvaloniaDrawingContext ctx = new AvaloniaDrawingContext(this.Width, this.Height, false, textOption, this.Images, filterOption);
183
184 pag.Graphics.CopyToIGraphicsContext(ctx);
185 this.RenderActions[i] = ctx.RenderActions;
186 }
187
188 if (animation.RepeatCount > 0)
189 {
190 this.maxFrameIndex = animation.RepeatCount * frames;
191 }
192 else
193 {
194 this.maxFrameIndex = 0;
195 }
196
197 this.FrameCount = frames;
198
199 this.FrameRate = frameRate;
200 }

202 /// <inheritdoc/>
203 public override void Render(DrawingContext context)
204 {
205 context.FillRectangle(this.BackgroundBrush, new Avalonia.Rect(0, 0, Width, Height));
206
207 foreach (RenderAction act in this.RenderActions[this.CurrentFrame % this.RenderActions.Length])
208 {
209 if (act.ActionType == RenderAction.ActionTypes.Path)
210 {
211 DrawingContext.PushedState? state = null;
212
213 if (act.ClippingPath != null)
214 {
215 //Random draw operation needed due to https://github.com/AvaloniaUI/Avalonia/issues/4408
216 context.DrawGeometry(null, new Pen(Brushes.Transparent), act.ClippingPath);
217 state = context.PushGeometryClip(act.ClippingPath);
218 }
219
220 using (context.PushTransform(act.Transform))
221 {
222 context.DrawGeometry(act.Fill, act.Stroke, act.Geometry);
223 }
224
225 if (state != null)
226 {
227 state?.Dispose();
228 //Random draw operation needed due to https://github.com/AvaloniaUI/Avalonia/issues/4408
229 context.DrawGeometry(null, new Pen(Brushes.Transparent), act.ClippingPath);
230 }
231 }
232 else if (act.ActionType == RenderAction.ActionTypes.Text)
233 {
234 DrawingContext.PushedState? state = null;
235
236 if (act.ClippingPath != null)
237 {
238 //Random draw operation needed due to https://github.com/AvaloniaUI/Avalonia/issues/4408
239 context.DrawGeometry(null, new Pen(Brushes.Transparent), act.ClippingPath);
240 state = context.PushGeometryClip(act.ClippingPath);
241 }
242
243 using (context.PushTransform(act.Transform))
244 {
245 context.DrawText(act.Text, Origin);
246 }
247
248 if (state != null)
249 {
250 state?.Dispose();
251 //Random draw operation needed due to https://github.com/AvaloniaUI/Avalonia/issues/4408
252 context.DrawGeometry(null, new Pen(Brushes.Transparent), act.ClippingPath);
253 }
254 }
255 else if (act.ActionType == RenderAction.ActionTypes.RasterImage)
256 {
257 DrawingContext.PushedState? state = null;
258
259 if (act.ClippingPath != null)
260 {
261 //Random draw operation needed due to https://github.com/AvaloniaUI/Avalonia/issues/4408
262 context.DrawGeometry(null, new Pen(Brushes.Transparent), act.ClippingPath);
263 state = context.PushGeometryClip(act.ClippingPath);
264 }
265
266 (IImage, bool) image = Images[act.ImageId];
267
268 using (context.PushTransform(act.Transform))
269 {
270 RenderOptions.SetBitmapInterpolationMode(this, image.Item2 ? Avalonia.Media.Imaging.BitmapInterpolationMode.HighQuality : Avalonia.Media.Imaging.BitmapInterpolationMode.None);
271 context.DrawImage(image.Item1, act.ImageSource.Value, act.ImageDestination.Value);
272 }
273
274 if (state != null)
275 {
276 state?.Dispose();
277 //Random draw operation needed due to https://github.com/AvaloniaUI/Avalonia/issues/4408
278 context.DrawGeometry(null, new Pen(Brushes.Transparent), act.ClippingPath);
279 }
280 }
281 }
