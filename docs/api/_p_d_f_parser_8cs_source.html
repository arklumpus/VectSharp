31 private static Page FromMuPDFDocument(MuPDFContext context, MuPDFDocument document, int pageNumber, SVGCreationOptions.TextOption textOption, bool includeAnnotations, Dictionary<string, MuPDFLinkDestination> linkDestinations)
32 {
33 string tempFile = Path.GetTempFileName();
34
35 try
36 {
37 MuPDFDocument.Create.SVGDocument(context, tempFile, document.Pages[pageNumber], new SVGCreationOptions() { TextRendering = textOption, IncludeAnnotations = includeAnnotations });
38 Parser.ParseImageURI = ImageURIParser.Parser(Parser.ParseSVGURI);
39
40 Page parsedPage = Parser.FromFile(tempFile);
41
42 if (linkDestinations != null)
43 {
44 foreach (MuPDFLink link in document.Pages[pageNumber].Links)
45 {
46 if (link.IsVisible)
47 {
48 string tag = Guid.NewGuid().ToString("N");
49 parsedPage.Graphics.FillRectangle(link.ActiveArea.X0, link.ActiveArea.Y0, link.ActiveArea.Width, link.ActiveArea.Height, Colour.FromRgba(0, 0, 0, 0), tag: tag);
50 linkDestinations.Add(tag, link.Destination);
51 }
52 }
53 }
54
55 return parsedPage;
56 }
57 finally
58 {
59 File.Delete(tempFile);
60 }
61 }

63 private static Dictionary<string, string> SetUpLinks(Page page, int pageNumber, Dictionary<string, MuPDFLinkDestination> linkDestinations)
64 {
65 Dictionary<string, string> tbr = new Dictionary<string, string>();
66
67 foreach (KeyValuePair<string, MuPDFLinkDestination> link in linkDestinations)
68 {
69 if (link.Value.Type == MuPDFLinkDestination.DestinationType.External)
70 {
71 tbr.Add(link.Key, (link.Value as MuPDFExternalLinkDestination).Uri);
72 }
73 else if (link.Value.Type == MuPDFLinkDestination.DestinationType.Internal)
74 {
75 MuPDFInternalLinkDestination intern = link.Value as MuPDFInternalLinkDestination;
76 if (intern.PageNumber == pageNumber)
77 {
78 string tag = Guid.NewGuid().ToString("N");
79
80 switch (intern.InternalType)
81 {
82 case MuPDFInternalLinkDestination.InternalDestinationType.XYZoom:
83 page.Graphics.FillRectangle(intern.X, intern.Y, 1, 1, Colour.FromRgba(0, 0, 0, 0), tag);
84 break;
85 case MuPDFInternalLinkDestination.InternalDestinationType.FitRectangle:
86 page.Graphics.FillRectangle(intern.X, intern.Y, intern.Width, intern.Height, Colour.FromRgba(0, 0, 0, 0), tag);
87 break;
88 case MuPDFInternalLinkDestination.InternalDestinationType.Fit:
89 case MuPDFInternalLinkDestination.InternalDestinationType.FitBoundingBox:
90 page.Graphics.FillRectangle(0, 0, page.Width, page.Height, Colour.FromRgba(0, 0, 0, 0), tag);
91 break;
92 case MuPDFInternalLinkDestination.InternalDestinationType.FitWidth:
93 case MuPDFInternalLinkDestination.InternalDestinationType.FitBoundingBoxWidth:
94 page.Graphics.FillRectangle(0, intern.Y, page.Width, page.Height - intern.Y, Colour.FromRgba(0, 0, 0, 0), tag);
95 break;
96 case MuPDFInternalLinkDestination.InternalDestinationType.FitHeight:
97 case MuPDFInternalLinkDestination.InternalDestinationType.FitBoundingBoxHeight:
98 page.Graphics.FillRectangle(intern.X, 0, page.Width - intern.X, page.Height, Colour.FromRgba(0, 0, 0, 0), tag);
99 break;
100 }
101
102 tbr.Add(link.Key, "#" + tag);
103 }
104 }
105 }
106
107 return tbr;
108 }

163 /// <param name="includeAnnotations">Whether annotations (e.g., signatures) should be included in the imported document.</param>

169 using (MuPDFDocument document = new MuPDFDocument(context, fileName))

171 return FromMuPDFDocument(context, document, pageNumber, textOption, includeAnnotations, null);

177 /// Import the specified page from a document.

179 /// <param name="fileName">The path to the document file.</param>
180 /// <param name="pageNumber">The page number (starting at 0).</param>
181 /// <param name="linkDestinations">When this method returns, this variable will contain a dictionary used to associate graphic action tags to hyperlinks. This can be used to enable such links when rendering the <see cref="Page"/> to a file.</param>
182 /// <param name="textOption">This parameter determines whether text elements are rendered as paths (which provides more accurate results) or as text.</param>
183 /// <param name="includeAnnotations">Whether annotations (e.g., signatures) should be included in the imported document.</param>
184 /// <returns>A <see cref="Page"/> containing a representation of the specified page from the document.</returns>
185 public static Page FromFile(string fileName, int pageNumber, out Dictionary<string, string> linkDestinations, SVGCreationOptions.TextOption textOption = SVGCreationOptions.TextOption.TextAsPath, bool includeAnnotations = true)
186 {
187 using (MuPDFContext context = new MuPDFContext())
188 {
189 using (MuPDFDocument document = new MuPDFDocument(context, fileName))
190 {
191 Dictionary<string, MuPDFLinkDestination> dests = new Dictionary<string, MuPDFLinkDestination>();
192 Page pag = FromMuPDFDocument(context, document, pageNumber, textOption, includeAnnotations, dests);
193 linkDestinations = SetUpLinks(pag, pageNumber, dests);
194 return pag;
