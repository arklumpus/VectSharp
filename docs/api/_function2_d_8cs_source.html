26 /// <summary>
27 /// Represents a function of two variables that has been sampled in some points.
28 /// </summary>
29 public class Function2DGrid
30 {
31 /// <summary>
32 /// Describes the arrangement of the points that have been sampled.
33 /// </summary>
34 public enum GridType
35 {
36 /// <summary>
37 /// Points have been sampled along a rectangular grid.
38 /// </summary>
39 Rectangular,
40
41 /// <summary>
42 /// Points have been sampled along a grid composed of hexagons with a diagonal
43 /// parallel to the horizontal.
44 /// </summary>
45 HexagonHorizontal,
46
47 /// <summary>
48 /// Points have been sampled along a grid composed of hexagons with a diagonal
49 /// parallel to the vertical.
50 /// </summary>
51 HexagonVertical,
52
53 /// <summary>
54 /// Points have been sampled without any particular criterion.
55 /// </summary>
56 Irregular
57 }
58
59 /// <summary>
60 /// The points where the function has been sampled.
61 /// </summary>
62 public IReadOnlyList<IReadOnlyList<double>> DataPoints { get; }
63
64 /// <summary>
65 /// The minimum X value that has been sampled.
66 /// </summary>
67 public double MinX { get; }
68
69 /// <summary>
70 /// The maximum X value that has been sampled.
71 /// </summary>
72 public double MaxX { get; }
73
74 /// <summary>
75 /// The minimum Y value that has been sampled.
76 /// </summary>
77 public double MinY { get; }
78
79 /// <summary>
80 /// The maximum Y value that has been sampled.
81 /// </summary>
82 public double MaxY { get; }
83
84 /// <summary>
85 /// The minimum value that has been obtained for the function.
86 /// </summary>
87 public double MinZ { get; }
88
89 /// <summary>
90 /// The maximum value that has been obtained for the function.
91 /// </summary>
92 public double MaxZ { get; }
93
94 /// <summary>
95 /// If the function has been sampled along a regular grid, the number of steps
96 /// between <see cref="MinX"/> and <see cref="MaxX"/> on the X axis.
97 /// </summary>
98 public int StepsX { get; private set; }
99
100 /// <summary>
101 /// If the function has been sampled along a regular grid, the number of steps
102 /// between <see cref="MinY"/> and <see cref="MaxY"/> on the X axis.
103 /// </summary>
104 public int StepsY { get; private set; }
105
106 /// <summary>
107 /// The type of grid.
108 /// </summary>
109 public GridType Type { get; private set; }
110
111 /// <summary>
112 /// Create a new <see cref="Function2DGrid"/> from a list of sampled values.
113 /// </summary>
114 /// <param name="dataPoints">The sampled values. Each element of this <see cref="IReadOnlyList{T}"/> should
115 /// be a collection of three values: the X coordinate of the sampled point, the Y coordinate of the sampled
116 /// point, and the value of the function at that point.</param>
117 public Function2DGrid(IReadOnlyList<IReadOnlyList<double>> dataPoints)
118 {
119 this.Type = GridType.Irregular;
120 this.DataPoints = dataPoints.ToArray();
121
122 MinX = double.MaxValue;
123 MaxX = double.MinValue;
124 MinY = double.MaxValue;
125 MaxY = double.MinValue;
126 MinZ = double.MaxValue;
127 MaxZ = double.MinValue;
128
129 this.StepsX = 0;
130 this.StepsY = 0;
131
132 for (int i = 0; i < dataPoints.Count; i++)
133 {
134 MinX = Math.Min(MinX, dataPoints[i][0]);
135 MinY = Math.Min(MinY, dataPoints[i][1]);
136 MinZ = Math.Min(MinZ, dataPoints[i][2]);
137
138 MaxX = Math.Max(MaxX, dataPoints[i][0]);
139 MaxY = Math.Max(MaxY, dataPoints[i][1]);
140 MaxZ = Math.Max(MaxZ, dataPoints[i][2]);
141 }
142 }

144 /// <summary>
145 /// Create a new <see cref="Function2DGrid"/> by sampling the provided function.
146 /// </summary>
147 /// <param name="function">The function to sample.</param>
148 /// <param name="minX">The minimum X value at which the function should be sampled.</param>
149 /// <param name="minY">The minimum Y value at which the function should be sampled.</param>
150 /// <param name="maxX">The maximum X value at which the function should be sampled.</param>
151 /// <param name="maxY">The maximum Y value at which the function should be sampled.</param>
152 /// <param name="stepsX">If <paramref name="type"/> is not <see cref="GridType.Irregular"/>,
153 /// the number of steps between <paramref name="minX"/> and <paramref name="maxX"/> on the X axis.
154 /// Otherwise, the number of sampled points is determined by multiplying <paramref name="stepsX"/> and
155 /// <paramref name="stepsY"/> together.</param>
156 /// <param name="stepsY">If <paramref name="type"/> is not <see cref="GridType.Irregular"/>,
157 /// the number of steps between <paramref name="minY"/> and <paramref name="maxY"/> on the Y axis.
158 /// Otherwise, the number of sampled points is determined by multiplying <paramref name="stepsX"/> and
159 /// <paramref name="stepsY"/> together.</param>
160 /// <param name="type">The strategy used to select points to sample. If this is <see cref="GridType.Irregular"/>,
161 /// uniformly distributed random points between (<paramref name="minX"/>, <paramref name="minY"/>) and
162 /// (<paramref name="maxX"/>, <paramref name="maxY"/>) are sampled.</param>
163 public Function2DGrid(Func<double[], double> function, double minX, double minY, double maxX, double maxY, int stepsX, int stepsY, GridType type)
164 {
165 MinX = minX;
166 MaxX = maxX;
167 MinY = minY;
168 MaxY = maxY;
169 StepsX = stepsX;
170 StepsY = stepsY;
171 Type = type;
172 MinZ = double.MaxValue;
173 MaxZ = double.MinValue;
174
175 if (type == GridType.HexagonHorizontal)
176 {
177 List<double[]> dataPoints = new List<double[]>((stepsX + 1) * (stepsY + 1));
178
179 for (int y = 0; y <= stepsY; y++)
180 {
181 for (int x = 0; x <= stepsX; x++)
182 {
183 if (x % 2 == 0 || y < stepsY)
184 {
185
186 double realY;
187
188 if (x % 2 == 0)
189 {
190 realY = minY + (maxY - minY) / stepsY * y;
191 }
192 else
193 {
194 realY = minY + (maxY - minY) / stepsY * (y + 0.5);
195 }
196
197 double realX = minX + (maxX - minX) / stepsX * x;
198 double z = function(new double[] { realX, realY });
199
200 if (!double.IsNaN(z))
201 {
202 MinZ = Math.Min(MinZ, z);
203 MaxZ = Math.Max(MaxZ, z);
204
205 dataPoints.Add(new double[] { realX, realY, z });
206 }
207 else
208 {
209 Type = GridType.Irregular;
210 }
211 }
212 }
213 }
214
215 this.DataPoints = dataPoints;
216 }
217 else if (type == GridType.HexagonVertical)
218 {
219 List<double[]> dataPoints = new List<double[]>((stepsX + 1) * (stepsY + 1));
220
221 for (int y = 0; y <= stepsY; y++)
222 {
223 double realY = minY + (maxY - minY) / stepsY * y;
224
225 for (int x = 0; x <= stepsX; x++)
226 {
227 if (y % 2 == 0 || x < stepsX)
228 {
229
230 double realX;
231
232 if (y % 2 == 0)
233 {
234 realX = minX + (maxX - minX) / stepsX * x;
235 }
236 else
237 {
238 realX = minX + (maxX - minX) / stepsX * (x + 0.5);
239 }
240
241 double z = function(new double[] { realX, realY });
242
243 if (!double.IsNaN(z))
244 {
245 MinZ = Math.Min(MinZ, z);
246 MaxZ = Math.Max(MaxZ, z);
247
248 dataPoints.Add(new double[] { realX, realY, z });
249 }
250 else
251 {
252 Type = GridType.Irregular;
253 }
254 }
255 }
256 }
257
258 this.DataPoints = dataPoints;
259 }
260 else if (type == GridType.Rectangular)
261 {
262 List<double[]> dataPoints = new List<double[]>((stepsX + 1) * (stepsY + 1));
263
264 for (int y = 0; y <= stepsY; y++)
265 {
266 double realY = minY + (maxY - minY) / stepsY * y;
267 for (int x = 0; x <= stepsX; x++)
268 {
269 double realX = minX + (maxX - minX) / stepsX * x;
270 double z = function(new double[] { realX, realY });
271
272 if (!double.IsNaN(z))
273 {
274 MinZ = Math.Min(MinZ, z);
275 MaxZ = Math.Max(MaxZ, z);
276
277 dataPoints.Add(new double[] { realX, realY, z });
278 }
279 else
280 {
281 Type = GridType.Irregular;
282 }
283 }
284 }
285
286 this.DataPoints = dataPoints;
287 }
288 else //type == GridType.Irregular
289 {
290 double[] xs = ContinuousUniform.Samples(minX, maxX).Take((stepsX + 1) * (stepsY + 1)).ToArray();
291 double[] ys = ContinuousUniform.Samples(minY, maxY).Take((stepsX + 1) * (stepsY + 1)).ToArray();
292
293 List<double[]> dataPoints = new List<double[]>((stepsX + 1) * (stepsY + 1));
294
295 for (int i = 0; i < (stepsX + 1) * (stepsY + 1); i++)
296 {
297 double z = function(new double[] { xs[i], ys[i] });
298
299 if (!double.IsNaN(z))
300 {
301 MinZ = Math.Min(MinZ, z);
302 MaxZ = Math.Max(MaxZ, z);
303
304 dataPoints.Add(new double[] { xs[i], ys[i], z });
305 }
306 }
307
308 this.DataPoints = dataPoints;
309 }
310 }

312 /// <summary>
313 /// Converts a hexagonal grid into a rectangular grid.
314 /// </summary>
315 /// <returns>If <see cref="Type"/> is <see cref="GridType.Rectangular"/>, this method returns the current instance. <br/>
316 /// If <see cref="Type"/> is <see cref="GridType.HexagonHorizontal"/> or <see cref="GridType.HexagonVertical"/>, a
317 /// new <see cref="Function2DGrid"/> with <see cref="Type"/> equal to <see cref="GridType.Rectangular"/> is returned.
318 /// The sampled points in this grid are obtained by performing a bilinear interpolation on the sampled points from
319 /// this grid. The returned grid will always be "denser" than the current instance.<br/>
320 /// If <see cref="Type"/> is <see cref="GridType.Irregular"/>, an <see cref="InvalidOperationException"/> is thrown.</returns>
321 /// <exception cref="InvalidOperationException">Thrown if <see cref="Type"/> is <see cref="GridType.Irregular"/>.</exception>
322 public Function2DGrid ToRectangular()
323 {
324 if (this.Type == GridType.Rectangular)
325 {
326 return this;
327 }
328 else if (this.Type == GridType.HexagonHorizontal)
329 {
330 IReadOnlyList<double>[] data = new IReadOnlyList<double>[(2 * this.StepsY + 1) * (this.StepsX + 1)];
331
332 int newStepsX = this.StepsX;
333 int newStepsY = this.StepsY * 2;
334
335 for (int i = 0; i < this.DataPoints.Count; i++)
336 {
337 int x = (int)Math.Round((this.DataPoints[i][0] - this.MinX) / (this.MaxX - this.MinX) * newStepsX);
338 int y = (int)Math.Round((this.DataPoints[i][1] - this.MinY) / (this.MaxY - this.MinY) * newStepsY);
339
340 data[y * (newStepsX + 1) + x] = this.DataPoints[i];
341 }
342
343 for (int y = 0; y <= newStepsY; y++)
344 {
345 for (int x = 0; x <= newStepsX; x++)
346 {
347 if (data[y * (newStepsX + 1) + x] == null)
348 {
349 double xVal = this.MinX + (this.MaxX - this.MinX) / newStepsX * x;
350 double yVal = this.MinY + (this.MaxY - this.MinY) / newStepsY * y;
351
352 double zVal;
353
354 if (x > 0)
355 {
356 if (y > 0)
357 {
358 if (x < newStepsX)
359 {
360 if (y < newStepsY)
361 {
362 zVal = (data[(y - 1) * (newStepsX + 1) + x][2] + data[(y + 1) * (newStepsX + 1) + x][2] + data[y * (newStepsX + 1) + (x - 1)][2] + data[y * (newStepsX + 1) + (x + 1)][2]) * 0.25;
363 }
364 else
365 {
366 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[y * (newStepsX + 1) + (x + 1)][2]) * 0.5;
367 }
368 }
369 else
370 {
371 if (y < newStepsY)
372 {
373 zVal = (data[(y - 1) * (newStepsX + 1) + x][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
374 }
375 else
376 {
377 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[(y - 1) * (newStepsX + 1) + x][2]) * 0.5;
378 }
379 }
380 }
381 else
382 {
383 if (x < newStepsX)
384 {
385 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[y * (newStepsX + 1) + (x + 1)][2]) * 0.5;
386 }
387 else
388 {
389 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
390 }
391 }
392 }
393 else
394 {
395 if (y > 0)
396 {
397 if (y < newStepsY)
398 {
399 zVal = (data[(y - 1) * (newStepsX + 1) + x][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
400 }
401 else
402 {
403 zVal = (data[y * (newStepsX + 1) + (x + 1)][2] + data[(y - 1) * (newStepsX + 1) + x][2]) * 0.5;
404 }
405 }
406 else
407 {
408 zVal = (data[y * (newStepsX + 1) + (x + 1)][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
409 }
410 }
411
412 data[y * (newStepsX + 1) + x] = new double[] { xVal, yVal, zVal };
413 }
414 }
415 }
416
417 return new Function2DGrid(data) { Type = GridType.Rectangular, StepsX = newStepsX, StepsY = newStepsY };
418 }
419 else if (this.Type == GridType.HexagonVertical)
420 {
421 IReadOnlyList<double>[] data = new IReadOnlyList<double>[(this.StepsY + 1) * (2 * this.StepsX + 1)];
422
423 int newStepsX = this.StepsX * 2;
424 int newStepsY = this.StepsY;
425
426 for (int i = 0; i < this.DataPoints.Count; i++)
427 {
428 int x = (int)Math.Round((this.DataPoints[i][0] - this.MinX) / (this.MaxX - this.MinX) * newStepsX);
429 int y = (int)Math.Round((this.DataPoints[i][1] - this.MinY) / (this.MaxY - this.MinY) * newStepsY);
430
431 data[y * (newStepsX + 1) + x] = this.DataPoints[i];
432 }
433
434 for (int y = 0; y <= newStepsY; y++)
435 {
436 for (int x = 0; x <= newStepsX; x++)
437 {
438 if (data[y * (newStepsX + 1) + x] == null)
439 {
440 double xVal = this.MinX + (this.MaxX - this.MinX) / newStepsX * x;
441 double yVal = this.MinY + (this.MaxY - this.MinY) / newStepsY * y;
442
443 double zVal;
444
445 if (x > 0)
446 {
447 if (y > 0)
448 {
449 if (x < newStepsX)
450 {
451 if (y < newStepsY)
452 {
453 zVal = (data[(y - 1) * (newStepsX + 1) + x][2] + data[(y + 1) * (newStepsX + 1) + x][2] + data[y * (newStepsX + 1) + (x - 1)][2] + data[y * (newStepsX + 1) + (x + 1)][2]) * 0.25;
454 }
455 else
456 {
457 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[y * (newStepsX + 1) + (x + 1)][2]) * 0.5;
458 }
459 }
460 else
461 {
462 if (y < newStepsY)
463 {
464 zVal = (data[(y - 1) * (newStepsX + 1) + x][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
465 }
466 else
467 {
468 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[(y - 1) * (newStepsX + 1) + x][2]) * 0.5;
469 }
470 }
471 }
472 else
473 {
474 if (x < newStepsX)
475 {
476 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[y * (newStepsX + 1) + (x + 1)][2]) * 0.5;
477 }
478 else
479 {
480 zVal = (data[y * (newStepsX + 1) + (x - 1)][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
481 }
482 }
483 }
484 else
485 {
486 if (y > 0)
487 {
488 if (y < newStepsY)
489 {
490 zVal = (data[(y - 1) * (newStepsX + 1) + x][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
491 }
492 else
493 {
494 zVal = (data[y * (newStepsX + 1) + (x + 1)][2] + data[(y - 1) * (newStepsX + 1) + x][2]) * 0.5;
495 }
496 }
497 else
498 {
499 zVal = (data[y * (newStepsX + 1) + (x + 1)][2] + data[(y + 1) * (newStepsX + 1) + x][2]) * 0.5;
500 }
501 }
502
503 data[y * (newStepsX + 1) + x] = new double[] { xVal, yVal, zVal };
504 }
505 }
506 }
507
508 return new Function2DGrid(data) { Type = GridType.Rectangular, StepsX = newStepsX, StepsY = newStepsY };
509 }
510 else
511 {
512 throw new InvalidOperationException("Cannot convert an irregular grid into a rectangular grid!");
513 }
514 }

517 /// <summary>
518 /// A plot element that plots a function of two variables.
519 /// </summary>
520 public class Function2D : IPlotElement
521 {
522 /// <summary>
523 /// Describes the kind of plots that can be produced.
524 /// </summary>
525 public enum PlotType
526 {
527 /// <summary>
528 /// A symbol is drawn at each sampled point, whose
529 /// colour depends on the value of the function at that point.
530 /// </summary>
531 SampledPoints,
532
533 /// <summary>
534 /// The plot area is tessellated with cells whose colour
535 /// depends on the value of the function at a point within the cell.
536 /// </summary>
537 Tessellation,
538
539 /// <summary>
540 /// A rasterised tessellation is created and then stretched and interpolated
541 /// to fill the plot area.
542 /// </summary>
543 Raster
544 }
545
546 /// <summary>
547 /// The symbol to draw at the sampled points.
548 /// </summary>
549 public IDataPointElement SampledPointElement { get; set; } = new PathDataPointElement();
550
551 /// <summary>
552 /// Resolution on the X axis for the rasterised tesselation.
553 /// </summary>
554 public int RasterResolutionX { get; set; } = 512;
555
556 /// <summary>
557 /// Resolution on the Y axis for the rasterised tesselation.
558 /// </summary>
559 public int RasterResolutionY { get; set; } = 512;
560
561 /// <summary>
562 /// The function to plot.
563 /// </summary>
564 public Function2DGrid Function { get; set; }
565
566 /// <summary>
567 /// The kind of plot that is produced.
568 /// </summary>
569 public PlotType Type { get; set; } = PlotType.SampledPoints;
570
571 /// <summary>
572 /// A function associating sampled function values to a <see cref="Colour"/>. You should
573 /// set this to a function accepting a single <see langword="double"/> argument ranging between
574 /// 0 and 1, and returning the corresponding colour. The default value returns black 0 and white for 1.
575 /// </summary>
576 public Func<double, Colour> Colouring { get; set; } = x => { x = Math.Max(0, Math.Min(1, x)); return Colour.FromRgb(x, x, x); };
577
578 /// <summary>
579 /// A tag to identify the function in the plot.
580 /// </summary>
581 public string Tag { get; set; }
582
583 /// <summary>
584 /// The coordinate system used to transform the points from data space to plot space.
585 /// </summary>
586 public IContinuousInvertibleCoordinateSystem CoordinateSystem { get; set; }
587 ICoordinateSystem IPlotElement.CoordinateSystem => this.CoordinateSystem;
588
589 /// <summary>
590 /// Create a new <see cref="Function2D"/> instance.
591 /// </summary>
592 /// <param name="function">The function to plot.</param>
593 /// <param name="coordinateSystem">The coordinate system used to transform the points from data space to plot space.</param>
594 public Function2D(Function2DGrid function, IContinuousInvertibleCoordinateSystem coordinateSystem)
595 {
596 this.Function = function;
597 this.CoordinateSystem = coordinateSystem;
598 }
599
600 /// <inheritdoc/>
601 public unsafe void Plot(Graphics target)
602 {
603 if (Type == PlotType.SampledPoints)
604 {
605 Point topLeft = CoordinateSystem.ToPlotCoordinates(new double[] { Function.MinX, Function.MaxY });
606 Point bottomRight = CoordinateSystem.ToPlotCoordinates(new double[] { Function.MaxX, Function.MinY });
607
608 double size = Math.Min(Math.Abs(topLeft.X - bottomRight.X), Math.Abs(topLeft.Y - bottomRight.Y)) * 0.0075;
609
610 if (Function.Type != Function2DGrid.GridType.Irregular)
611 {
612 size = Math.Min(Math.Abs(topLeft.X - bottomRight.X) / Function.StepsX, Math.Abs(topLeft.Y - bottomRight.Y) / Function.StepsY) * 0.4;
613 }
614
615 for (int i = 0; i < Function.DataPoints.Count; i++)
616 {
617 Point pt = CoordinateSystem.ToPlotCoordinates(Function.DataPoints[i]);
618 Colour col = Colouring((Function.DataPoints[i][2] - Function.MinZ) / (Function.MaxZ - Function.MinZ));
619
620 string tag = Tag;
621 if (!string.IsNullOrEmpty(tag) && target.UseUniqueTags)
622 {
623 tag += "@" + i.ToString();
624 }
625
626 target.Save();
627 target.Translate(pt);
628 target.Scale(size, size);
629
630 SampledPointElement.Plot(target, new PlotElementPresentationAttributes() { Fill = col, Stroke = null }, tag);
631
632 target.Restore();
633 }
634 }
635 else if (Type == PlotType.Tessellation || (Type == PlotType.Raster && Function.Type == Function2DGrid.GridType.Irregular))
636 {
637 double width = (Function.MaxX - Function.MinX) / Function.StepsX;
638 double height = (Function.MaxY - Function.MinY) / Function.StepsY;
639
640 Point topLeft = CoordinateSystem.ToPlotCoordinates(new double[] { Function.MinX, Function.MaxY });
641 Point topRight = CoordinateSystem.ToPlotCoordinates(new double[] { Function.MaxX, Function.MaxY });
642 Point bottomRight = CoordinateSystem.ToPlotCoordinates(new double[] { Function.MaxX, Function.MinY });
643 Point bottomLeft = CoordinateSystem.ToPlotCoordinates(new double[] { Function.MinX, Function.MinY });
644
645
646 Graphics strokes = new Graphics();
647 Graphics fills = new Graphics();
648
649 bool anyTransparent = false;
650
651 if (Function.Type != Function2DGrid.GridType.Irregular)
652 {
653 for (int i = 0; i < Function.DataPoints.Count; i++)
654 {
655 Colour col = Colouring((Function.DataPoints[i][2] - Function.MinZ) / (Function.MaxZ - Function.MinZ));
656
657 if (col.A != 1)
658 {
659 anyTransparent = true;
660 }
661
662 if (Function.Type == Function2DGrid.GridType.Rectangular)
663 {
664 Point p1 = CoordinateSystem.ToPlotCoordinates(new double[] { Function.DataPoints[i][0] - width * 0.5, Function.DataPoints[i][1] - height * 0.5 });
665 Point p2 = CoordinateSystem.ToPlotCoordinates(new double[] { Function.DataPoints[i][0] - width * 0.5, Function.DataPoints[i][1] + height * 0.5 });
