216 private void RenderHTMLImage(HtmlTag imgTag, bool isInline, ref MarkdownContext context, ref Graphics graphics, NewPageAction newPageAction)
217 {
218 if (imgTag != null)
219 {
220 if (imgTag.Attributes.TryGetValue("src", out string imageSrc))
221 {
222 string imageFile;
223 bool wasDownloaded;
224
225 if (imageSrc.StartsWith("emoji://") || imageSrc.StartsWith("unicode://"))
226 {
227 (imageFile, wasDownloaded) = this.EmojiUriResolver(imageSrc);
228 }
229 else
230 {
231 (imageFile, wasDownloaded) = this.ImageUriResolver(imageSrc, this.BaseImageUri);
232 }
233
234 Page imagePage = null;
235
236 if (imageFile != null)
237 {
238 if (System.IO.Path.GetExtension(imageFile) == ".svg")
239 {
240 try
241 {
242 imagePage = VectSharp.SVG.Parser.FromFile(imageFile);
243 }
244 catch
245 {
246 imagePage = null;
247 }
248 }
249 else if (RasterImageLoader != null)
250 {
251 try
252 {
253 RasterImage raster = RasterImageLoader(imageFile);
254 imagePage = new Page(raster.Width, raster.Height);
255 imagePage.Graphics.DrawRasterImage(0, 0, raster, tag: context.Tag);
256 }
257 catch
258 {
259 imagePage = null;
260 }
261 }
262
263
264 if (wasDownloaded)
265 {
266 System.IO.File.Delete(imageFile);
267 System.IO.Directory.Delete(System.IO.Path.GetDirectoryName(imageFile));
268 }
269 }
270 else if (imageSrc.StartsWith("data:"))
271 {
272 try
273 {
274 imagePage = VectSharp.SVG.Parser.ParseImageURI(imageSrc, false);
275 }
276 catch
277 {
278
279 imagePage = null;
280 }
281 }
282
283 if (imagePage != null)
284 {
285 double scaleX = 1;
286 double scaleY = 1;
287
288 bool hasWidth = false;
289 bool hasHeight = false;
290
291 if (imgTag.Attributes.TryGetValue("width", out string widthString) && double.TryParse(widthString, System.Globalization.NumberStyles.Any, System.Globalization.CultureInfo.InvariantCulture, out double width))
292 {
293 hasWidth = true;
294 scaleX = width * this.ImageUnitMultiplier / imagePage.Width;
295 }
296
297 if (imgTag.Attributes.TryGetValue("height", out string heightString) && double.TryParse(heightString, System.Globalization.NumberStyles.Any, System.Globalization.CultureInfo.InvariantCulture, out double height))
298 {
299 hasHeight = true;
300 scaleY = height * this.ImageUnitMultiplier / imagePage.Height;
301 }
302
303 if (hasWidth && !hasHeight)
304 {
305 scaleY = scaleX;
306 }
307 else if (hasHeight && !hasWidth)
308 {
309 scaleX = scaleY;
310 }
311
312 scaleX *= this.ImageMultiplier;
313 scaleY *= this.ImageMultiplier;
314
315 if (!isInline)
316 {
317 if (!imgTag.Attributes.TryGetValue("align", out string alignValue))
318 {
319 alignValue = null;
320 }
321
322 if (alignValue == "center")
323 {
324 if (context.CurrentLine != null)
325 {
326 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);
327 context.CurrentLine = null;
328 }
329
330 double minX = context.GetMinX(context.Cursor.Y + context.Translation.Y, context.Cursor.Y + context.Translation.Y + scaleY * imagePage.Height);
331 double maxX = context.GetMaxX(context.Cursor.Y + context.Translation.Y, context.Cursor.Y + context.Translation.Y + scaleY * imagePage.Height, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X);
332
333 if (scaleX * imagePage.Width > maxX - minX)
334 {
335 scaleX = (maxX - minX) / imagePage.Width;
336 scaleY = scaleX;
337 }
338
339 double finalY = context.Cursor.Y + scaleY * imagePage.Height;
340
341 if (finalY + context.Translation.Y > this.PageSize.Height - this.Margins.Bottom + this.ImageMarginTolerance - context.MarginBottomRight.Y)
342 {
343 newPageAction(ref context, ref graphics);
344 finalY = context.Cursor.Y + scaleY * imagePage.Height;
345 }
346
347 graphics.Save();
348 graphics.Translate((minX + maxX - scaleX * imagePage.Width) * 0.5, context.Cursor.Y);
349 graphics.Scale(scaleX, scaleY);
350 graphics.SetClippingPath(0, 0, imagePage.Width, imagePage.Height);
351 graphics.DrawGraphics(0, 0, imagePage.Graphics);
352
353 graphics.Restore();
354
355 context.Cursor = new Point(0, finalY + SpaceAfterParagraph * context.Font.FontSize + SpaceAfterLine * context.Font.FontSize);
356 }
357 else if (alignValue == "right")
358 {
359 if (context.CurrentLine != null)
360 {
361 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);
362 context.CurrentLine = null;
363 }
364
365 double finalY = context.Cursor.Y + scaleY * imagePage.Height;
366
367 if (finalY + context.Translation.Y > this.PageSize.Height - this.Margins.Bottom + this.ImageMarginTolerance - context.MarginBottomRight.Y)
368 {
369 newPageAction(ref context, ref graphics);
370 }
371
372 graphics.Save();
373 graphics.Translate(this.PageSize.Width - this.Margins.Right - context.Translation.X - scaleX * imagePage.Width - context.MarginBottomRight.X, context.Cursor.Y);
374 graphics.Scale(scaleX, scaleY);
375 graphics.SetClippingPath(0, 0, imagePage.Width, imagePage.Height);
376
377 graphics.DrawGraphics(0, 0, imagePage.Graphics);
378
379 graphics.Restore();
380
381 context.ForbiddenAreasRight.Add((this.PageSize.Width - this.Margins.Right - scaleX * imagePage.Width - this.ImageSideMargin - context.MarginBottomRight.X, context.Cursor.Y + context.Translation.Y, context.Cursor.Y + context.Translation.Y + scaleY * imagePage.Height));
382 }
383 else if (alignValue == "left")
384 {
385 if (context.CurrentLine != null)
386 {
387 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);
388 context.CurrentLine = null;
389 }
390
391 double finalY = context.Cursor.Y + scaleY * imagePage.Height;
392
393 if (finalY + context.Translation.Y > this.PageSize.Height - this.Margins.Bottom + this.ImageMarginTolerance - context.MarginBottomRight.Y)
394 {
395 newPageAction(ref context, ref graphics);
396 }
397
398 graphics.Save();
399 graphics.Translate(0, context.Cursor.Y);
400 graphics.Scale(scaleX, scaleY);
401 graphics.SetClippingPath(0, 0, imagePage.Width, imagePage.Height);
402
403 graphics.DrawGraphics(0, 0, imagePage.Graphics);
404
405 graphics.Restore();
406
407 context.ForbiddenAreasLeft.Add((scaleX * imagePage.Width + context.Translation.X + this.ImageSideMargin, context.Cursor.Y + context.Translation.Y, context.Cursor.Y + context.Translation.Y + scaleY * imagePage.Height));
408 }
409 else
410 {
411 isInline = true;
412 }
413 }
414
415 if (isInline)
416 {
417 Graphics scaledImage = new Graphics();
418 scaledImage.Scale(scaleX, scaleY);
419 scaledImage.DrawGraphics(0, 0, imagePage.Graphics);
420
421 if (context.CurrentLine == null)
422 {
423 context.CurrentLine = new Line(context.Font.Ascent);
424
425 context.Cursor = new Point(0, context.Cursor.Y);
426 double minX = context.GetMinX(context.Cursor.Y - scaleY * imagePage.Height, context.Cursor.Y);
427 context.Cursor = new Point(minX, context.Cursor.Y);
428 }
429
430 double currLineMaxX = context.GetMaxX(context.Cursor.Y - scaleY * imagePage.Height, context.Cursor.Y, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X);
431
432 double finalX = context.Cursor.X + imagePage.Width;
433
434 if (finalX > currLineMaxX)
435 {
436 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);
437
438 context.CurrentLine = new Line(context.Font.Ascent);
439 context.Cursor = new Point(0, context.Cursor.Y - context.Font.Descent + SpaceAfterLine * context.Font.FontSize + context.Font.Ascent);
440
441 double minX = context.GetMinX(context.Cursor.Y - scaleY * imagePage.Height * scaleY, context.Cursor.Y);
442
443 context.Cursor = new Point(minX, context.Cursor.Y);
444 }
445
446 context.CurrentLine.Fragments.Add(new GraphicsFragment(new Point(context.Cursor.X, context.Cursor.Y - scaleY * imagePage.Height), scaledImage, scaleY * imagePage.Height));
447 context.Cursor = new Point(context.Cursor.X + scaleX * imagePage.Width, context.Cursor.Y);
448 }
449 }
450 }
451 }

454 static (int, string)[] RomanNumbers = new (int, string)[] { (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"), (100, "C"), (90, "XC"), (50, "L"), (40, "XL"), (10, "X"), (9, "IX"), (5, "V"), (4, "IV"), (1, "I") };

456 private static string GetRomanNumeral(int number)
457 {
458 StringBuilder tbr = new StringBuilder();
459
460 for (int i = 0; i < RomanNumbers.Length; i++)
461 {
462 while (number >= RomanNumbers[i].Item1)
463 {
464 tbr.Append(RomanNumbers[i].Item2);
465 number -= RomanNumbers[i].Item1;
466 }
467 }
468
469 return tbr.ToString();
470 }
