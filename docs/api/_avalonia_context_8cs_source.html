68 public static double[] Multiply(double[,] matrix, double[] vector)
69 {
70 double[] tbr = new double[2];
71
72 tbr[0] = matrix[0, 0] * vector[0] + matrix[0, 1] * vector[1] + matrix[0, 2];
73 tbr[1] = matrix[1, 0] * vector[0] + matrix[1, 1] * vector[1] + matrix[1, 2];
74
75 return tbr;
76 }

78 public static double[,] Multiply(double[,] matrix1, double[,] matrix2)
79 {
80 double[,] tbr = new double[3, 3];
81
82 for (int i = 0; i < 3; i++)
83 {
84 for (int j = 0; j < 3; j++)
85 {
86 for (int k = 0; k < 3; k++)
87 {
88 tbr[i, j] += matrix1[i, k] * matrix2[k, j];
89 }
90 }
91 }
92
93 return tbr;
94 }

96 public static double[,] Rotate(double[,] matrix, double angle)
97 {
98 double[,] rotationMatrix = new double[3, 3];
99 rotationMatrix[0, 0] = Math.Cos(angle);
100 rotationMatrix[0, 1] = -Math.Sin(angle);
101 rotationMatrix[1, 0] = Math.Sin(angle);
102 rotationMatrix[1, 1] = Math.Cos(angle);
103 rotationMatrix[2, 2] = 1;
104
105 return Multiply(matrix, rotationMatrix);

136 tbr[0, 1] = -(m[0, 1] * m[2, 2] - m[0, 2] * m[2, 1]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);

138 tbr[1, 0] = -(m[1, 0] * m[2, 2] - m[1, 2] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
139 tbr[1, 1] = (m[0, 0] * m[2, 2] - m[0, 2] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
140 tbr[1, 2] = -(m[0, 0] * m[1, 2] - m[0, 2] * m[1, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
141 tbr[2, 0] = (m[1, 0] * m[2, 1] - m[1, 1] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
142 tbr[2, 1] = -(m[0, 0] * m[2, 1] - m[0, 1] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);

286 private void PathText(string text, double x, double y)
287 {
288 Utils.CoerceNaNAndInfinityToZero(ref x, ref y);
289
290 GraphicsPath textPath = new GraphicsPath().AddText(x, y, text, Font, TextBaseline);
291
292 for (int j = 0; j < textPath.Segments.Count; j++)
293 {
294 switch (textPath.Segments[j].Type)
295 {
296 case VectSharp.SegmentType.Move:
297 this.MoveTo(textPath.Segments[j].Point.X, textPath.Segments[j].Point.Y);
298 break;
299 case VectSharp.SegmentType.Line:
300 this.LineTo(textPath.Segments[j].Point.X, textPath.Segments[j].Point.Y);
301 break;
302 case VectSharp.SegmentType.CubicBezier:
303 this.CubicBezierTo(textPath.Segments[j].Points[0].X, textPath.Segments[j].Points[0].Y, textPath.Segments[j].Points[1].X, textPath.Segments[j].Points[1].Y, textPath.Segments[j].Points[2].X, textPath.Segments[j].Points[2].Y);
304 break;
305 case VectSharp.SegmentType.Close:
306 this.Close();
307 break;
308 }

320 public void FillText(string text, double x, double y)
321 {
322 Utils.CoerceNaNAndInfinityToZero(ref x, ref y);
323
324 if (_textOption == AvaloniaContextInterpreter.TextOptions.NeverConvert || (_textOption == AvaloniaContextInterpreter.TextOptions.ConvertIfNecessary && Font.FontFamily.IsStandardFamily && Font.FontFamily.FileName != "ZapfDingbats" && Font.FontFamily.FileName != "Symbol"))
325 {
326
327 TextBlock blk = new TextBlock() { ClipToBounds = false, Text = text, FontFamily = Avalonia.Media.FontFamily.Parse(FontFamily), FontSize = Font.FontSize, FontStyle = (Font.FontFamily.IsOblique ? FontStyle.Oblique : Font.FontFamily.IsItalic ? FontStyle.Italic : FontStyle.Normal), FontWeight = (Font.FontFamily.IsBold ? FontWeight.Bold : FontWeight.Regular), TextAlignment = TextAlignment.Left };
328
329 double top = y;
330 double left = x;
331
332 double[,] currTransform = null;
333 double[,] deltaTransform = MatrixUtils.Identity;
334
335 if (Font.FontFamily.TrueTypeFile != null)
336 {
337 currTransform = MatrixUtils.Translate(_transform, x, y);
338 }
339
340 Font.DetailedFontMetrics metrics = Font.MeasureTextAdvanced(text);
341
342 if (text.StartsWith(" "))
343 {
344 var spaceMetrics = Font.MeasureTextAdvanced(" ");
345
346 for (int i = 0; i < text.Length; i++)
347 {
348 if (text[i] == ' ')
349 {
350 left -= spaceMetrics.AdvanceWidth;
351 }
352 else
353 {
354 break;
355 }
356 }
357 }
358
359 if (TextBaseline == TextBaselines.Top)
360 {
361 blk.VerticalAlignment = Avalonia.Layout.VerticalAlignment.Top;
362
363 if (Font.FontFamily.TrueTypeFile != null)
364 {
365 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
366 {
367 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top + metrics.Top - Font.WinAscent);
368 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top + metrics.Top - Font.WinAscent);
369 }
370 else
371 {
372 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top + metrics.Top - Font.Ascent);
373 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top + metrics.Top - Font.Ascent);
374 }
375 }
376 }
377 else if (TextBaseline == TextBaselines.Middle)
378 {
379 blk.VerticalAlignment = Avalonia.Layout.VerticalAlignment.Top;
380
381 if (Font.FontFamily.TrueTypeFile != null)
382 {
383 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
384 {
385 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top + metrics.Top / 2 + metrics.Bottom / 2 - Font.WinAscent);
386 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top + metrics.Top / 2 + metrics.Bottom / 2 - Font.WinAscent);
387 }
388 else
389 {
390 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top + metrics.Top / 2 + metrics.Bottom / 2 - Font.Ascent);
391 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top + metrics.Top / 2 + metrics.Bottom / 2 - Font.Ascent);
392 }
393
394 }
395 }
396 else if (TextBaseline == TextBaselines.Baseline)
397 {
398 blk.VerticalAlignment = Avalonia.Layout.VerticalAlignment.Top;
399
400 if (Font.FontFamily.TrueTypeFile != null)
401 {
402 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
403 {
404 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top - Font.WinAscent);
405 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top - Font.YMax);
406 }
407 else
408 {
409 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top - Font.Ascent);
410 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top - Font.YMax);
411 }
412 }
413 }
414 else if (TextBaseline == TextBaselines.Bottom)
415 {
416 blk.VerticalAlignment = Avalonia.Layout.VerticalAlignment.Bottom;
417
418 if (Font.FontFamily.TrueTypeFile != null)
419 {
420 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
421 {
422 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top - Font.WinAscent + metrics.Bottom);
423 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top - Font.WinAscent + metrics.Bottom);
424 }
425 else
426 {
427 currTransform = MatrixUtils.Translate(_transform, left - metrics.LeftSideBearing * 2, top - Font.Ascent + metrics.Bottom);
428 deltaTransform = MatrixUtils.Translate(deltaTransform, left - metrics.LeftSideBearing * 2, top - Font.Ascent + metrics.Bottom);
429 }
430 }
431 }
432
433 blk.RenderTransform = new MatrixTransform(currTransform.ToAvaloniaMatrix());
434 blk.RenderTransformOrigin = new Avalonia.RelativePoint(0, 0, Avalonia.RelativeUnit.Absolute);
435
436 Avalonia.Media.Brush foreground = null;
437
438 if (this.FillStyle is SolidColourBrush solid)
439 {
440 foreground = new SolidColorBrush(Color.FromArgb(FillAlpha, (byte)(solid.R * 255), (byte)(solid.G * 255), (byte)(solid.B * 255)));
441 }
442 else if (this.FillStyle is LinearGradientBrush linearGradient)
443 {
444 foreground = linearGradient.ToLinearGradientBrush(deltaTransform);
445 }
446 else if (this.FillStyle is RadialGradientBrush radialGradient)
447 {
448 foreground = radialGradient.ToRadialGradientBrush(metrics.Width + metrics.LeftSideBearing + metrics.RightSideBearing, deltaTransform);
449 }
450
451 blk.Foreground = foreground;
452
453 currControlElement.Children.Add(blk);
454
455 if (!string.IsNullOrEmpty(Tag))
456 {
457 if (TaggedActions.ContainsKey(Tag))
458 {
459 TaggedActions[Tag].DynamicInvoke(blk);
460
461 if (removeTaggedActions)
462 {
463 TaggedActions.Remove(Tag);
464 }
465 }
466 }
467 }
468 else
469 {
470 PathText(text, x, y);
471 Fill(FillRule.NonZeroWinding);
472 }
473 }
