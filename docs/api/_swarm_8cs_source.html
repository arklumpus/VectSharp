90 private static (Point, double) GetMinDistance(Point pt, List<Point> plottedPoints)
91 {
92 double minDist = double.MaxValue;
93 Point minPoint = new Point();
94
95 for (int i = 0; i < plottedPoints.Count; i++)
96 {
97 double dist = (plottedPoints[i] - pt).Modulus();
98 if (dist < minDist)
99 {
100 minDist = dist;
101 minPoint = plottedPoints[i];
102 }
103 }
104
105 return (minPoint, minDist);
106 }

108 private IReadOnlyList<double> ProposePoint(IReadOnlyList<double> pt, List<Point> plottedPoints, IReadOnlyList<double> perpendicularDirection)
109 {
110 Point projPt = CoordinateSystem.ToPlotCoordinates(pt);
111
112 (Point closestPoint, double minDist) = GetMinDistance(projPt, plottedPoints);
113
114 while (minDist < PointSize * 2 + PointMargin * PointSize)
115 {
116 Point localPerpDirectionPt = (CoordinateSystem.ToPlotCoordinates(CoordinateSystem.GetAround(CoordinateSystem.ToDataCoordinates(projPt), perpendicularDirection)) - projPt).Normalize();
117 double parallDist = (projPt - closestPoint).Select((x, i) => x * localPerpDirectionPt[i]).Sum();
118 projPt = projPt + localPerpDirectionPt * (-parallDist + Math.Sqrt(parallDist * parallDist - (projPt - closestPoint).Select(x => x * x).Sum() + (PointSize * 2 + PointMargin * PointSize) * (PointSize * 2 + PointMargin * PointSize))) * 1.0001;
119 (closestPoint, minDist) = GetMinDistance(projPt, plottedPoints);
120 }
121
122 return CoordinateSystem.ToDataCoordinates(projPt);
123 }

125 private static double[] GetPerpendicularVector(IReadOnlyList<double> v)
126 {
127 double[] tbr = new double[v.Count];
128
129 int m = -1;
130
131 for (int i = 0; i < v.Count; i++)
132 {
133 if (v[i] != 0)
134 {
135 m = i; break;
136 }
137 }
138
139 if (m < 0)
140 {
141 return tbr;
142 }
143
144 int n = m == 0 ? 1 : 0;
145
146 double mod = Math.Sqrt(v[m] * v[m] + v[n] * v[n]);
147
148 tbr[m] = -v[n] / mod;
149 tbr[n] = v[m] / mod;
150
151 return tbr;
152 }

154 /// <inheritdoc/>
155 public void Plot(Graphics target)
156 {
157 List<Point> plottedPoints = new List<Point>();
158
159 double[] perpDirection1 = GetPerpendicularVector(this.Direction);
160 double[] perpDirection2 = perpDirection1.Select(x => -x).ToArray();
161
162 for (int i = 0; i < Data.Count; i++)
163 {
164 double[] point = new double[this.Position.Count];
165
166 for (int j = 0; j < this.Position.Count; j++)
167 {
168 point[j] = this.Position[j] + Data[i] * this.Direction[j];
169 }
170
171 IReadOnlyList<double> point1 = ProposePoint(point, plottedPoints, perpDirection1);
172 IReadOnlyList<double> point2 = ProposePoint(point, plottedPoints, perpDirection2);
173
174 Point proj = CoordinateSystem.ToPlotCoordinates(point);
175 Point pt1 = CoordinateSystem.ToPlotCoordinates(point1);
176 Point pt2 = CoordinateSystem.ToPlotCoordinates(point2);
177
178 Point pt = (pt1 - proj).Modulus() < (pt2 - proj).Modulus() ? pt1 : pt2;
179
180 plottedPoints.Add(pt);
181
182 string tag = (Tag == null || !target.UseUniqueTags) ? this.Tag : (this.Tag + "@" + i.ToString());
183
184 target.Save();
185 target.Translate(pt);
186 target.Scale(PointSize, PointSize);
187 SwarmPointElement.Plot(target, this.PresentationAttributes, tag);
188 target.Restore();
189 }
190 }
