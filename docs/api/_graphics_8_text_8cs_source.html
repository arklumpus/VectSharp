43 }
44 }

123 /// <summary>
124 /// Fill a text string along a <see cref="GraphicsPath"/>.
125 /// </summary>
126 /// <param name="path">The <see cref="GraphicsPath"/> along which the text will flow.</param>
127 /// <param name="text">The string to draw.</param>
128 /// <param name="font">The font with which to draw the text.</param>
129 /// <param name="fillColour">The <see cref="Brush"/> to use to fill the text.</param>
130 /// <param name="reference">The (relative) starting point on the path starting from which the text should be drawn (0 is the start of the path, 1 is the end of the path).</param>
131 /// <param name="anchor">The anchor in the text string that will correspond to the point specified by the <paramref name="reference"/>.</param>
132 /// <param name="textBaseline">The text baseline (determines which the position of the text in relation to the <paramref name="path"/>.</param>
133 /// <param name="tag">A tag to identify the filled text.</param>
134 public void FillTextOnPath(GraphicsPath path, string text, Font font, Brush fillColour, double reference = 0, TextAnchors anchor = TextAnchors.Left, TextBaselines textBaseline = TextBaselines.Top, string tag = null)
135 {
136 if (!string.IsNullOrEmpty(text))
137 {
138 if (font.Underline != null)
139 {
140 font = new Font(font.FontFamily, font.FontSize, false);
141 }
142
143 double currDelta = 0;
144 double pathLength = path.MeasureLength();
145
146 Font.DetailedFontMetrics fullMetrics = font.MeasureTextAdvanced(text);
147
148 switch (anchor)
149 {
150 case TextAnchors.Left:
151 break;
152 case TextAnchors.Center:
153 currDelta = -fullMetrics.Width * 0.5 / pathLength;
154 break;
155 case TextAnchors.Right:
156 currDelta = -fullMetrics.Width / pathLength;
157 break;
158 }
159
160 Point currentGlyphPlacementDelta = new Point();
161 Point currentGlyphAdvanceDelta = new Point();
162 Point nextGlyphPlacementDelta = new Point();
163 Point nextGlyphAdvanceDelta = new Point();
164
165 for (int i = 0; i < text.Length; i++)
166 {
167 string c = text.Substring(i, 1);
168
169 if (Font.EnableKerning && i < text.Length - 1)
170 {
171 currentGlyphPlacementDelta = nextGlyphPlacementDelta;
172 currentGlyphAdvanceDelta = nextGlyphAdvanceDelta;
173 nextGlyphAdvanceDelta = new Point();
174 nextGlyphPlacementDelta = new Point();
175
176 TrueTypeFile.PairKerning kerning = font.FontFamily.TrueTypeFile.Get1000EmKerning(text[i], text[i + 1]);
177
178 if (kerning != null)
179 {
180 currentGlyphPlacementDelta = new Point(currentGlyphPlacementDelta.X + kerning.Glyph1Placement.X, currentGlyphPlacementDelta.Y + kerning.Glyph1Placement.Y);
181 currentGlyphAdvanceDelta = new Point(currentGlyphAdvanceDelta.X + kerning.Glyph1Advance.X, currentGlyphAdvanceDelta.Y + kerning.Glyph1Advance.Y);
182
183 nextGlyphPlacementDelta = new Point(nextGlyphPlacementDelta.X + kerning.Glyph2Placement.X, nextGlyphPlacementDelta.Y + kerning.Glyph2Placement.Y);
184 nextGlyphAdvanceDelta = new Point(nextGlyphAdvanceDelta.X + kerning.Glyph2Advance.X, nextGlyphAdvanceDelta.Y + kerning.Glyph2Advance.Y);
185 }
186 }
187
188 Font.DetailedFontMetrics metrics = font.MeasureTextAdvanced(c);
189
190 Point origin = path.GetPointAtRelative(reference + currDelta + currentGlyphPlacementDelta.X * font.FontSize / 1000);
191
192 Point tangent = path.GetTangentAtRelative(reference + currDelta + currentGlyphPlacementDelta.X * font.FontSize / 1000 + (metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing) / pathLength * 0.5);
193
194 origin = new Point(origin.X - tangent.Y * currentGlyphPlacementDelta.Y * font.FontSize / 1000, origin.Y + tangent.X * currentGlyphPlacementDelta.Y * font.FontSize / 1000);
195
196 this.Save();
197
198 this.Translate(origin);
199 this.Rotate(Math.Atan2(tangent.Y, tangent.X));
200
201 string glyphTag = (!string.IsNullOrEmpty(tag) && UseUniqueTags) ? (tag + "@" + i.ToString()) : tag;
202
203 switch (textBaseline)
204 {
205 case TextBaselines.Top:
206 if (i > 0)
207 {
208 this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Top), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
209 }
210 else
211 {
212 this.FillText(new Point(0, fullMetrics.Top), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
213 }
214 break;
215 case TextBaselines.Baseline:
216 if (i > 0)
217 {
218 this.FillText(new Point(metrics.LeftSideBearing, 0), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
219 }
220 else
221 {
222 this.FillText(new Point(0, 0), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
223 }
224 break;
225 case TextBaselines.Bottom:
226 if (i > 0)
227 {
228 this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Bottom), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
229 }
230 else
231 {
232 this.FillText(new Point(0, fullMetrics.Bottom), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
233 }
234 break;
235 case TextBaselines.Middle:
236 if (i > 0)
237 {
238 this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Bottom + fullMetrics.Height / 2), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
239 }
240 else
241 {
242 this.FillText(new Point(0, fullMetrics.Bottom + fullMetrics.Height / 2), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
243 }
244 break;
245 }
246
247 this.Restore();
248
249 if (i > 0)
250 {
251 currDelta += (metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing + currentGlyphAdvanceDelta.X * font.FontSize / 1000) / pathLength;
252 }
253 else
254 {
255 currDelta += (metrics.Width + metrics.RightSideBearing + currentGlyphAdvanceDelta.X * font.FontSize / 1000) / pathLength;
256 }
257 }
258 }
259 }

403 /// <summary>
404 /// Fill a formatted text string.
405 /// </summary>
406 /// <param name="origin">The text origin. See <paramref name="textBaseline"/>.</param>
407 /// <param name="text">The <see cref="FormattedText"/> to draw.</param>
408 /// <param name="fillColour">The default <see cref="Brush"/> to use to fill the text. This can be overridden by each <paramref name="text"/> element.</param>
409 /// <param name="textBaseline">The text baseline (determines what the vertical component of <paramref name="origin"/> represents).</param>
410 /// <param name="tag">A tag to identify the filled text.</param>
411 public void FillText(Point origin, IEnumerable<FormattedText> text, Brush fillColour, TextBaselines textBaseline = TextBaselines.Top, string tag = null)
412 {
413 if (text != null)
414 {
415 List<FormattedText> enumeratedText = new List<FormattedText>();
416 List<Font.DetailedFontMetrics> allMetrics = new List<Font.DetailedFontMetrics>();
417
418 Font.DetailedFontMetrics fullMetrics = text.Measure(enumeratedText, allMetrics);
419
420 if (enumeratedText.Count > 0)
421 {
422 Point baselineOrigin = origin;
423
424 switch (textBaseline)
425 {
426 case TextBaselines.Baseline:
427 baselineOrigin = origin;
428 break;
429 case TextBaselines.Top:
430 baselineOrigin = new Point(origin.X, origin.Y + fullMetrics.Top);
431 break;
432 case TextBaselines.Bottom:
433 baselineOrigin = new Point(origin.X, origin.Y + fullMetrics.Bottom);
434 break;
435 case TextBaselines.Middle:
436 baselineOrigin = new Point(origin.X, origin.Y + fullMetrics.Top * 0.5 + fullMetrics.Bottom * 0.5);
437 break;
438 }
439
440 for (int i = 0; i < enumeratedText.Count; i++)
441 {
442 FormattedText txt = enumeratedText[i];
443
444 string spanTag = (!string.IsNullOrEmpty(tag) && UseUniqueTags) ? (tag + "@" + i.ToString()) : tag;
445
446 if (txt.Script == Script.Normal)
447 {
448 Font.DetailedFontMetrics metrics = allMetrics[i];
449
450 if (i > 0)
451 {
452 FillText(baselineOrigin.X + metrics.LeftSideBearing, baselineOrigin.Y, txt.Text, txt.Font, txt.Brush ?? fillColour, TextBaselines.Baseline, spanTag);
453 }
454 else
455 {
456 FillText(baselineOrigin, txt.Text, txt.Font, txt.Brush ?? fillColour, TextBaselines.Baseline, spanTag);
457 }
458
459 if (i > 0)
460 {
461 baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing, baselineOrigin.Y);
462 }
463 else
464 {
465 baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing, baselineOrigin.Y);
466 }
467 }
468 else
469 {
470 Font newFont = new Font(txt.Font.FontFamily, txt.Font.FontSize * 0.7, txt.Font.Underline);
471
472 Font.DetailedFontMetrics metrics = allMetrics[i];
473
474 if (i == 0)
475 {
476 baselineOrigin = new Point(baselineOrigin.X - metrics.LeftSideBearing, baselineOrigin.Y);
477 }
478
479 if (txt.Script == Script.Subscript)
480 {
481 FillText(baselineOrigin.X + metrics.LeftSideBearing, baselineOrigin.Y + txt.Font.FontSize * 0.14, txt.Text, newFont, txt.Brush ?? fillColour, TextBaselines.Baseline, spanTag);
482 }
483 else if (txt.Script == Script.Superscript)
484 {
485 FillText(baselineOrigin.X + metrics.LeftSideBearing, baselineOrigin.Y - txt.Font.FontSize * 0.33, txt.Text, newFont, txt.Brush ?? fillColour, TextBaselines.Baseline, spanTag);
486 }
487
488
489 if (i > 0)
490 {
491 baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing, baselineOrigin.Y);
492 }
493 else
494 {
495 baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing, baselineOrigin.Y);
496 }
497 }
498 }
499 }
500 }
