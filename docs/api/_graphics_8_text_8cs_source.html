221 /// <summary>
222 /// Fill a text string along a <see cref="GraphicsPath"/>.
223 /// </summary>
224 /// <param name="path">The <see cref="GraphicsPath"/> along which the text will flow.</param>
225 /// <param name="text">The string to draw.</param>
226 /// <param name="font">The font with which to draw the text.</param>
227 /// <param name="fillColour">The <see cref="Brush"/> to use to fill the text.</param>
228 /// <param name="reference">The (relative) starting point on the path starting from which the text should be drawn (0 is the start of the path, 1 is the end of the path).</param>
229 /// <param name="anchor">The anchor in the text string that will correspond to the point specified by the <paramref name="reference"/>.</param>
230 /// <param name="textBaseline">The text baseline (determines which the position of the text in relation to the <paramref name="path"/>.</param>
231 /// <param name="tag">A tag to identify the filled text.</param>
232 public void FillTextOnPath(GraphicsPath path, string text, Font font, Brush fillColour, double reference = 0, TextAnchors anchor = TextAnchors.Left, TextBaselines textBaseline = TextBaselines.Top, string tag = null)
233 {
234 if (!string.IsNullOrEmpty(text))
235 {
236 if (font.Underline != null)
237 {
238 font = new Font(font.FontFamily, font.FontSize, false);
239 }
240
241 double currDelta = 0;
242 double pathLength = path.MeasureLength();
243
244 Font.DetailedFontMetrics fullMetrics = font.MeasureTextAdvanced(text);
245
246 switch (anchor)
247 {
248 case TextAnchors.Left:
249 break;
250 case TextAnchors.Center:
251 currDelta = -fullMetrics.Width * 0.5 / pathLength;
252 break;
253 case TextAnchors.Right:
254 currDelta = -fullMetrics.Width / pathLength;
255 break;
256 }
257
258 Point currentGlyphPlacementDelta = new Point();
259 Point currentGlyphAdvanceDelta = new Point();
260 Point nextGlyphPlacementDelta = new Point();
261 Point nextGlyphAdvanceDelta = new Point();
262
263 for (int i = 0; i < text.Length; i++)
264 {
265 string c = text.Substring(i, 1);
266
267 if (Font.EnableKerning && i < text.Length - 1)
268 {
269 currentGlyphPlacementDelta = nextGlyphPlacementDelta;
270 currentGlyphAdvanceDelta = nextGlyphAdvanceDelta;
271 nextGlyphAdvanceDelta = new Point();
272 nextGlyphPlacementDelta = new Point();
273
274 TrueTypeFile.PairKerning kerning = font.FontFamily.TrueTypeFile.Get1000EmKerning(text[i], text[i + 1]);
275
276 if (kerning != null)
277 {
278 currentGlyphPlacementDelta = new Point(currentGlyphPlacementDelta.X + kerning.Glyph1Placement.X, currentGlyphPlacementDelta.Y + kerning.Glyph1Placement.Y);
279 currentGlyphAdvanceDelta = new Point(currentGlyphAdvanceDelta.X + kerning.Glyph1Advance.X, currentGlyphAdvanceDelta.Y + kerning.Glyph1Advance.Y);
280
281 nextGlyphPlacementDelta = new Point(nextGlyphPlacementDelta.X + kerning.Glyph2Placement.X, nextGlyphPlacementDelta.Y + kerning.Glyph2Placement.Y);
282 nextGlyphAdvanceDelta = new Point(nextGlyphAdvanceDelta.X + kerning.Glyph2Advance.X, nextGlyphAdvanceDelta.Y + kerning.Glyph2Advance.Y);
283 }
284 }
285
286 Font.DetailedFontMetrics metrics = font.MeasureTextAdvanced(c);
287
288 Point origin = path.GetPointAtRelative(reference + currDelta + currentGlyphPlacementDelta.X * font.FontSize / 1000);
289
290 Point tangent = path.GetTangentAtRelative(reference + currDelta + currentGlyphPlacementDelta.X * font.FontSize / 1000 + (metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing) / pathLength * 0.5);
291
292 origin = new Point(origin.X - tangent.Y * currentGlyphPlacementDelta.Y * font.FontSize / 1000, origin.Y + tangent.X * currentGlyphPlacementDelta.Y * font.FontSize / 1000);
293
294 this.Save();
295
296 this.Translate(origin);
297 this.Rotate(Math.Atan2(tangent.Y, tangent.X));
298
299 string glyphTag = (!string.IsNullOrEmpty(tag) && UseUniqueTags) ? (tag + "@" + i.ToString()) : tag;
300
301 switch (textBaseline)
302 {
303 case TextBaselines.Top:
304 if (i > 0)
305 {
306 this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Top), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
307 }
308 else
309 {
310 this.FillText(new Point(0, fullMetrics.Top), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
311 }
312 break;
313 case TextBaselines.Baseline:
314 if (i > 0)
315 {
316 this.FillText(new Point(metrics.LeftSideBearing, 0), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
317 }
318 else
319 {
320 this.FillText(new Point(0, 0), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
321 }
322 break;
323 case TextBaselines.Bottom:
324 if (i > 0)
325 {
326 this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Bottom), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
327 }
328 else
329 {
330 this.FillText(new Point(0, fullMetrics.Bottom), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
331 }
332 break;
333 case TextBaselines.Middle:
334 if (i > 0)
335 {
336 this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Bottom + fullMetrics.Height / 2), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
337 }
338 else
339 {
340 this.FillText(new Point(0, fullMetrics.Bottom + fullMetrics.Height / 2), c, font, fillColour, textBaseline: TextBaselines.Baseline, glyphTag);
341 }
342 break;
343 }
344
345 this.Restore();
346
347 if (i > 0)
348 {
349 currDelta += (metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing + currentGlyphAdvanceDelta.X * font.FontSize / 1000) / pathLength;
350 }
351 else
352 {
353 currDelta += (metrics.Width + metrics.RightSideBearing + currentGlyphAdvanceDelta.X * font.FontSize / 1000) / pathLength;
354 }
355 }
356 }
357 }
