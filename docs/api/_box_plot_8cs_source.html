115 /// <summary>
116 /// Create a new <see cref="BoxPlot"/> instance.
117 /// </summary>
118 /// <param name="position">The position of the centre (e.g., median or mean) of the box plot, in data space coordinates.</param>
119 /// <param name="direction">The direction along which the box plot is drawn, in data space coordinates.</param>
120 /// <param name="whisker1">The distance between the centre of the box plot and the first whisker, expressed as a multiple
121 /// of <paramref name="direction"/>.</param>

127 /// of <paramref name="direction"/>.</param>

129 public BoxPlot(IReadOnlyList<double> position, IReadOnlyList<double> direction, double whisker1, double box1, double box2, double whisker2, ICoordinateSystem<IReadOnlyList<double>> coordinateSystem)

140 /// <inheritdoc/>
141 public void Plot(Graphics target)
142 {
143 double[] perpDirection = new double[] { -Direction[1], Direction[0] };
144
145 Point whisker1 = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Whisker1, this.Position[1] + this.Direction[1] * this.Whisker1 });
146 Point whisker2 = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Whisker2, this.Position[1] + this.Direction[1] * this.Whisker2 });
147
148 Point whisker1Left = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Whisker1 + perpDirection[0] * this.Width * this.WhiskerWidth, this.Position[1] + this.Direction[1] * this.Whisker1 + perpDirection[1] * this.Width * this.WhiskerWidth });
149 Point box1Left = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Box1 + perpDirection[0] * this.Width, this.Position[1] + this.Direction[1] * this.Box1 + perpDirection[1] * this.Width });
150 Point box2Left = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Box2 + perpDirection[0] * this.Width, this.Position[1] + this.Direction[1] * this.Box2 + perpDirection[1] * this.Width });
151 Point whisker2Left = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Whisker2 + perpDirection[0] * this.Width * this.WhiskerWidth, this.Position[1] + this.Direction[1] * this.Whisker2 + perpDirection[1] * this.Width * this.WhiskerWidth });
152
153 Point whisker1Right = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Whisker1 - perpDirection[0] * this.Width * this.WhiskerWidth, this.Position[1] + this.Direction[1] * this.Whisker1 - perpDirection[1] * this.Width * this.WhiskerWidth });
154 Point box1Right = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Box1 - perpDirection[0] * this.Width, this.Position[1] + this.Direction[1] * this.Box1 - perpDirection[1] * this.Width });
155 Point box2Right = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Box2 - perpDirection[0] * this.Width, this.Position[1] + this.Direction[1] * this.Box2 - perpDirection[1] * this.Width });
156 Point whisker2Right = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.Whisker2 - perpDirection[0] * this.Width * this.WhiskerWidth, this.Position[1] + this.Direction[1] * this.Whisker2 - perpDirection[1] * this.Width * this.WhiskerWidth });
157
158 Point medianLeft = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + perpDirection[0] * this.Width, this.Position[1] + perpDirection[1] * this.Width });
159 Point medianRight = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] - perpDirection[0] * this.Width, this.Position[1] - perpDirection[1] * this.Width });
160
161 string whiskerTag = Tag;
162 string boxFillTag = Tag;
163 string boxStrokeTag = Tag;
164 string medianTag = Tag;
165 string medianSymbolTag = Tag;
166
167 if (!string.IsNullOrEmpty(Tag) && target.UseUniqueTags)
168 {
169 whiskerTag += "@whiskers";
170 boxStrokeTag += "@stroke";
171 medianTag += "@median";
172 medianSymbolTag += "@medianSymbol";
173 }
174
175 if (NotchSize != 0 && NotchWidth != 1)
176 {
177 medianLeft = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + perpDirection[0] * this.Width * this.NotchWidth, this.Position[1] + perpDirection[1] * this.Width * this.NotchWidth });
178 medianRight = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] - perpDirection[0] * this.Width * this.NotchWidth, this.Position[1] - perpDirection[1] * this.Width * this.NotchWidth });
179
180 Point notch2Left = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.NotchSize + perpDirection[0] * this.Width, this.Position[1] + this.Direction[1] * this.NotchSize + perpDirection[1] * this.Width });
181 Point notch1Left = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] - this.Direction[0] * this.NotchSize + perpDirection[0] * this.Width, this.Position[1] - this.Direction[1] * this.NotchSize + perpDirection[1] * this.Width });
182
183 Point notch2Right = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] + this.Direction[0] * this.NotchSize - perpDirection[0] * this.Width, this.Position[1] + this.Direction[1] * this.NotchSize - perpDirection[1] * this.Width });
184 Point notch1Right = CoordinateSystem.ToPlotCoordinates(new double[] { this.Position[0] - this.Direction[0] * this.NotchSize - perpDirection[0] * this.Width, this.Position[1] - this.Direction[1] * this.NotchSize - perpDirection[1] * this.Width });
185
186
187 GraphicsPath whiskers = new GraphicsPath();
188
189 whiskers.MoveTo(whisker1Left).LineTo(whisker1Right);
190 whiskers.MoveTo(whisker1).LineTo(whisker2);
191 whiskers.MoveTo(whisker2Left).LineTo(whisker2Right);
192
193 GraphicsPath box = new GraphicsPath().MoveTo(box1Left).LineTo(box1Right).LineTo(notch1Right).LineTo(medianRight).LineTo(notch2Right).LineTo(box2Right).LineTo(box2Left).LineTo(notch2Left).LineTo(medianLeft).LineTo(notch1Left).Close();
194 GraphicsPath median = new GraphicsPath().MoveTo(medianLeft).LineTo(medianRight);
195
196 if (WhiskersPresentationAttributes.Stroke != null)
197 {
198 target.StrokePath(whiskers, WhiskersPresentationAttributes.Stroke, WhiskersPresentationAttributes.LineWidth, WhiskersPresentationAttributes.LineCap, WhiskersPresentationAttributes.LineJoin, WhiskersPresentationAttributes.LineDash, whiskerTag);
199 }
200
201 if (BoxPresentationAttributes.Fill != null)
202 {
203 target.FillPath(box, BoxPresentationAttributes.Fill, boxFillTag);
204 }
205
206 if (BoxPresentationAttributes.Stroke != null)
207 {
208 target.StrokePath(box, BoxPresentationAttributes.Stroke, BoxPresentationAttributes.LineWidth, BoxPresentationAttributes.LineCap, BoxPresentationAttributes.LineJoin, BoxPresentationAttributes.LineDash, boxStrokeTag);
209 target.StrokePath(median, BoxPresentationAttributes.Stroke, BoxPresentationAttributes.LineWidth * 2, BoxPresentationAttributes.LineCap, BoxPresentationAttributes.LineJoin, BoxPresentationAttributes.LineDash, medianTag);
210 }
211 }
212 else
213 {
214 GraphicsPath whiskers = new GraphicsPath();
215
216 whiskers.MoveTo(whisker1Left).LineTo(whisker1Right);
217 whiskers.MoveTo(whisker1).LineTo(whisker2);
218 whiskers.MoveTo(whisker2Left).LineTo(whisker2Right);
219
220 GraphicsPath box = new GraphicsPath().MoveTo(box1Left).LineTo(box1Right).LineTo(box2Right).LineTo(box2Left).Close();
221 GraphicsPath median = new GraphicsPath().MoveTo(medianLeft).LineTo(medianRight);
222
223 if (WhiskersPresentationAttributes.Stroke != null)
224 {
225 target.StrokePath(whiskers, WhiskersPresentationAttributes.Stroke, WhiskersPresentationAttributes.LineWidth, WhiskersPresentationAttributes.LineCap, WhiskersPresentationAttributes.LineJoin, WhiskersPresentationAttributes.LineDash, whiskerTag);
226 }
227
228 if (BoxPresentationAttributes.Fill != null)
229 {
230 target.FillPath(box, BoxPresentationAttributes.Fill, boxFillTag);
231 }
232
233 if (BoxPresentationAttributes.Stroke != null)
234 {
235 target.StrokePath(box, BoxPresentationAttributes.Stroke, BoxPresentationAttributes.LineWidth, BoxPresentationAttributes.LineCap, BoxPresentationAttributes.LineJoin, BoxPresentationAttributes.LineDash, boxStrokeTag);
236 target.StrokePath(median, BoxPresentationAttributes.Stroke, BoxPresentationAttributes.LineWidth * 2, LineCaps.Butt, BoxPresentationAttributes.LineJoin, BoxPresentationAttributes.LineDash, medianTag);
237 }
238 }
239
240 if (CentreSymbolPresentationAttributes.Fill != null || CentreSymbolPresentationAttributes.Stroke != null)
241 {
242 double medianCircleRadius = Math.Min(Math.Sqrt((medianLeft.X - medianRight.X) * (medianLeft.X - medianRight.X) + (medianLeft.Y - medianRight.Y) * (medianLeft.Y - medianRight.Y)),
243 Math.Min(Math.Sqrt((box1Left.X - box2Left.X) * (box1Left.X - box2Left.X) + (box1Left.Y - box2Left.Y) * (box1Left.Y - box2Left.Y)),
244 Math.Sqrt((box1Right.X - box2Right.X) * (box1Right.X - box2Right.X) + (box1Right.Y - box2Right.Y) * (box1Right.Y - box2Right.Y)))) * 0.45;
245
246 target.Save();
247 target.Translate((medianLeft.X + medianRight.X) * 0.5, (medianLeft.Y + medianRight.Y) * 0.5);
248 target.Scale(medianCircleRadius, medianCircleRadius);
249
250 CentreSymbol.Plot(target, CentreSymbolPresentationAttributes, medianSymbolTag);
251
252 target.Restore();
253 }
254 }
255 }
