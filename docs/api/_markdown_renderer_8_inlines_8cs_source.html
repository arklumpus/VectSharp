33 private string RenderInline(Inline inline, ref MarkdownContext context, ref Graphics graphics, NewPageAction newPageAction)
34 {
35 this.OnInlineRendering(ref context, ref graphics, ref inline);
36
37 if (inline != null)
38 {
39 string returnValue;
40 HtmlAttributes attributes = inline.TryGetAttributes();
41
42 if (attributes != null && !string.IsNullOrEmpty(attributes.Id))
43 {
44 Point cursor = context.Cursor;
45 RenderHTMLBlock("<a name=\"" + attributes.Id + "\"></a>", true, ref context, ref graphics, newPageAction, false, false);
46 context.Cursor = cursor;
47 }
48
49 if (inline is EmojiInline emoji)
50 {
51 int headingLevel = 0;
52 Block currBlock = emoji.Parent?.ParentBlock;
53 while (currBlock != null)
54 {
55 if (currBlock is HeadingBlock heading)
56 {
57 headingLevel = heading.Level;
58 break;
59 }
60 currBlock = currBlock.Parent;
61 }
62
63 inline = new LinkInline(emoji.Content.ToString() + "_heading:" + headingLevel.ToString(), "") { IsImage = true };
64 }
65
66 if (inline is LeafInline)
67 {
68 if (inline is AutolinkInline autoLink)
69 {
70 LinkInline link = new LinkInline((autoLink.IsEmail ? "mailto:" : "") + autoLink.Url, "");
71 link.AppendChild(new LiteralInline(autoLink.Url));
72
73 returnValue = RenderLinkInline(link, ref context, ref graphics, newPageAction);
74 }
75 else if (inline is CodeInline code)
76 {
77 returnValue = RenderCodeInline(code, ref context, ref graphics, newPageAction);
78 }
79 else if (inline is HtmlEntityInline htmlEntity)
80 {
81 returnValue = RenderLiteralInline(new LiteralInline(htmlEntity.Transcoded), ref context, ref graphics, newPageAction);
82 }
83 else if (inline is HtmlInline html)
84 {
85 RenderHTMLBlock(html.Tag, true, ref context, ref graphics, newPageAction, true, true);
86 returnValue = "";
87 }
88 else if (inline is LineBreakInline lineBreak)
89 {
90 RenderLineBreakInline(lineBreak.IsHard, false, ref context, ref graphics, newPageAction);
91 returnValue = "\n";
92 }
93 else if (inline is LiteralInline literal)
94 {
95 returnValue = RenderLiteralInline(literal, ref context, ref graphics, newPageAction);
96 }
97 else if (inline is Markdig.Extensions.Mathematics.MathInline math)
98 {
99 byte[] svgData;
100
101 using (MemoryStream ms = new MemoryStream())
102 {
103 if (math.DelimiterCount == 1)
104 {
105 MathPainter.LineStyle = global::CSharpMath.Atom.LineStyle.Text;
106 }
107 else
108 {
109 MathPainter.LineStyle = global::CSharpMath.Atom.LineStyle.Display;
110 }
111
112 MathPainter.FontSize = (float)(context.Font.FontSize * MathFontScalingFactor / ImageMultiplier);
113
114 MathPainter.LaTeX = math.Content.ToString();
115 MathPainter.TextColor = context.Colour;
116 Page pag = MathPainter.DrawToPage();
117 VectSharp.SVG.SVGContextInterpreter.SaveAsSVG(pag, ms);
118 svgData = ms.ToArray();
119 }
120
121 string imageUri = "<img src=\"data:image/svg+xml;base64," + Convert.ToBase64String(svgData) + "\">";
122 RenderHTMLBlock(imageUri, true, ref context, ref graphics, newPageAction, true, true);
123 returnValue = math.Content.ToString();
124 }
125 else if (inline is Markdig.Extensions.SmartyPants.SmartyPant smartyPant)
126 {
127 returnValue = smartyPant.Type switch
128 {
129 Markdig.Extensions.SmartyPants.SmartyPantType.LeftDoubleQuote => RenderLiteralInline(new LiteralInline("“"), ref context, ref graphics, newPageAction),
130 Markdig.Extensions.SmartyPants.SmartyPantType.RightDoubleQuote => RenderLiteralInline(new LiteralInline("”"), ref context, ref graphics, newPageAction),
131 Markdig.Extensions.SmartyPants.SmartyPantType.LeftQuote => RenderLiteralInline(new LiteralInline("‘"), ref context, ref graphics, newPageAction),
132 Markdig.Extensions.SmartyPants.SmartyPantType.RightQuote => RenderLiteralInline(new LiteralInline("’"), ref context, ref graphics, newPageAction),
133 Markdig.Extensions.SmartyPants.SmartyPantType.Dash2 => RenderLiteralInline(new LiteralInline("–"), ref context, ref graphics, newPageAction),
134 Markdig.Extensions.SmartyPants.SmartyPantType.Dash3 => RenderLiteralInline(new LiteralInline("—"), ref context, ref graphics, newPageAction),
135 Markdig.Extensions.SmartyPants.SmartyPantType.DoubleQuote => RenderLiteralInline(new LiteralInline("\""), ref context, ref graphics, newPageAction),
136 Markdig.Extensions.SmartyPants.SmartyPantType.Ellipsis => RenderLiteralInline(new LiteralInline("…"), ref context, ref graphics, newPageAction),
137 Markdig.Extensions.SmartyPants.SmartyPantType.LeftAngleQuote => RenderLiteralInline(new LiteralInline("«"), ref context, ref graphics, newPageAction),
138 Markdig.Extensions.SmartyPants.SmartyPantType.Quote => RenderLiteralInline(new LiteralInline("'"), ref context, ref graphics, newPageAction),
139 Markdig.Extensions.SmartyPants.SmartyPantType.RightAngleQuote => RenderLiteralInline(new LiteralInline("»"), ref context, ref graphics, newPageAction),
140 _ => "",
141 };
142 }
143 else if (inline is Markdig.Extensions.TaskLists.TaskList)
144 {
145 // Nothing to render here (the checkbox has already been rendered)
146 returnValue = "";
147 }
148 else
149 {
150 returnValue = "";
151 }
152 }
153 else if (inline is ContainerInline)
154 {
155 if (inline is DelimiterInline)
156 {
157 // Nothing to render here
158 returnValue = "";
159 }
160 else if (inline is EmphasisInline emphasis)
161 {
162 returnValue = RenderEmphasisInline(emphasis, ref context, ref graphics, newPageAction);
163 }
164 else if (inline is LinkInline link)
165 {
166 returnValue = RenderLinkInline(link, ref context, ref graphics, newPageAction);
167 }
168 else
169 {
170 returnValue = "";
171 }
172 }
173 else
174 {
175 returnValue = "";
176 }
177
178 this.OnInlineRendered(ref context, ref graphics, inline);
179 return returnValue;
180 }
181 else
182 {
183 return "";
184 }

188 {
189 if (isHard)
190 {
191 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);

194

196
197 context.Cursor = new Point(minX, context.Cursor.Y);

203 }

207 context.Cursor = new Point(context.Cursor.X + spaceWidth, context.Cursor.Y);
208 }

211 private string RenderLiteralInline(LiteralInline literal, ref MarkdownContext context, ref Graphics graphics, NewPageAction newPageAction)
212 {
213 string text = literal.Content.ToString();
214
215 double spaceWidth = context.Font.FontFamily.TrueTypeFile.Get1000EmGlyphWidth(' ') / 1000.0 * context.Font.FontSize;
216
217 List<Word> words = Word.GetWords(text, context.Font, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X).ToList();
218
219 double underlineStart = context.Cursor.X;
220 double underlineEnd = context.Cursor.X;
221
222 double currLineMaxX = context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X);
223
224 bool ignoreNextWhitespace = false;
225
226 bool broken = false;
227
228 for (int i = 0; i < words.Count; i++)
229 {
230 Word w = words[i];
231
232 if (!string.IsNullOrEmpty(w.Text))
233 {
234 if (!ignoreNextWhitespace)
235 {
236 context.Cursor = new Point(context.Cursor.X + spaceWidth * w.WhitespaceCount * (w.PrecedingWhitespace == '\t' ? 4 : 1), context.Cursor.Y);
237 }
238 else
239 {
240 ignoreNextWhitespace = false;
241 }
242
243 Font.DetailedFontMetrics wordMetrics = w.Metrics;
244
245 double finalX = context.Cursor.X + wordMetrics.Width + wordMetrics.RightSideBearing + wordMetrics.LeftSideBearing;
246
247 if (finalX <= currLineMaxX || broken)
248 {
249 context.CurrentLine.Fragments.Add(new TextFragment(new Point(context.Cursor.X + wordMetrics.LeftSideBearing, context.Cursor.Y), w.Text, context.Font, context.Colour, context.Tag));
250 context.Cursor = new Point(context.Cursor.X + wordMetrics.Width + wordMetrics.RightSideBearing + wordMetrics.LeftSideBearing, context.Cursor.Y);
251
252 broken = false;
253
254 if (context.Underline || context.StrikeThrough)
255 {
256 underlineEnd = context.Cursor.X;
257 }
258 }
259 else
260 {
261 if (context.Underline && underlineStart != underlineEnd)
262 {
263 context.CurrentLine.Fragments.Add(new UnderlineFragment(new Point(underlineStart, context.Cursor.Y + context.Font.FontSize * 0.2), new Point(underlineEnd, context.Cursor.Y + context.Font.FontSize * 0.2), context.Colour, context.Font.FontSize * (context.Font.FontFamily.IsBold ? this.BoldUnderlineThickness : this.UnderlineThickness), context.Tag));
264 }
265 else if (context.StrikeThrough && underlineStart != underlineEnd)
266 {
267 context.CurrentLine.Fragments.Add(new UnderlineFragment(new Point(underlineStart, context.Cursor.Y - context.Font.Ascent * 0.5 - context.Font.Descent * 0.5), new Point(underlineEnd, context.Cursor.Y - context.Font.Ascent * 0.5 - context.Font.Descent * 0.5), context.Colour, context.Font.FontSize * (context.Font.FontFamily.IsBold ? this.BoldUnderlineThickness : this.UnderlineThickness), context.Tag));
268 }
269
270 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);
271
272
273
274 context.CurrentLine = new Line(context.Font.Ascent);
275 context.Cursor = new Point(0, context.Cursor.Y - context.Font.Descent + SpaceAfterLine * context.Font.FontSize + context.Font.Ascent);
276 currLineMaxX = context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X);
277
278 double minX = context.GetMinX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent);
279
280 context.Cursor = new Point(minX, context.Cursor.Y);
281
282 underlineStart = minX;
283 underlineEnd = minX;
284
285 i--;
286 ignoreNextWhitespace = true;
287 broken = true;
288 }
289 }
290 else
291 {
292 context.Cursor = new Point(context.Cursor.X + spaceWidth * w.WhitespaceCount * (w.PrecedingWhitespace == '\t' ? 4 : 1), context.Cursor.Y);
293 }
294 }
295
296 if (context.Underline && underlineStart != underlineEnd)
297 {
298 context.CurrentLine.Fragments.Add(new UnderlineFragment(new Point(underlineStart, context.Cursor.Y + context.Font.FontSize * 0.2), new Point(underlineEnd, context.Cursor.Y + context.Font.FontSize * 0.2), context.Colour, context.Font.FontSize * (context.Font.FontFamily.IsBold ? this.BoldUnderlineThickness : this.UnderlineThickness), context.Tag));
299 }
300 else if (context.StrikeThrough && underlineStart != underlineEnd)
301 {
302 context.CurrentLine.Fragments.Add(new UnderlineFragment(new Point(underlineStart, context.Cursor.Y - context.Font.Ascent * 0.5 - context.Font.Descent * 0.5), new Point(underlineEnd, context.Cursor.Y - context.Font.Ascent * 0.5 - context.Font.Descent * 0.5), context.Colour, context.Font.FontSize * (context.Font.FontFamily.IsBold ? this.BoldUnderlineThickness : this.UnderlineThickness), context.Tag));
303 }
304
305 return text;
306 }

308 private string RenderCodeInline(CodeInline code, ref MarkdownContext context, ref Graphics graphics, NewPageAction newPageAction)
309 {
310 MarkdownContext prevContext = context.Clone();
311
312 context.Font = new Font(this.CodeFont, context.Font.FontSize);
313
314 double spaceWidth = context.Font.FontFamily.TrueTypeFile.Get1000EmGlyphWidth(' ') / 1000.0 * context.Font.FontSize;

351 context.Cursor = new Point(context.Cursor.X + wordMetrics.Width + wordMetrics.RightSideBearing + wordMetrics.LeftSideBearing, context.Cursor.Y);
352
353 broken = false;
354
355 if (context.Underline || context.StrikeThrough)
356 {
357 underlineEnd = context.Cursor.X;
358 }
359 }
360 else
361 {

363 {

365 }
366 else if (context.StrikeThrough && underlineStart != underlineEnd)
367 {
368 context.CurrentLine.Fragments.Add(new UnderlineFragment(new Point(underlineStart, context.Cursor.Y - context.Font.Ascent * 0.5 - context.Font.Descent * 0.5), new Point(underlineEnd, context.Cursor.Y - context.Font.Ascent * 0.5 - context.Font.Descent * 0.5), context.Colour, context.Font.FontSize * (context.Font.FontFamily.IsBold ? this.BoldUnderlineThickness : this.UnderlineThickness), context.Tag));
369 }

371 context.CurrentLine.Render(ref graphics, ref context, newPageAction, this.PageSize.Height - this.Margins.Bottom - context.Translation.Y - context.MarginBottomRight.Y, context.GetMaxX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent, this.PageSize.Width - this.Margins.Right - context.Translation.X - context.MarginBottomRight.X), this);
372
373 context.CurrentLine = new Line(prevContext.Font.Ascent);
374
375 context.Cursor = new Point(0, context.Cursor.Y - prevContext.Font.Descent + SpaceAfterLine * prevContext.Font.FontSize + prevContext.Font.Ascent);
376
377
378 double minX = context.GetMinX(context.Cursor.Y - context.Font.Ascent, context.Cursor.Y - context.Font.Descent);
379
380 context.Cursor = new Point(minX, context.Cursor.Y);
381
382 if (i == 0)
383 {
384 context.Cursor = new Point(context.Cursor.X + this.CodeInlineMargin * context.Font.FontSize, context.Cursor.Y);

393 }

398 {
399 context.CurrentLine.Fragments.Add(new UnderlineFragment(new Point(underlineStart, context.Cursor.Y + context.Font.FontSize * 0.2), new Point(underlineEnd, context.Cursor.Y + context.Font.FontSize * 0.2), context.Colour, context.Font.FontSize * (context.Font.FontFamily.IsBold ? this.BoldUnderlineThickness : this.UnderlineThickness), context.Tag));
400 }

404 }

407
408 context.Cursor = new Point(context.Cursor.X + this.CodeInlineMargin * context.Font.FontSize, context.Cursor.Y);
409
410 prevContext.Cursor = context.Cursor;
411 prevContext.BottomRight = context.BottomRight;
412 prevContext.CurrentPage = context.CurrentPage;
413 prevContext.CurrentLine = context.CurrentLine;
414
415 context = prevContext;
416 return text;

598 case '~':

603 translationToUndo = new Point(translationToUndo.X, translationToUndo.Y + context.Font.FontSize * this.SubscriptShift);
604 context.Font = new Font(context.Font.FontFamily, context.Font.FontSize * this.SubSuperscriptFontSize);
605 }

610 }

616 context.Cursor = new Point(context.Cursor.X, context.Cursor.Y - context.Font.FontSize * this.SuperscriptShift);
617 translationToUndo = new Point(translationToUndo.X, translationToUndo.Y - context.Font.FontSize * this.SuperscriptShift);
618 context.Font = new Font(context.Font.FontFamily, context.Font.FontSize * this.SubSuperscriptFontSize);
619 }
620 break;
621 case '+':
622 context.Colour = this.InsertedColour;
623 break;
624 case '=':
625 context.Colour = this.MarkedColour;
626 break;
627 }
628
629 StringBuilder text = new StringBuilder();
630
631 foreach (Inline innerInline in emphasis)
632 {
633 text.Append(RenderInline(innerInline, ref context, ref graphics, newPageAction));
634 }
635
636 if (translationToUndo.X != 0 || translationToUndo.Y != 0)
637 {
638 context.Cursor = new Point(context.Cursor.X - translationToUndo.X, context.Cursor.Y - translationToUndo.Y);
639 }
640
641
642 prevContext.Cursor = context.Cursor;
643 prevContext.BottomRight = context.BottomRight;
644 prevContext.CurrentPage = context.CurrentPage;
645 prevContext.CurrentLine = context.CurrentLine;
646
647 context = prevContext;
648
649 return text.ToString();

652 private string RenderLinkInline(LinkInline link, ref MarkdownContext context, ref Graphics graphics, NewPageAction newPageAction)
653 {
654 if (!link.IsImage)
655 {
656 MarkdownContext prevContext = context.Clone();
657
658 context.Colour = this.LinkColour;
659 context.Underline = true;
660 string tag = Guid.NewGuid().ToString("N");
661
662 if (!link.Url.StartsWith("#"))
663 {
664 if (Uri.TryCreate(this.BaseLinkUri, link.Url, out Uri uri))
665 {
666 context.LinkDestinations[tag] = this.LinkUriResolver(uri.ToString());
667 }
668 else
669 {
670 context.LinkDestinations[tag] = this.LinkUriResolver(link.Url);
671 }
672 }
673 else
674 {
675 if (!context.InternalAnchors.TryGetValue(link.Url, out string anchor))
676 {
677 anchor = Guid.NewGuid().ToString("N");
678 context.InternalAnchors[link.Url] = anchor;
679 }
680
681 context.LinkDestinations[tag] = "#" + anchor;
682 }
683
684 context.Tag = tag;
685
686 StringBuilder text = new StringBuilder();
687
688 foreach (Inline innerInline in link)
689 {
690 text.Append(RenderInline(innerInline, ref context, ref graphics, newPageAction));
691 }
692
693 prevContext.Cursor = context.Cursor;
694 prevContext.BottomRight = context.BottomRight;
695 prevContext.CurrentPage = context.CurrentPage;
696 prevContext.CurrentLine = context.CurrentLine;
697
698 context = prevContext;
699
700 return text.ToString();
701 }
702 else
703 {
704 HtmlTag tag = HtmlTag.Parse("<img src=\"" + link.Url.Replace("\"", "\\\"") + "\">").FirstOrDefault();
705
706 RenderHTMLImage(tag, true, ref context, ref graphics, newPageAction);
707
708 return "";
709 }
710 }
