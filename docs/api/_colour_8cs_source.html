223 /// <summary>
224 /// Convert a CSS colour string into a <see cref="Colour"/> object.
225 /// </summary>
226 /// <param name="cssString">The CSS colour string. In addition to 148 standard colour names (case-insensitive), #RGB, #RGBA, #RRGGBB and #RRGGBBAA hex strings and rgb(r, g, b) and rgba(r, g, b, a) functional colour notations are supported.</param>
227 /// <returns></returns>
228 public static Colour? FromCSSString(string cssString)
229 {
230 if (cssString.StartsWith("#"))
231 {
232 cssString = cssString.Substring(1);
233
234 try
235 {
236 if (cssString.Length == 3)
237 {
238 byte r = byte.Parse(cssString.Substring(0, 1) + cssString.Substring(0, 1), System.Globalization.NumberStyles.HexNumber);
239 byte g = byte.Parse(cssString.Substring(1, 1) + cssString.Substring(1, 1), System.Globalization.NumberStyles.HexNumber);
240 byte b = byte.Parse(cssString.Substring(2, 1) + cssString.Substring(2, 1), System.Globalization.NumberStyles.HexNumber);
241
242 return Colour.FromRgb(r, g, b);
243 }
244 else if (cssString.Length == 4)
245 {
246 byte r = byte.Parse(cssString.Substring(0, 1) + cssString.Substring(0, 1), System.Globalization.NumberStyles.HexNumber);
247 byte g = byte.Parse(cssString.Substring(1, 1) + cssString.Substring(1, 1), System.Globalization.NumberStyles.HexNumber);
248 byte b = byte.Parse(cssString.Substring(2, 1) + cssString.Substring(2, 1), System.Globalization.NumberStyles.HexNumber);
249 byte a = byte.Parse(cssString.Substring(3, 1) + cssString.Substring(3, 1), System.Globalization.NumberStyles.HexNumber);
250
251 return Colour.FromRgba(r, g, b, a);
252 }
253 else if (cssString.Length == 6)
254 {
255 byte r = byte.Parse(cssString.Substring(0, 2), System.Globalization.NumberStyles.HexNumber);
256 byte g = byte.Parse(cssString.Substring(2, 2), System.Globalization.NumberStyles.HexNumber);
257 byte b = byte.Parse(cssString.Substring(4, 2), System.Globalization.NumberStyles.HexNumber);
258
259 return Colour.FromRgb(r, g, b);
260 }
261 else if (cssString.Length == 8)
262 {
263 byte r = byte.Parse(cssString.Substring(0, 2), System.Globalization.NumberStyles.HexNumber);
264 byte g = byte.Parse(cssString.Substring(2, 2), System.Globalization.NumberStyles.HexNumber);
265 byte b = byte.Parse(cssString.Substring(4, 2), System.Globalization.NumberStyles.HexNumber);
266 byte a = byte.Parse(cssString.Substring(6, 2), System.Globalization.NumberStyles.HexNumber);
267
268 return Colour.FromRgba(r, g, b, a);
269 }
270 else
271 {
272 return null;
273 }
274 }
275 catch
276 {
277 return null;
278 }
279 }
280 else if (cssString.StartsWith("rgb(") || cssString.StartsWith("rgba("))
281 {
282 try
283 {
284 cssString = cssString.Substring(cssString.IndexOf("(") + 1).Replace(")", "").Replace(" ", "");
285 string[] splitCssString = cssString.Split(',');
286
287 double R = ParseColourValueOrPercentage(splitCssString[0]);
288 double G = ParseColourValueOrPercentage(splitCssString[1]);
289 double B = ParseColourValueOrPercentage(splitCssString[2]);
290
291 double A = 1;
292
293 if (splitCssString.Length == 4)
294 {
295 A = double.Parse(splitCssString[3], System.Globalization.CultureInfo.InvariantCulture);
296 }
297
298 return Colour.FromRgba(R, G, B, A);
299 }
300 catch
301 {
302 return null;
303 }
304 }
305 else
306 {
307 if (StandardColours.TryGetValue(cssString, out Colour tbr))
308 {
309 return tbr;
310 }
311 else
312 {
313 return null;
314 }
315 }
316 }

378 /// <summary>
379 /// Converts a <see cref="Colour"/> to the CIE XYZ colour space.
380 /// </summary>
381 /// <returns>A <see cref="ValueTuple"/> containing the X, Y and Z components of the <see cref="Colour"/>.</returns>
382 public (double X, double Y, double Z) ToXYZ()
383 {
384 double r, g, b;
385
386 if (R <= 0.04045)
387 {
388 r = 25 * R / 323;
389 }
390 else
391 {
392 r = Math.Pow((200 * R + 11) / 211, 2.4);
393 }
394
395 if (G <= 0.04045)
396 {
397 g = 25 * G / 323;
398 }
399 else
400 {
401 g = Math.Pow((200 * G + 11) / 211, 2.4);
402 }
403
404 if (B <= 0.04045)
405 {
406 b = 25 * B / 323;
407 }
408 else
409 {
410 b = Math.Pow((200 * B + 11) / 211, 2.4);
411 }
412
413 double x = 0.41239080 * r + 0.35758434 * g + 0.18048079 * b;
414 double y = 0.21263901 * r + 0.71516868 * g + 0.07219232 * b;
415 double z = 0.01933082 * r + 0.11919478 * g + 0.95053215 * b;
416
417 return (x, y, z);
418 }

420 /// <summary>
421 /// Creates a <see cref="Colour"/> from CIE XYZ coordinates.
422 /// </summary>
423 /// <param name="x">The X coordinate.</param>
424 /// <param name="y">The Y coordinate.</param>
425 /// <param name="z">The Z coordinate.</param>
426 /// <returns>An sRGB <see cref="Colour"/> created from the specified components.</returns>
427 public static Colour FromXYZ(double x, double y, double z)
428 {
429 double r = +3.24096994 * x - 1.53738318 * y - 0.49861076 * z;
430 double g = -0.96924364 * x + 1.8759675 * y + 0.04155506 * z;
431 double b = 0.05563008 * x - 0.20397696 * y + 1.05697151 * z;
432
433 if (r <= 0.0031308)
434 {
435 r = 323 * r / 25;
436 }
437 else
438 {
439 r = (211 * Math.Pow(r, 1 / 2.4) - 11) / 200;
440 }
441
442 if (g <= 0.0031308)
443 {
444 g = 323 * g / 25;
445 }
446 else
447 {
448 g = (211 * Math.Pow(g, 1 / 2.4) - 11) / 200;
449 }
450
451 if (b <= 0.0031308)
452 {
453 b = 323 * b / 25;
454 }
455 else
456 {
457 b = (211 * Math.Pow(b, 1 / 2.4) - 11) / 200;
458 }
459
460 r = Math.Min(Math.Max(0, r), 1);
461 g = Math.Min(Math.Max(0, g), 1);
462 b = Math.Min(Math.Max(0, b), 1);
463
464 return Colour.FromRgb(r, g, b);
465 }

467 /// <summary>
468 /// Converts a <see cref="Colour"/> to the CIE Lab colour space (under Illuminant D65).
469 /// </summary>
470 /// <returns>A <see cref="ValueType"/> containing the L*, a* and b* components of the <see cref="Colour"/>.</returns>
471 public (double L, double a, double b) ToLab()
472 {
473 double f(double t)
474 {
475 const double d = 6.0 / 29;
476
477 if (t > d * d * d)
478 {
479 return Math.Pow(t, 1.0 / 3);
480 }
481 else
482 {
483 return t / (3 * d * d) + 4.0 / 29;
484 }
485 }
486
487 const double xN = 0.950489;
488 const double yN = 1;
489 const double zN = 1.088840;
490
491 (double x, double y, double z) = this.ToXYZ();
492
493 double fY = f(y / yN);
494
495 double l = 1.16 * fY - 0.16;
496 double a = 5 * (f(x / xN) - fY);
497 double b = 2 * (fY - f(z / zN));
498
499 return (l, a, b);
500 }

502 /// <summary>
503 /// Creates a <see cref="Colour"/> from CIE Lab coordinates (under Illuminant D65).
504 /// </summary>
505 /// <param name="L">The L* component.</param>
506 /// <param name="a">The a* component.</param>
507 /// <param name="b">The b* component.</param>
508 /// <returns>An sRGB <see cref="Colour"/> created from the specified components.</returns>
509 public static Colour FromLab(double L, double a, double b)
510 {
511 double f(double t)
512 {
513 const double d = 6.0 / 29;
514
515 if (t > d)
516 {
517 return t * t * t;
518 }
519 else
520 {
521 return 3 * d * d * (t - 4.0 / 29);
522 }
523 }
524
525 const double xN = 0.950489;
526 const double yN = 1;
527 const double zN = 1.088840;
528
529 double x = xN * f((L + 0.16) / 1.16 + a / 5);
530 double y = yN * f((L + 0.16) / 1.16);
531 double z = zN * f((L + 0.16) / 1.16 - b / 2);
532
533 return Colour.FromXYZ(x, y, z);
534 }

536 /// <summary>
537 /// Converts a <see cref="Colour"/> to the HSL colour space.
538 /// </summary>
539 /// <returns>A <see cref="ValueType"/> containing the H, S and L components of the <see cref="Colour"/>. Each component has range [0, 1].</returns>
540 public (double H, double S, double L) ToHSL()
541 {
542 double xMax = Math.Max(Math.Max(R, G), B);
543 double xMin = Math.Min(Math.Min(R, G), B);
544
545 double l = (xMax + xMin) * 0.5;
546
547 double h;
548
549 if (xMax == xMin)
550 {
551 h = 0;
552 }
553 else if (xMax == R)
554 {
555 h = (G - B) / (xMax - xMin) / 6;
556 }
557 else if (xMax == G)
558 {
559 h = (2 + (B - R) / (xMax - xMin)) / 6;
560 }
561 else
562 {
563 h = (4 + (R - G) / (xMax - xMin)) / 6;
564 }
565
566 double s;
567
568 if (l == 0 || l == 1)
569 {
570 s = 0;
571 }
572 else
573 {
574 s = (xMax - l) / Math.Min(l, 1 - l);
575 }
576
577 return (h, s, l);
578 }

580 /// <summary>
581 /// Creates a <see cref="Colour"/> from HSL coordinates.
582 /// </summary>
583 /// <param name="h">The H component. Should be in range [0, 1].</param>
584 /// <param name="s">The S component. Should be in range [0, 1].</param>
585 /// <param name="l">The L component. Should be in range [0, 1].</param>
586 /// <returns>A <see cref="Colour"/> created from the specified components.</returns>
587 public static Colour FromHSL(double h, double s, double l)
588 {
589 double c = (1 - Math.Abs(2 * l - 1)) * s;
590
591 double hp = h * 6;
592
593 double x = c * (1 - Math.Abs((hp % 2) - 1));
594
595 double r1, g1, b1;
596
597 if (hp <= 1)
598 {
599 r1 = c;
600 g1 = x;
601 b1 = 0;
602 }
603 else if (hp <= 2)
604 {
605 r1 = x;
606 g1 = c;
607 b1 = 0;
608 }
609 else if (hp <= 3)
610 {
611 r1 = 0;
612 g1 = c;
613 b1 = x;
614 }
615 else if (hp <= 4)
616 {
617 r1 = 0;
618 g1 = x;
619 b1 = c;
620 }
621 else if (hp <= 5)
622 {
623 r1 = x;
624 g1 = 0;
625 b1 = c;
626 }
627 else if (hp <= 6)
628 {
629 r1 = c;
630 g1 = 0;
631 b1 = x;
632 }
633 else
634 {
635 r1 = 0;
636 g1 = 0;
637 b1 = 0;
638 }
639
640 double m = l - c / 2;
641
642 return Colour.FromRgb(r1 + m, g1 + m, b1 + m);
643 }
