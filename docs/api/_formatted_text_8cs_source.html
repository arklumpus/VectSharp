95 /// <summary>
96 /// Parse the formatting information contained in a text string into a collection of <see cref="FormattedText"/> objects.
97 /// </summary>
98 /// <param name="text">
99 /// The string containing formatting information. Format information is specified using HTML-like tags:
100 /// <list type="bullet">
101 /// <item><c>&lt;b&gt;&lt;/b&gt;</c> or <c>&lt;strong&gt;&lt;/strong&gt;</c> are used for bold text;</item>
102 /// <item><c>&lt;i&gt;&lt;/i&gt;</c> or <c>&lt;em&gt;&lt;/em&gt;</c> are used for text in italics;</item>
103 /// <item><c>&lt;u&gt;&lt;/u&gt;</c></item> are used for underlined text;
104 /// <item><c>&lt;sup&gt;&lt;/sup&gt;</c> and <c>&lt;sub&gt;&lt;/sub&gt;</c> are used, respectively, for superscript and subscript text;</item>
105 /// <item><c>&lt;#COLOUR&gt;&lt;/#&gt;</c> is used to specify the colour of the text, where <c>COLOUR</c> is a CSS colour string (e.g. <c>&lt;#red&gt;</c>, <c>&lt;#0080FF&gt;</c>, or <c>&lt;#rgba(128, 80, 52, 0.5)&gt;</c>).</item>
106 /// </list></param>
107 /// <param name="normalFont">The font that will be used for text that is neither bold nor italic.</param>
108 /// <param name="boldFont">The font that will be used for text that is bold. Note that this does not necessarily have to be a bold font; this is just the font that is applied to text contained within <c>&lt;b&gt;&lt;/b&gt;</c> tags.</param>
109 /// <param name="italicFont">The font that will be used for text that is in italics. Note that this does not necessarily have to be an italic font; this is just the font that is applied to text contained within <c>&lt;i&gt;&lt;/i&gt;</c> tags.</param>
110 /// <param name="boldItalicFont">The font that will be used for text that is both in bold and in italics.</param>
111 /// <param name="defaultBrush">The default <see cref="Brush"/> that will be used for text runs that do not specify a colour. If this is <see langword="null"/>, the default <see cref="Brush"/> will be the one specified in the painting call.</param>
112 /// <param name="spacing">The text spacing parameters used to draw the text.</param>
113 /// <returns>A lazy collection of <see cref="FormattedText"/> objects. Note that every enumeration of this collection causes the text to be parsed again; if you need to enumerate this collection more than once, you should probably convert it e.g. to a <see cref="List{T}"/>.</returns>

116 if (string.IsNullOrEmpty(text))
117 {
118 yield break;

120
121 StringBuilder currentRun = new StringBuilder();
122 int boldDepth = 0;
123 int italicsDepth = 0;
124 int underlineDepth = 0;
125 int superscriptDepth = 0;
126 int subscriptDepth = 0;
127 Stack<Brush> brushes = new Stack<Brush>();
128 brushes.Push(defaultBrush);
129
130 for (int i = 0; i < text.Length; i++)
131 {
132 if (text[i] != '<')
133 {
134 currentRun.Append(text[i]);
135 }
136 else
137 {
138 Tags tag = GetTag(text, i, out int tagEnd, out Brush tagBrush);
139

142 currentRun.Append(text[i]);
143 }
144 else
145 {
146 i = tagEnd;
147

149
150 switch (tag)

Script.Superscript : subscriptDepth > superscriptDepth ? Script.Subscript : Script.Normal, brushes.Peek(), spacing);
156 }

158 boldDepth++;
159 break;

178 break;

208 }

214 yield return new FormattedText(txt, GetFont(boldDepth % 2 == 1, italicsDepth % 2 == 1, underlineDepth % 2 == 1, normalFont, boldFont, italicFont, boldItalicFont), superscriptDepth > subscriptDepth ? Script.Superscript : subscriptDepth > superscriptDepth ? Script.Subscript : Script.Normal, brushes.Peek(), spacing);

374 private static Font GetFont(bool bold, bool italic, bool underlined, Font normalFont, Font boldFont, Font italicFont, Font boldItalicFont)
375 {
376 if (!bold && !italic)
377 {
378 if (underlined)
379 {
380 return new Font(normalFont.FontFamily, normalFont.FontSize, normalFont.Underline == null);
381 }
382 else
383 {
384 return normalFont;
385 }
386 }
387 else if (bold && !italic)
388 {
389 if (underlined)
390 {
391 return new Font(boldFont.FontFamily, boldFont.FontSize, boldFont.Underline == null);
392 }
393 else
394 {
395 return boldFont;
396 }
397 }
398 else if (!bold && italic)
399 {
400 if (underlined)
401 {
402 return new Font(italicFont.FontFamily, italicFont.FontSize, italicFont.Underline == null);
403 }
404 else
405 {
406 return italicFont;
407 }
408 }
409 else
410 {
411 if (underlined)
412 {
413 return new Font(boldItalicFont.FontFamily, boldItalicFont.FontSize, boldItalicFont.Underline == null);
414 }
415 else
416 {
417 return boldItalicFont;
418 }
419 }

425 BoldClose,

427 ItalicsClose,

439 private static Tags GetTag(string text, int start, out int tagEnd, out Brush tagBrush)
440 {
441 StringBuilder tag = new StringBuilder();
442 bool closed = false;
443
444 int i = start;
445
446 for (; i < text.Length; i++)
447 {
448 tag.Append(text[i]);
449
450 if (text[i] == '>')
451 {
452 closed = true;
453 break;
454 }
455 }
456
457 tagEnd = i;
458 tagBrush = null;
459
460 if (!closed)
461 {
462 return Tags.None;
463 }
464 else
465 {
466 string tagString = tag.Replace(" ", "").ToString().ToLowerInvariant();
467
468 if (tagString == "<b>" || tagString == "<strong>")
469 {
470 return Tags.BoldOpen;
471 }
472 else if (tagString == "</b>" || tagString == "</strong>")
473 {
474 return Tags.BoldClose;
475 }
476 else if (tagString == "<i>" || tagString == "<em>")
477 {
478 return Tags.ItalicsOpen;
479 }
480 else if (tagString == "</i>" || tagString == "</em>")
481 {
482 return Tags.ItalicsClose;
483 }
484 else if (tagString == "<u>")
485 {
486 return Tags.UnderlineOpen;
487 }
488 else if (tagString == "</u>")
489 {
490 return Tags.UnderlineClose;
491 }
492 else if (tagString == "<sup>")
493 {
494 return Tags.SupOpen;
495 }
496 else if (tagString == "</sup>")
497 {
498 return Tags.SupClose;
499 }
500 else if (tagString == "<sub>")
501 {
502 return Tags.SubOpen;
503 }
504 else if (tagString == "</sub>")
505 {
506 return Tags.SubClose;
507 }
508 else if (tagString.StartsWith("<#"))
509 {
510 string colour = tagString.Substring(1, tagString.Length - 2);
511
512 Colour? col = null;
513
514 try
515 {
516 col = Colour.FromCSSString(colour);
517 }
518 catch { }
519
520 if (col == null)
521 {
522 colour = tagString.Substring(2, tagString.Length - 3);
523
524 col = Colour.FromCSSString(colour);
525 }
526
527 if (col != null)
528 {
529 tagBrush = col.Value;
530 return Tags.ColourOpen;
531 }
532 else
533 {
534 return Tags.None;
535 }
536 }
537 else if (tagString == "</#>")
538 {
539 return Tags.ColourClose;
540 }
541 else
542 {
543 return Tags.None;
544 }
545 }
546 }

584 }

588 }

601 }
602 else
603 {
604 width += metrics.Width + metrics.RightSideBearing;
605 lsb = metrics.LeftSideBearing;
606 }
607 }
608 else
609 {

611
612 Font.DetailedFontMetrics metrics;
613
614 if (txt.Spacing == TextSpacing.Default)
615 {
616 metrics = newFont.MeasureTextAdvanced(txt.Text);
617 }
618 else
619 {
620 metrics = Graphics.MeasureTextWithSpacing(0, 0, txt.Text, newFont, i == fText.Length - 1, txt.Spacing);
621 }
622
623 allMetrics?.Add(metrics);
624
625 advanceWidth += metrics.AdvanceWidth;
626
627 if (txt.Script == Script.Subscript)
628 {
629 top = Math.Max(top, metrics.Top - txt.Font.FontSize * 0.14);
630 bottom = Math.Min(bottom, metrics.Bottom - txt.Font.FontSize * 0.14);
631 }
632 else if (txt.Script == Script.Superscript)
633 {
634 top = Math.Max(top, metrics.Top + txt.Font.FontSize * 0.33);
635 bottom = Math.Min(bottom, metrics.Bottom + txt.Font.FontSize * 0.33);
636 }
637
638 rsb = metrics.RightSideBearing;
639
640 if (!isFirst)
641 {
642 width += metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing;
643 }
644 else
645 {
646 width += metrics.Width + metrics.RightSideBearing;
647 lsb = metrics.LeftSideBearing;
648 }
649 }
650
651 if (isFirst)
652 {
653 isFirst = false;
654 }
655 }
656 }
657
658 width -= rsb;
659
660 return new Font.DetailedFontMetrics(width, top - bottom, lsb, rsb, top, bottom, advanceWidth);
661 }
