22namespace VectSharp.Filters

24 /// <summary>
25 /// Represents a filter that corresponds to applying multiple <see cref="ILocationInvariantFilter"/>s one after the other.
26 /// </summary>
27 public class CompositeLocationInvariantFilter : ILocationInvariantFilter
28 {
29 /// <inheritdoc/>
30 public Point TopLeftMargin { get; }
31
32 /// <inheritdoc/>
33 public Point BottomRightMargin { get; }
34
35 /// <summary>
36 /// The filters that are applied by this filter.
37 /// </summary>
38 public ImmutableList<ILocationInvariantFilter> Filters { get; }
39
40 /// <summary>
41 /// Creates a new <see cref="CompositeLocationInvariantFilter"/> with the specified filters.
42 /// </summary>
43 /// <param name="filters">The filters that will be applied by the new filter.</param>
44 public CompositeLocationInvariantFilter(IEnumerable<ILocationInvariantFilter> filters)
45 {
46 IEnumerable<ILocationInvariantFilter> flattenedFilters = FlattenFilters(filters);
47 this.Filters = ImmutableList.CreateRange(flattenedFilters);
48
49 bool initialised = false;
50
51 foreach (IFilter filter in flattenedFilters)
52 {
53 if (!initialised)
54 {
55 this.TopLeftMargin = filter.TopLeftMargin;
56 this.BottomRightMargin = filter.BottomRightMargin;
57 initialised = true;
58 }
59 else
60 {
61 this.TopLeftMargin = Point.Max(this.TopLeftMargin, filter.TopLeftMargin);
62 this.BottomRightMargin = Point.Max(this.BottomRightMargin, filter.BottomRightMargin);
63 }
64 }
65 }
66
67 /// <summary>
68 /// Creates a new <see cref="CompositeLocationInvariantFilter"/> with the specified filters.
69 /// </summary>
70 /// <param name="filters">The filters that will be applied by the new filter.</param>
71 public CompositeLocationInvariantFilter(params ILocationInvariantFilter[] filters) : this((IEnumerable<ILocationInvariantFilter>)filters) { }
72
73 private IEnumerable<ILocationInvariantFilter> FlattenFilters(IEnumerable<ILocationInvariantFilter> filters)
74 {
75 foreach (ILocationInvariantFilter filter in filters)
76 {
77 if (filter is CompositeLocationInvariantFilter composite)
78 {
79 foreach (ILocationInvariantFilter filter2 in FlattenFilters(composite.Filters))
80 {
81 yield return filter2;
82 }
83 }
84 else
85 {
86 yield return filter;
87 }
88 }
89 }
90
91 /// <inheritdoc/>
92 [Pure]
93 public RasterImage Filter(RasterImage image, double scale)
94 {
95 RasterImage currImage = image;
96
97 foreach (ILocationInvariantFilter filter in this.Filters)
98 {
99 RasterImage prevImage = currImage;
100 currImage = filter.Filter(prevImage, scale);
101
102 if (prevImage != image)
103 {
104 prevImage.Dispose();
105 }
106 }
107
108 return currImage;
109 }
110 }
