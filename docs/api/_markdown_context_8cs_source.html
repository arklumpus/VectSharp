189 /// <summary>
190 /// Get the maximum x value for text drawn at the specified <paramref name="y"/> position.
191 /// </summary>
192 /// <param name="y">The vertical position at which text should be drawn.</param>
193 /// <param name="pageMaxX">The overall maximum x value for the page, after taking margins into account.</param>
194 /// <returns>The maximum x value for text drawn at the specified <paramref name="y"/> position.</returns>
195 public double GetMaxX(double y, double pageMaxX)
196 {
197 y += Translation.Y;
198
199 double maxX = pageMaxX;
200
201 foreach ((double MaxX, double MinY, double MaxY) in ForbiddenAreasRight)
202 {
203 if (MinY <= y && MaxY >= y)
204 {
205 maxX = Math.Min(maxX, MaxX - this.Translation.X);
206 }
207 }
208
209 return maxX;
210 }
211
212 /// <summary>
213 /// Get the maximum x value for text that will extend from <paramref name="y0"/> to <paramref name="y1"/>.
214 /// </summary>
215 /// <param name="y0">The vertical position of the start of the text block.</param>
216 /// <param name="y1">The vertical position of the end of the text block.</param>
217 /// <param name="pageMaxX">The overall maximum x value for the page, after taking margins into account.</param>
218 /// <returns>The maximum x value for text that will extend from <paramref name="y0"/> to <paramref name="y1"/>.</returns>
219 public double GetMaxX(double y0, double y1, double pageMaxX)
220 {
221 y0 += Translation.Y;
222 y1 += Translation.Y;
223
224 double maxX = pageMaxX;
225
226 foreach ((double MaxX, double MinY, double MaxY) in ForbiddenAreasRight)
227 {
228 if (!((MinY < y0 && MaxY < y0) || (MinY > y1 && MaxY > y1)))
229 {
230 maxX = Math.Min(maxX, MaxX - this.Translation.X);
231 }
232 }
233
234 return maxX;
235 }
