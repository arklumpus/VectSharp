201 /// <summary>
202 /// Creates a new <see cref="RasterImage"/> instance copying the specified pixel data.
203 /// </summary>
204 /// <param name="data">The image pixel data that will be copied.</param>
205 /// <param name="width">The width in pixels of the image.</param>
206 /// <param name="height">The height in pixels of the image.</param>
207 /// <param name="pixelFormat">The format of the pixel data.</param>
208 /// <param name="interpolate">Whether the image should be interpolated when it is resized.</param>
209 public RasterImage(byte[] data, int width, int height, PixelFormats pixelFormat, bool interpolate)
210 {
211 this.ImageDataAddress = Marshal.AllocHGlobal(data.Length);
212 GC.AddMemoryPressure(data.Length);
213 this.DataHolder = new DisposableIntPtr(this.ImageDataAddress);
214 this.Id = Guid.NewGuid().ToString();
215 this.Width = width;
216 this.Height = height;
217 this.Interpolate = interpolate;
218
219 switch (pixelFormat)
220 {
221 case PixelFormats.RGB:
222 case PixelFormats.RGBA:
223 Marshal.Copy(data, 0, this.ImageDataAddress, data.Length);
224 this.HasAlpha = pixelFormat == PixelFormats.RGBA;
225 break;
226
227 case PixelFormats.BGRA:
228 case PixelFormats.BGR:
229 this.HasAlpha = pixelFormat == PixelFormats.BGRA;
230
231 int pixelSize = pixelFormat == PixelFormats.BGRA ? 4 : 3;
232 int pixelCount = width * height;
233
234 unsafe
235 {
236 byte* dataPointer = (byte*)this.ImageDataAddress;
237 for (int i = 0; i < pixelCount; i++)
238 {
239 dataPointer[i * pixelSize] = data[i * pixelSize + 2];
240 dataPointer[i * pixelSize + 1] = data[i * pixelSize + 1];
241 dataPointer[i * pixelSize + 2] = data[i * pixelSize];
242 if (pixelSize == 4)
243 {
244 dataPointer[i * pixelSize + 3] = data[i * pixelSize + 3];
245 }
246 }
247 }
248 break;
249 }
250 }

252 private void EncodeAsPNG(Stream outputStream)
253 {
254 unsafe
255 {
256 PNGEncoder.SavePNG((byte*)this.ImageDataAddress, this.Width, this.Height, this.HasAlpha, outputStream, PNGEncoder.FilterModes.Adaptive);
257 }

262 /// </summary>
263 public void ClearPNGCache()
264 {
265 _PNGStream?.Dispose();

269 private bool disposedValue;
270
271 /// <inheritdoc/>
272 protected virtual void Dispose(bool disposing)
273 {
274 if (!disposedValue)
275 {
276 if (disposing)
277 {
278 _PNGStream?.Dispose();
279
280 if (DataHolder != null)
281 {
282 DataHolder.Dispose();
283 GC.RemoveMemoryPressure(Height * Width * (HasAlpha ? 4 : 3));
284 }
285 }
286 disposedValue = true;
287 }
