556 {
557 GraphicsPath tbr = new GraphicsPath();
558
559 foreach (Segment seg in segments)
560 {
561 switch (seg.Type)
562 {

565 break;
566 case SegmentType.Move:

568 break;
569 case SegmentType.Line:

571 break;
572 case SegmentType.CubicBezier:

574 break;
575 }
576 }
577
578 return tbr;

740 public void DrawRasterImage(int sourceX, int sourceY, int sourceWidth, int sourceHeight, double destinationX, double destinationY, double destinationWidth, double destinationHeight, RasterImage image)
741 {
742 Save();
743
744 MoveTo(destinationX, destinationY);
745 LineTo(destinationX + destinationWidth, destinationY);
746 LineTo(destinationX + destinationWidth, destinationY + destinationHeight);
747 LineTo(destinationX, destinationY + destinationHeight);
748 Close();
749 SetClippingPath();
750
751 double sourceRectX = (double)sourceX / image.Width;
752 double sourceRectY = 1 - (double)sourceY / image.Height;
753 double sourceRectWidth = (double)sourceWidth / image.Width;
754 double sourceRectHeight = -(double)sourceHeight / image.Height;
755
756 double scaleX = destinationWidth / sourceRectWidth;
757 double scaleY = destinationHeight / sourceRectHeight;
758
759 double translationX = destinationX / scaleX - sourceRectX;
760 double translationY = destinationY / scaleY - sourceRectY;
761
762
763 Scale(scaleX, scaleY);
764 Translate(translationX, translationY);
765
766 _figures.Add(new RasterImageFigure(image, this.Tag));
767
768 Restore();
769 }

771 public void DrawFilteredGraphics(Graphics graphics, IFilter filter)
772 {
773 if (this._filterOption.Operation == PDFContextInterpreter.FilterOption.FilterOperations.RasteriseAll)
774 {
775 double scale = this._filterOption.RasterisationResolution;
776
777 Rectangle bounds = graphics.GetBounds();
778
779 bounds = new Rectangle(bounds.Location.X - filter.TopLeftMargin.X, bounds.Location.Y - filter.TopLeftMargin.Y, bounds.Size.Width + filter.TopLeftMargin.X + filter.BottomRightMargin.X, bounds.Size.Height + filter.TopLeftMargin.Y + filter.BottomRightMargin.Y);
780
781 if (bounds.Size.Width > 0 && bounds.Size.Height > 0)
782 {
783 if (!this._filterOption.RasterisationResolutionRelative)
784 {
785 scale = scale / Math.Min(bounds.Size.Width, bounds.Size.Height);
786 }
787
788 if (graphics.TryRasterise(bounds, scale, true, out RasterImage rasterised))
789 {
790 RasterImage filtered = null;
791
792 if (filter is ILocationInvariantFilter locInvFilter)
793 {
794 filtered = locInvFilter.Filter(rasterised, scale);
795 }
796 else if (filter is IFilterWithLocation filterWithLoc)
797 {
798 filtered = filterWithLoc.Filter(rasterised, bounds, scale);
799 }
800
801 if (filtered != null)
802 {
803 rasterised.Dispose();

917 private static string EscapeStringForPDF(string str)
918 {
919 StringBuilder sb = new StringBuilder();
920
921 for (int i = 0; i < str.Length; i++)
922 {
923 char ch = str[i];
924
925 if (CP1252Chars.Contains(ch))
926 {
927 if ((int)ch < 128)
928 {
929 if (!"\n\r\t\b\f()\\".Contains(ch))
930 {
931 sb.Append(ch);
932 }
933 else
934 {
935 switch (ch)
936 {
937 case '\n':
938 sb.Append("\\n");
939 break;
940 case '\r':
941 sb.Append("\\r");
942 break;
943 case '\t':
944 sb.Append("\\t");
945 break;
946 case '\b':
947 sb.Append("\\b");
948 break;
949 case '\f':
950 sb.Append("\\f");
951 break;
952 case '\\':
953 sb.Append("\\\\");
954 break;
955 case '(':
956 sb.Append("\\(");
957 break;
958 case ')':
959 sb.Append("\\)");
960 break;
961 }
962 }
963 }
964 else
965 {
966 string octal = Convert.ToString((int)ch, 8);
967 while (octal.Length < 3)
968 {
969 octal = "0" + octal;
970 }
971 sb.Append("\\" + octal);
972 }
973 }
974 else
975 {
976 sb.Append('?');
977 }
978 }
979 return sb.ToString();
