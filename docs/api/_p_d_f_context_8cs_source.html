200 private static GraphicsPath GetGraphicsPath(IEnumerable<Figures.Segment> segments)
201 {
202 GraphicsPath tbr = new GraphicsPath();
203
204 foreach (Figures.Segment seg in segments)
205 {
206 switch (seg.Type)
207 {
208 case Figures.SegmentType.Close:
209 tbr.Close();
210 break;
211 case Figures.SegmentType.Move:
212 tbr.MoveTo(seg.Point);
213 break;
214 case Figures.SegmentType.Line:
215 tbr.LineTo(seg.Point);
216 break;
217 case Figures.SegmentType.CubicBezier:
218 tbr.CubicBezierTo(seg.Points[0], seg.Points[1], seg.Points[2]);
219 break;
220 }
221 }
222
223 return tbr;
224 }

342 private void PathText(string text, double x, double y)
343 {
344 GraphicsPath textPath = new GraphicsPath().AddText(x, y, text, Font, TextBaseline);
345
346 for (int j = 0; j < textPath.Segments.Count; j++)
347 {
348 switch (textPath.Segments[j].Type)
349 {
350 case VectSharp.SegmentType.Move:
351 this.MoveTo(textPath.Segments[j].Point.X, textPath.Segments[j].Point.Y);
352 break;
353 case VectSharp.SegmentType.Line:
354 this.LineTo(textPath.Segments[j].Point.X, textPath.Segments[j].Point.Y);
355 break;
356 case VectSharp.SegmentType.CubicBezier:
357 this.CubicBezierTo(textPath.Segments[j].Points[0].X, textPath.Segments[j].Points[0].Y, textPath.Segments[j].Points[1].X, textPath.Segments[j].Points[1].Y, textPath.Segments[j].Points[2].X, textPath.Segments[j].Points[2].Y);
358 break;
359 case VectSharp.SegmentType.Close:
360 this.Close();
361 break;
362 }
363 }

708 public void DrawFilteredGraphics(Graphics graphics, IFilter filter)
709 {
710 if (this._filterOption.Operation == PDFContextInterpreter.FilterOption.FilterOperations.RasteriseAll)
711 {
712 double scale = this._filterOption.RasterisationResolution;
713
714 Rectangle bounds = graphics.GetBounds();
715
716 bounds = new Rectangle(bounds.Location.X - filter.TopLeftMargin.X, bounds.Location.Y - filter.TopLeftMargin.Y, bounds.Size.Width + filter.TopLeftMargin.X + filter.BottomRightMargin.X, bounds.Size.Height + filter.TopLeftMargin.Y + filter.BottomRightMargin.Y);
717
718 if (bounds.Size.Width > 0 && bounds.Size.Height > 0)
719 {
720 if (!this._filterOption.RasterisationResolutionRelative)
721 {
722 scale /= Math.Min(bounds.Size.Width, bounds.Size.Height);
723 }
724
725 if (graphics.TryRasterise(bounds, scale, true, out RasterImage rasterised))
726 {
727 RasterImage filtered = null;
728
729 if (filter is ILocationInvariantFilter locInvFilter)
730 {
731 filtered = locInvFilter.Filter(rasterised, scale);
732 }
733 else if (filter is IFilterWithLocation filterWithLoc)
734 {
735 filtered = filterWithLoc.Filter(rasterised, bounds, scale);
736 }
737
738 if (filtered != null)
739 {
740 rasterised.Dispose();
741
742 DrawRasterImage(0, 0, filtered.Width, filtered.Height, bounds.Location.X, bounds.Location.Y, bounds.Size.Width, bounds.Size.Height, filtered);
743 }
744 }
745 else
746 {
747 throw new NotImplementedException(@"The filter could not be rasterised! You can avoid this error by doing one of the following:
748 Add a reference to VectSharp.Raster or VectSharp.Raster.ImageSharp (you may also need to add a using directive somewhere to force the assembly to be loaded).
749 Provide your own implementation of Graphics.RasterisationMethod.
750 Set the FilterOption.Operation to ""IgnoreAll"" or ""SkipAll"".");
751 }
752 }
753 }
754 else if (this._filterOption.Operation == PDFContextInterpreter.FilterOption.FilterOperations.IgnoreAll)
755 {
756 graphics.CopyToIGraphicsContext(this);
757 }
758 else
759 {
760
761 }
762 }
