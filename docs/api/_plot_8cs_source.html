202 /// <returns>The first <see cref="IPlotElement"/> of the specified type, or the first <see cref="ICoordinateSystem"/> of the specified type, or <see langword="null"/> if none could be found.</returns>
203 public T GetFirst<T>() where T : class
204 {

240 /// <summary>
241 /// Render the plot to a suitably cropped <see cref="Page"/> object.
242 /// </summary>
243 /// <returns>A <see cref="Page"/> object containing the rendered plot.</returns>
244 public Page Render()
245 {
246 Page pag = new Page(1, 1);
247
248 this.Render(pag.Graphics);
249
250 Rectangle bounds = pag.Graphics.GetBounds();
251
252 pag.Crop(new Point(bounds.Location.X - bounds.Size.Width * 0.01, bounds.Location.Y - bounds.Size.Height * 0.01), new Size(bounds.Size.Width * 1.02, bounds.Size.Height * 1.02));
253
254 return pag;
255 }

265 private static (double minX, double minY, double maxX, double maxY, double rangeX, double rangeY) GetDataRange(IReadOnlyList<IReadOnlyList<double>> data)
266 {
267 double minX = double.MaxValue;
268 double minY = double.MaxValue;
269
270 double maxX = double.MinValue;
271 double maxY = double.MinValue;
272
273 for (int i = 0; i < data.Count; i++)
274 {
275 minX = Math.Min(minX, data[i][0]);
276 minY = Math.Min(minY, data[i][1]);
277
278 maxX = Math.Max(maxX, data[i][0]);
279 maxY = Math.Max(maxY, data[i][1]);
280 }
281
282 double rangeX = maxX - minX;
283 double rangeY = maxY - minY;
284
285 return (minX, minY, maxX, maxY, rangeX, rangeY);
286 }
