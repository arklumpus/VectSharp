25 /// <summary>
26 /// A plot element that draws bars.
27 /// </summary>
28 /// <typeparam name="T">The type of data elements.</typeparam>
29 public class Bars<T> : IPlotElement
30 {
31 private double margin = 0;
32
33 /// <summary>
34 /// The data points corresponding to the tips of the bars.
35 /// </summary>
36 public SortedSet<T> Data { get; set; }
37
38 /// <summary>
39 /// A function that returns the bottom for each bar. This function should accept
40 /// a single parameter of type <typeparamref name="T"/> and return another <typeparamref name="T"/>
41 /// object, representing the bottom of the bar in data space.
42 /// </summary>
43 public Func<T, T> GetBaseline { get; set; }
44
45 /// <summary>
46 /// The margin between consecutive bars. This should range between 0 and 1.
47 /// </summary>
48 public double Margin
49 {
50 get => margin;
51 set
52 {
53 if (value >= 0 && value <= 1)
54 {
55 margin = value;
56 }
57 else
58 {
59 throw new ArgumentOutOfRangeException(nameof(value), value, "The value for the margin must be within 0 and 1 (inclusive)!");
60 }
61 }
62 }
63
64 /// <summary>
65 /// The coordinate system used to transform the points from data space to plot space.
66 /// </summary>
67 public ICoordinateSystem<T> CoordinateSystem { get; set; }
68 ICoordinateSystem IPlotElement.CoordinateSystem => CoordinateSystem;
69
70 /// <summary>
71 /// Presentation attributes for the bars.
72 /// </summary>
73 public PlotElementPresentationAttributes PresentationAttributes { get; set; } = new PlotElementPresentationAttributes();
74
75 /// <summary>
76 /// A tag to identify the bars in the plot.
77 /// </summary>
78 public string Tag { get; set; }

80 /// <summary>
81 /// Create a new <see cref="Bars{T}"/> instance.
82 /// </summary>
83 /// <param name="data">The data points corresponding to the tips of the bars.</param>
84 /// <param name="sorting">A comparer used to sort the bars.</param>
85 /// <param name="getBaseline">A function that returns the bottom for each bar. This function should accept
86 /// a single parameter of type <typeparamref name="T"/> and return another <typeparamref name="T"/>
87 /// object, representing the bottom of the bar in data space.</param>
88 /// <param name="coordinateSystem">The coordinate system used to transform the points from data space to plot space.</param>
89 public Bars(IEnumerable<T> data, IComparer<T> sorting, Func<T, T> getBaseline, ICoordinateSystem<T> coordinateSystem)
90 {
91 this.Data = new SortedSet<T>(data, sorting);
92 this.CoordinateSystem = coordinateSystem;
93 this.GetBaseline = getBaseline;
94 }

96 /// <summary>
97 /// Create a new <see cref="Bars{T}"/> instance.
98 /// </summary>
99 /// <param name="data">The data points corresponding to the tips of the bars.</param>
100 /// <param name="sorting">A comparer used to sort the bars.</param>
101 /// <param name="getBaseline">A function that returns the bottom for each bar. This function should accept
102 /// a single parameter of type <typeparamref name="T"/> and return another <typeparamref name="T"/>
103 /// object, representing the bottom of the bar in data space.</param>
104 /// <param name="coordinateSystem">The coordinate system used to transform the points from data space to plot space.</param>
105 public Bars(IEnumerable<T> data, Comparison<T> sorting, Func<T, T> getBaseline, ICoordinateSystem<T> coordinateSystem) : this(data, Comparer<T>.Create(sorting), getBaseline, coordinateSystem) { }

107 /// <summary>
108 /// Create a new <see cref="Bars{T}"/> instance.
109 /// </summary>
110 /// <param name="data">The data points corresponding to the tips of the bars. These should already be sorted.</param>
111 /// <param name="getBaseline">A function that returns the bottom for each bar. This function should accept
112 /// a single parameter of type <typeparamref name="T"/> and return another <typeparamref name="T"/>
113 /// object, representing the bottom of the bar in data space.</param>
114 /// <param name="coordinateSystem">The coordinate system used to transform the points from data space to plot space.</param>
115 public Bars(IReadOnlyList<T> data, Func<T, T> getBaseline, ICoordinateSystem<T> coordinateSystem)
116 {
117 Dictionary<T, int> indices = new Dictionary<T, int>();
118
119 for (int i = 0; i < data.Count; i++)
120 {
121 indices[data[i]] = i;
122 }
123
124 this.Data = new SortedSet<T>(data, Comparer<T>.Create((x, y) => Math.Sign(indices[x] - indices[y])));
125 this.CoordinateSystem = coordinateSystem;
126 this.GetBaseline = getBaseline;
127 }

129 private static bool AnyNan(params Point[] points)
130 {
131 for (int i = 0; i < points.Length; i++)
132 {
133 if (double.IsNaN(points[i].X) || double.IsNaN(points[i].Y))
134 {
135 return true;
136 }
137 }
138
139 return false;
140 }

142 /// <inheritdoc/>
143 public void Plot(Graphics target)
144 {
145 List<(Point, Point)> bars = new List<(Point, Point)>();
146
147 foreach (T data in Data)
148 {
149 Point pt = CoordinateSystem.ToPlotCoordinates(data);
150 Point baseline = CoordinateSystem.ToPlotCoordinates(GetBaseline(data));
151 bars.Add((baseline, pt));
152 }
153
154 for (int i = 0; i < bars.Count; i++)
155 {
156 GraphicsPath pth = new GraphicsPath();
157
158 if (i > 0 && i < bars.Count - 1)
159 {
160 Point prevBaselineMid = new Point(bars[i].Item1.X * (0.5 + Margin * 0.5) + bars[i - 1].Item1.X * (0.5 - Margin * 0.5), bars[i].Item1.Y * (0.5 + Margin * 0.5) + bars[i - 1].Item1.Y * (0.5 - Margin * 0.5));
161
162 Point perpDir = new Point(bars[i].Item2.Y - bars[i].Item1.Y, -(bars[i].Item2.X - bars[i].Item1.X));
163 perpDir = perpDir.Normalize();
164
165 double t = (bars[i].Item2.X - prevBaselineMid.X) * perpDir.X + (bars[i].Item2.Y - prevBaselineMid.Y) * perpDir.Y;
166 Point prevTop = new Point(bars[i].Item2.X - perpDir.X * t, bars[i].Item2.Y - perpDir.Y * t);
167
168 Point nextBaselineMid = new Point(bars[i].Item1.X * (0.5 + Margin * 0.5) + bars[i + 1].Item1.X * (0.5 - Margin * 0.5), bars[i].Item1.Y * (0.5 + Margin * 0.5) + bars[i + 1].Item1.Y * (0.5 - Margin * 0.5));
169
170 double t2 = (bars[i].Item2.X - nextBaselineMid.X) * perpDir.X + (bars[i].Item2.Y - nextBaselineMid.Y) * perpDir.Y;
171 Point nextTop = new Point(bars[i].Item2.X - perpDir.X * t2, bars[i].Item2.Y - perpDir.Y * t2);
172
173 if (!AnyNan(prevBaselineMid, prevTop, nextTop, nextBaselineMid))
174 {
175 pth.MoveTo(prevBaselineMid).LineTo(prevTop).LineTo(nextTop).LineTo(nextBaselineMid).Close();
176 }
177 }
178 else if (i == 0)
179 {
180 Point perpDir = new Point(bars[i].Item2.Y - bars[i].Item1.Y, -(bars[i].Item2.X - bars[i].Item1.X));
181 perpDir = perpDir.Normalize();
182
183 Point nextBaselineMid = new Point(bars[i].Item1.X * (0.5 + Margin * 0.5) + bars[i + 1].Item1.X * (0.5 - Margin * 0.5), bars[i].Item1.Y * (0.5 + Margin * 0.5) + bars[i + 1].Item1.Y * (0.5 - Margin * 0.5));
184
185 double t2 = (bars[i].Item2.X - nextBaselineMid.X) * perpDir.X + (bars[i].Item2.Y - nextBaselineMid.Y) * perpDir.Y;
186 Point nextTop = new Point(bars[i].Item2.X - perpDir.X * t2, bars[i].Item2.Y - perpDir.Y * t2);
187
188
189 Point prevBaselineMid = new Point(2 * bars[i].Item1.X - nextBaselineMid.X, 2 * bars[i].Item1.Y - nextBaselineMid.Y);
190 double t = -t2;
191 Point prevTop = new Point(bars[i].Item2.X - perpDir.X * t, bars[i].Item2.Y - perpDir.Y * t);
192
193 if (!AnyNan(prevBaselineMid, prevTop, nextTop, nextBaselineMid))
194 {
195 pth.MoveTo(prevBaselineMid).LineTo(prevTop).LineTo(nextTop).LineTo(nextBaselineMid).Close();
196 }
197 }
198 else if (i == bars.Count - 1)
199 {
200 Point prevBaselineMid = new Point(bars[i].Item1.X * (0.5 + Margin * 0.5) + bars[i - 1].Item1.X * (0.5 - Margin * 0.5), bars[i].Item1.Y * (0.5 + Margin * 0.5) + bars[i - 1].Item1.Y * (0.5 - Margin * 0.5));
201
202 Point perpDir = new Point(bars[i].Item2.Y - bars[i].Item1.Y, -(bars[i].Item2.X - bars[i].Item1.X));
203 perpDir = perpDir.Normalize();
204
205 double t = (bars[i].Item2.X - prevBaselineMid.X) * perpDir.X + (bars[i].Item2.Y - prevBaselineMid.Y) * perpDir.Y;
206 Point prevTop = new Point(bars[i].Item2.X - perpDir.X * t, bars[i].Item2.Y - perpDir.Y * t);
207
208 Point nextBaselineMid = new Point(2 * bars[i].Item1.X - prevBaselineMid.X, 2 * bars[i].Item1.Y - prevBaselineMid.Y);
209
210 double t2 = -t;
211 Point nextTop = new Point(bars[i].Item2.X - perpDir.X * t2, bars[i].Item2.Y - perpDir.Y * t2);
212
213 if (!AnyNan(prevBaselineMid, prevTop, nextTop, nextBaselineMid))
214 {
215 pth.MoveTo(prevBaselineMid).LineTo(prevTop).LineTo(nextTop).LineTo(nextBaselineMid).Close();
216 }
217 }
218
219 string tag = Tag;
220 string strokeTag = Tag;
221
222 if (!string.IsNullOrEmpty(Tag))
223 {
224 tag = tag + "@" + i.ToString();
225 strokeTag = tag + "@" + i.ToString() + "_stroke";
226 }
227
228 if (PresentationAttributes.Fill != null)
229 {
230 target.FillPath(pth, PresentationAttributes.Fill, tag);
231 }
232
233 if (PresentationAttributes.Stroke != null)
234 {
235 target.StrokePath(pth, PresentationAttributes.Stroke, PresentationAttributes.LineWidth, PresentationAttributes.LineCap, PresentationAttributes.LineJoin, PresentationAttributes.LineDash, strokeTag);
236 }
237 }
238 }
239 }

432 /// <inheritdoc/>
433 public void Plot(Graphics target)
434 {
435 List<(Point, Point)> bars = new List<(Point, Point)>();
436 List<double[]> ratios = new List<double[]>();
437
438 foreach (IReadOnlyList<double> data in Data)
439 {
440 Point baseline = CoordinateSystem.ToPlotCoordinates(GetBaseline(data));
441
442 double total = 0;
443
444 double[] currRatios = new double[data.Count - 1];
445
446 for (int i = 0; i < data.Count - 1; i++)
447 {
448 if (i == 0)
449 {
450 total += Vertical ? data[1] : data[0];
451 }
452 else
453 {
454 total += data[i + 1];
455 }
456
457 Point currPt = CoordinateSystem.ToPlotCoordinates(Vertical ? new double[] { data[0], total } : new double[] { total, data[1] });
458
459 currRatios[i] = new Point(currPt.X - baseline.X, currPt.Y - baseline.Y).Modulus();
460 }
461
462 for (int i = 0; i < currRatios.Length; i++)
463 {
464 currRatios[i] /= currRatios[currRatios.Length - 1];
465 }
466
467 Point pt = CoordinateSystem.ToPlotCoordinates(Vertical ? new double[] { data[0], total } : new double[] { total, data[1] });
468
469 bars.Add((baseline, pt));
470 ratios.Add(currRatios);
471 }
472
473 for (int i = 0; i < bars.Count; i++)
474 {
475 List<GraphicsPath> paths = new List<GraphicsPath>();
476
477 if (i > 0 && i < bars.Count - 1)
478 {
479 Point prevBaselineMid = new Point(bars[i].Item1.X * (0.5 + Margin * 0.5) + bars[i - 1].Item1.X * (0.5 - Margin * 0.5), bars[i].Item1.Y * (0.5 + Margin * 0.5) + bars[i - 1].Item1.Y * (0.5 - Margin * 0.5));
480
481 Point perpDir = new Point(bars[i].Item2.Y - bars[i].Item1.Y, -(bars[i].Item2.X - bars[i].Item1.X));
482 perpDir = perpDir.Normalize();
