66 internal class UnicodeEmojiParser : InlineParser
67 {
68 public UnicodeEmojiParser()
69 {
70 OpeningCharacters = Enumerable.Range(0xD800, 0xDBFF - 0xD800 + 1).Select(x => (char)x).ToArray();
71 }
72
73 public override bool Match(InlineProcessor processor, ref StringSlice slice)
74 {
75 if (char.IsHighSurrogate(slice.CurrentChar))
76 {
77 StringInfo info = new StringInfo(slice.Text.Substring(slice.Start));
78 string emoji = info.SubstringByTextElements(0, 1);
79
80 processor.Inline = new EmojiInline("unicode://" + emoji)
81 {
82 Span = {
83 Start = processor.GetSourcePosition(slice.Start, out int line, out int column)
84 },
85 Line = line,
86 Column = column,
87 };
88 processor.Inline.Span.End = processor.Inline.Span.Start + emoji.Length - 1;
89
90 slice.Start += emoji.Length;
91
92 return true;
93 }
94 else
95 {
96 return false;
97 }
98 }
