39 static Func<RenderTargetBitmap, SKBitmap> GetBitmap;
40 static Func<RenderTargetBitmap, object> GetLock;
41
42 static SkiaBitmap()
43 {
44 ParameterExpression bitmapParameter = Expression.Parameter(typeof(RenderTargetBitmap));
45
46 PropertyInfo platformImplProperty = typeof(RenderTargetBitmap).GetProperty("PlatformImpl", BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.DeclaredOnly);
47 MemberExpression getPlatformImpl = Expression.Property(bitmapParameter, platformImplProperty);
48 MemberExpression getItem = Expression.Property(getPlatformImpl, "Item");
49
50 Type renderTargetImpl = typeof(Avalonia.Skia.SkiaPlatform).Assembly.GetType("Avalonia.Skia.WriteableBitmapImpl");
51
52 MemberExpression getBitmap = Expression.Field(Expression.Convert(getItem, renderTargetImpl), "_bitmap");
53 MemberExpression getLock = Expression.Field(Expression.Convert(getItem, renderTargetImpl), "_lock");
54
55 GetBitmap = Expression.Lambda<Func<RenderTargetBitmap, SKBitmap>>(getBitmap, bitmapParameter).Compile();
56 GetLock = Expression.Lambda<Func<RenderTargetBitmap, object>>(getLock, bitmapParameter).Compile();
57 }
58
59 public SkiaBitmap(int width, int height)
60 {
61 this.AvaloniaBitmap = new RenderTargetBitmap(new PixelSize(width, height), new Vector(96, 96));
62 this.Lock = GetLock(this.AvaloniaBitmap);
63 this.Bitmap = GetBitmap(this.AvaloniaBitmap);
64
65 this.SKCanvas = new SKCanvas(this.Bitmap);
66 this.Size = new Avalonia.Size(width, height);
67 this.Width = width;
68 this.Height = height;
69 }
70
71 private bool disposedValue;
72
73 public void Draw(DrawingContext context, Rect sourceRect, Rect destRect)
74 {
75 context.DrawImage(this.AvaloniaBitmap, sourceRect, destRect);
76 }
77
78 protected virtual void Dispose(bool disposing)
79 {
80 if (!disposedValue)
81 {
82 if (disposing)
83 {
84 this.SKCanvas?.Dispose();
85 this.AvaloniaBitmap?.Dispose();
86 }
87 disposedValue = true;
88 }
89 }
90
91 public void Dispose()
92 {
93 Dispose(disposing: true);
94 GC.SuppressFinalize(this);
95 }
96
97 public void Indispose()
98 {
99 if (!disposedValue)
100 {
101 this.SKCanvas?.Dispose();
102 this.disposedValue = true;
103 }
104 }
