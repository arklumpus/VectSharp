464 /// <inheritdoc/>
465 public void Plot(Graphics target)
466 {
467 int index = 0;
468
469 foreach (T data in Data)
470 {
471 IEnumerable<FormattedText> label = labelFunction(index, data);
472 if (label != null)
473 {
474 double rotation = Rotation(index, data);
475 Point pt = CoordinateSystem.ToPlotCoordinates(data);
476 Point margin = Margin(index, data);
477
478 pt = new Point(pt.X + margin.X, pt.Y + margin.Y);
479
480 target.Save();
481 target.Translate(pt);
482 target.Rotate(rotation);
483
484 string fillTag = Tag;
485 string strokeTag = Tag;
486
487
488 if (target.UseUniqueTags && !string.IsNullOrEmpty(Tag))
489 {
490 fillTag = fillTag + "@" + index.ToString();
491 strokeTag = strokeTag + "@stroke" + index.ToString();
492 }
493
494 if (PresentationAttributes.Fill != null)
495 {
496 target.FillText(Alignment == TextAnchors.Left ? 0 : Alignment == TextAnchors.Right ? -label.Measure().Width : -label.Measure().Width * 0.5, 0, label, PresentationAttributes.Fill, Baseline, fillTag);
497 }
498
499 if (PresentationAttributes.Stroke != null)
500 {
501 target.StrokeText(Alignment == TextAnchors.Left ? 0 : Alignment == TextAnchors.Right ? -label.Measure().Width : -label.Measure().Width * 0.5, 0, label, PresentationAttributes.Stroke, Baseline, PresentationAttributes.LineWidth, PresentationAttributes.LineCap, PresentationAttributes.LineJoin, PresentationAttributes.LineDash, fillTag);
502 }
503
504 target.Restore();
505 }
506 index++;
507 }

651 GraphicsPath pthStroke = new GraphicsPath();
652
653 if (!Smooth)
654 {
655 bool started = false;
656
657 List<Point> oppositePoints = new List<Point>();
658
659 foreach (T data in Data)
660 {
661 Point pt = CoordinateSystem.ToPlotCoordinates(data);
662
663 oppositePoints.Add(CoordinateSystem.ToPlotCoordinates(GetBaseline(data)));
664
665 if (!started)
666 {
667 pth.MoveTo(pt);
668 pthStroke.MoveTo(pt);
669 started = true;
670 }
671 else
672 {
673 pth.LineTo(pt);
674 pthStroke.LineTo(pt);
675 }
676 }
677
678 for (int i = oppositePoints.Count - 1; i >= 0; i--)
679 {
680 pth.LineTo(oppositePoints[i]);
681 }
682
683 pth.Close();
684 }
685 else
686 {
687 List<Point> points = new List<Point>();
688 List<Point> oppositePoints = new List<Point>();
689

691 {

693 points.Add(pt);
694 oppositePoints.Add(CoordinateSystem.ToPlotCoordinates(GetBaseline(data)));
695 }
696 pth.AddSmoothSpline(points.ToArray());
697 pthStroke.AddSmoothSpline(points.ToArray());
698 oppositePoints.Reverse();
699 pth.AddSmoothSpline(oppositePoints.ToArray());
700 pth.Close();
701 }

704 string strokeTag = tag;
705
706 if (!string.IsNullOrEmpty(tag) && target.UseUniqueTags)
707 {
708 strokeTag += "@stroke";
709 }

714 }
715
