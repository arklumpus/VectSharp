71 public double[,] GetMatrix()
72 {
73 if (this.Matrix != null)
74 {
75 return this.Matrix;
76 }
77 else if (this.Delta != null)
78 {
79 return Graphics.TranslationMatrix(this.Delta.Value.X, this.Delta.Value.Y);
80 }
81 else if (this.Angle != null)
82 {
83 return Graphics.RotationMatrix(this.Angle.Value);
84 }
85 else if (this.Scale != null)
86 {
87 return Graphics.ScaleMatrix(this.Scale.Value.Width, this.Scale.Value.Height);
88 }
89 else
90 {
91 return null;
92 }

129 public TextAction(Point origin, string text, Font font, TextBaselines textBaseLine, Brush fill, Brush stroke, double lineWidth, LineCaps lineCap, LineJoins lineJoin, LineDash lineDash, string tag)
130 {
131 this.Origin = origin;
132 this.Text = text;
133 this.Font = font;
134 this.TextBaseline = textBaseLine;
135 this.Fill = fill;
136 this.Stroke = stroke;
137 this.LineCap = lineCap;
138 this.LineJoin = lineJoin;
139 this.LineWidth = lineWidth;
140 this.Tag = tag;
141 this.LineDash = lineDash;
142 }
143
144 public Rectangle GetBounds()
145 {
146 Font.DetailedFontMetrics metrics = this.Font.MeasureTextAdvanced(this.Text);
147
148 switch (this.TextBaseline)
149 {
150 case TextBaselines.Top:
151 return new Rectangle(this.Origin, new Size(metrics.Width, metrics.Height));
152 case TextBaselines.Bottom:
153 return new Rectangle(this.Origin.X, this.Origin.Y - metrics.Height, metrics.Width, metrics.Height);
154 case TextBaselines.Middle:
155 return new Rectangle(this.Origin.X, this.Origin.Y - metrics.Height * 0.5, metrics.Width, metrics.Height);
156 case TextBaselines.Baseline:
157 return new Rectangle(this.Origin.X, this.Origin.Y - metrics.Top, metrics.Width, metrics.Height);
158 default:
159 throw new System.ArgumentOutOfRangeException(nameof(TextBaseline), this.TextBaseline, "Invalid text baseline!");
160 }
161 }
