137 public TextAction(Point origin, string text, Font font, TextBaselines textBaseLine, Brush fill, Brush stroke, double lineWidth, LineCaps lineCap, LineJoins lineJoin, LineDash lineDash, string tag)
138 {
139 this.Origin = origin;
140 this.Text = text;
141 this.Font = font;
142 this.TextBaseline = textBaseLine;
143 this.Fill = fill;
144 this.Stroke = stroke;
145 this.LineCap = lineCap;
146 this.LineJoin = lineJoin;
147 this.LineWidth = lineWidth;
148 this.Tag = tag;
149 this.LineDash = lineDash;
150 }
151
152 public Rectangle GetBounds()
153 {
154 Font.DetailedFontMetrics metrics = this.Font.MeasureTextAdvanced(this.Text);
155
156 switch (this.TextBaseline)
157 {
158 case TextBaselines.Top:
159 return new Rectangle(this.Origin, new Size(metrics.Width, metrics.Height));
160 case TextBaselines.Bottom:
161 return new Rectangle(this.Origin.X, this.Origin.Y - metrics.Height, metrics.Width, metrics.Height);
162 case TextBaselines.Middle:
163 return new Rectangle(this.Origin.X, this.Origin.Y - metrics.Height * 0.5, metrics.Width, metrics.Height);
164 case TextBaselines.Baseline:
165 return new Rectangle(this.Origin.X, this.Origin.Y - metrics.Top, metrics.Width, metrics.Height);
166 default:
167 throw new System.ArgumentOutOfRangeException(nameof(TextBaseline), this.TextBaseline, "Invalid text baseline!");
168 }
169 }
