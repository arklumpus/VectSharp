33 /// <summary>
34 /// Render the page to a PNG file.
35 /// </summary>
36 /// <param name="page">The <see cref="Page"/> to render.</param>
37 /// <param name="fileName">The full path to the file to save. If it exists, it will be overwritten.</param>
38 /// <param name="scale">The scale to be used when rasterising the page. This will determine the width and height of the image file.</param>
39 public static void SaveAsPNG(this Page page, string fileName, double scale = 1)
40 {
41 Document doc = new Document();
42 doc.Pages.Add(page);
43
44 MemoryStream ms = new MemoryStream();
45 doc.SaveAsPDF(ms, textOption: PDFContextInterpreter.TextOptions.ConvertIntoPaths, filterOption: new PDFContextInterpreter.FilterOption(PDFContextInterpreter.FilterOption.FilterOperations.RasteriseAll, scale, true));
46
47 using (MuPDFContext context = new MuPDFContext())
48 {
49 using (MuPDFDocument muDoc = new MuPDFDocument(context, ref ms, InputFileTypes.PDF))
50 {
51 muDoc.SaveImage(0, scale, MuPDFCore.PixelFormats.RGBA, fileName, RasterOutputFileTypes.PNG);
52 }
53 }
54
55 ms.Dispose();
56 }

83 /// <summary>
84 /// Rasterise a region of a <see cref="Graphics"/> object.
85 /// </summary>
86 /// <param name="graphics">The <see cref="Graphics"/> object that will be rasterised.</param>
87 /// <param name="region">The region of the <paramref name="graphics"/> that will be rasterised.</param>
88 /// <param name="scale">The scale at which the image will be rendered.</param>
89 /// <param name="interpolate">Whether the resulting image should be interpolated or not when it is drawn on another <see cref="Graphics"/> surface.</param>
90 /// <returns>A <see cref="RasterImage"/> containing the rasterised graphics.</returns>
91 public static RasterImage Rasterise(this Graphics graphics, Rectangle region, double scale, bool interpolate)
92 {
93 Page pag = new Page(1, 1);
94 pag.Graphics.DrawGraphics(0, 0, graphics);
95 pag.Crop(region.Location, region.Size);
96
97 Document doc = new Document();
98 doc.Pages.Add(pag);
99
100 MemoryStream ms = new MemoryStream();
101 doc.SaveAsPDF(ms, textOption: PDFContextInterpreter.TextOptions.ConvertIntoPaths, filterOption: new PDFContextInterpreter.FilterOption(PDFContextInterpreter.FilterOption.FilterOperations.RasteriseAll, scale, true));
102
103 IntPtr imageDataAddress;
104
105 int width;
106 int height;
107
108 using (MuPDFContext context = new MuPDFContext())
109 {
110 using (MuPDFDocument muDoc = new MuPDFDocument(context, ref ms, InputFileTypes.PDF))
111 {
112 int imageSize = muDoc.GetRenderedSize(0, scale, MuPDFCore.PixelFormats.RGBA);
113
114 RoundedRectangle roundedBounds = muDoc.Pages[0].Bounds.Round(scale);
115
116 width = roundedBounds.Width;
117 height = roundedBounds.Height;
118
119
120 imageDataAddress = Marshal.AllocHGlobal(imageSize);
121 GC.AddMemoryPressure(imageSize);
122
123 muDoc.Render(0, scale, MuPDFCore.PixelFormats.RGBA, imageDataAddress);
124 }
125 }
126
127 ms.Dispose();
128
129 DisposableIntPtr disposableAddress = new DisposableIntPtr(imageDataAddress);
130
131 return new RasterImage(ref disposableAddress, width, height, true, interpolate);
132 }

134 /// <summary>
135 /// Render the page to raw pixel data, in 32bpp RGBA format.
136 /// </summary>
137 /// <param name="pag">The <see cref="Page"/> to render.</param>
138 /// <param name="scale">The scale to be used when rasterising the page. This will determine the width and height of the image.</param>
139 /// <param name="width">The width of the rendered image.</param>
140 /// <param name="height">The height of the rendered image.</param>
141 /// <param name="totalSize">The size in bytes of the raw pixel data.</param>
142 /// <returns>A <see cref="DisposableIntPtr"/> containing a pointer to the raw pixel data, stored in unmanaged memory. Dispose this object to release the unmanaged memory.</returns>
143 public static DisposableIntPtr SaveAsRawBytes(this Page pag, out int width, out int height, out int totalSize, double scale = 1)
144 {
145 Document doc = new Document();
146 doc.Pages.Add(pag);
147
148 MemoryStream ms = new MemoryStream();
149 doc.SaveAsPDF(ms, textOption: PDFContextInterpreter.TextOptions.ConvertIntoPaths, filterOption: new PDFContextInterpreter.FilterOption(PDFContextInterpreter.FilterOption.FilterOperations.RasteriseAll, scale, true));
150
151 IntPtr renderedPage;
152
153 using (MuPDFContext context = new MuPDFContext())
154 {
155 using (MuPDFDocument muDoc = new MuPDFDocument(context, ref ms, InputFileTypes.PDF))
156 {
157 totalSize = muDoc.GetRenderedSize(0, scale, MuPDFCore.PixelFormats.RGBA);
158
159 renderedPage = Marshal.AllocHGlobal(totalSize);
160
161 RoundedRectangle bounds = muDoc.Pages[0].Bounds.Round(scale);
162
163 width = bounds.Width;
164 height = bounds.Height;
165
166 muDoc.Render(0, scale, MuPDFCore.PixelFormats.RGBA, renderedPage);
167 }
168 }
169
170 ms.Dispose();
171
172 return new DisposableIntPtr(renderedPage);
173 }

175 /// <summary>
176 /// Saves the animation to a stream in animated PNG format.
177 /// </summary>
178 /// <param name="animation">The animation to export.</param>
179 /// <param name="imageStream">The stream on which the animated PNG will be written.</param>
180 /// <param name="scale">The scale at which the animation will be rendered.</param>
181 /// <param name="frameRate">The target frame rate of the animation, in frames-per-second (fps). This is capped by the animated PNG specification at 90 fps.</param>
182 /// <param name="durationScaling">A scaling factor that will be applied to all durations in the animation. Values greater than 1 slow down the animation, values smaller than 1 accelerate it. Note that this does not affect the frame rate of the animation.</param>
183 /// <param name="interframeCompression">The kind of compression that will be used to reduce file size. Note that if the animation has a transparent background, no compression can be performed, and the value of this parameter is ignored.</param>
184 public static void SaveAsAnimatedPNG(this Animation animation, Stream imageStream, double scale = 1, double frameRate = 60, double durationScaling = 1, AnimatedPNG.InterframeCompression interframeCompression = AnimatedPNG.InterframeCompression.First)
185 {
186 frameRate = Math.Min(frameRate, 90);
187
188 int frames = (int)Math.Ceiling(animation.Duration * frameRate * durationScaling / 1000);
189
190 Stream fs = imageStream;
191
192 AnimatedPNG.CompressedFrame[] compressedFrames = new AnimatedPNG.CompressedFrame[frames];
193
194 int width = (int)(animation.Width * scale);
195 int height = (int)(animation.Height * scale);
196
197 if (animation.Background.A < 1 || interframeCompression == AnimatedPNG.InterframeCompression.None)
198 {
199 Parallel.For(0, frames, i =>
200 {
201 double frameTime = i / frameRate / durationScaling * 1000;
202
203 double frameDuration = Math.Min((animation.Duration - frameTime) * durationScaling, 1000 / frameRate);
204
205 Page pag = animation.GetFrameAtAbsolute(frameTime);
206
207 using (DisposableIntPtr rawFrame = pag.SaveAsRawBytes(out _, out _, out _, scale))
208 {
209 compressedFrames[i] = new AnimatedPNG.CompressedFrame(rawFrame, width, height, true, frameDuration);
210 }
211 });
212 }
213 else if (interframeCompression == AnimatedPNG.InterframeCompression.First)
214 {
215 DisposableIntPtr firstFrame = animation.GetFrameAtAbsolute(0).SaveAsRawBytes(out _, out _, out _, scale);
216 compressedFrames[0] = new AnimatedPNG.CompressedFrame(firstFrame, width, height, true, Math.Min(animation.Duration * durationScaling, 1000 / frameRate));
217
218 Parallel.For(1, frames, i =>
219 {
220 double frameTime = i / frameRate / durationScaling * 1000;
221
222 double frameDuration = Math.Min((animation.Duration - frameTime) * durationScaling, 1000 / frameRate);
223
224 Page pag = animation.GetFrameAtAbsolute(frameTime);
225
226 using (DisposableIntPtr rawFrame = pag.SaveAsRawBytes(out _, out _, out _, scale))
227 {
228 compressedFrames[i] = new AnimatedPNG.CompressedFrame(rawFrame, firstFrame, true, width, height, true, frameDuration);
229 }
230 });
231
232 firstFrame.Dispose();
233 }
234 else if (interframeCompression == AnimatedPNG.InterframeCompression.Previous)
235 {
236 DisposableIntPtr[] rawFrames = new DisposableIntPtr[frames];
237
238 Parallel.For(0, frames, i =>
239 {
240 double frameTime = i / frameRate / durationScaling * 1000;
241 Page pag = animation.GetFrameAtAbsolute(frameTime);
242 rawFrames[i] = pag.SaveAsRawBytes(out _, out _, out _, scale);
243 });
244
245 compressedFrames[0] = new AnimatedPNG.CompressedFrame(rawFrames[0], width, height, true, Math.Min(animation.Duration * durationScaling, 1000 / frameRate));
246
247 Parallel.For(1, frames, i =>
248 {
249 double frameTime = i / frameRate / durationScaling * 1000;
250 double frameDuration = Math.Min((animation.Duration - frameTime) * durationScaling, 1000 / frameRate);
251 compressedFrames[i] = new AnimatedPNG.CompressedFrame(rawFrames[i], rawFrames[i - 1], false, width, height, true, frameDuration);
252 });
253 }
254
255 AnimatedPNG.Create(fs, (int)(animation.Width * scale), (int)(animation.Height * scale), true, compressedFrames, animation.RepeatCount);
256 }
