25 /// <summary>
26 /// Represents a font library with methods to create <see cref="FontFamily"/> objects from a string or from <see cref="FontFamily.StandardFontFamilies"/>.
27 /// </summary>
28 public interface IFontLibrary
29 {
30 /// <summary>
31 /// Create a new font family from the specified family name or true type file. If the family name or the true type file are not valid, an exception might be raised.
32 /// </summary>
33 /// <param name="fontFamily">The name of the font family to create, or the path to a TTF file.</param>
34 /// <returns>If the font family name or the true type file is valid, a <see cref="FontFamily"/> object corresponding to the specified font family.</returns>
35 FontFamily ResolveFontFamily(string fontFamily);
36
37 /// <summary>
38 /// Create a new font family from the specified family name or true type file. If the family name or the true type file are not valid, try to instantiate the font family using
39 /// the <paramref name="fallback"/>. If none of the fallback family names or true type files are valid, an exception might be raised.
40 /// </summary>
41 /// <param name="fontFamily">The name of the font family to create, or the path to a TTF file.</param>
42 /// <param name="fallback">Names of additional font families or TTF files, which will be tried if the first <paramref name="fontFamily"/> is not valid.</param>
43 /// <returns>A <see cref="FontFamily"/> object corresponding to the first of the specified font families that is valid.</returns>
44 FontFamily ResolveFontFamily(string fontFamily, params string[] fallback);
45
46 /// <summary>
47 /// Create a new font family from the specified standard font family name.
48 /// </summary>
49 /// <param name="standardFontFamily">The standard name of the font family.</param>
50 /// <returns>A <see cref="FontFamily"/> object corresponding to the specified font family.</returns>
51 FontFamily ResolveFontFamily(FontFamily.StandardFontFamilies standardFontFamily);
52
53 /// <summary>
54 /// Create a new font family from the specified family name or true type file. If the family name or the true type file are not valid, try to instantiate the font family using
55 /// the <paramref name="fallback"/>. If none of the fallback family names or true type files are valid, instantiate a standard font family using the <paramref name="finalFallback"/>.
56 /// </summary>
57 /// <param name="fontFamily">The name of the font family to create, or the path to a TTF file.</param>
58 /// <param name="fallback">Names of additional font families or TTF files, which will be tried if the first <paramref name="fontFamily"/> is not valid.</param>
59 /// <param name="finalFallback">The standard name of the font family that will be used if none of the fallback families are valid.</param>
60 /// <returns>A <see cref="FontFamily"/> object corresponding to the first of the specified font families that is valid.</returns>
61 FontFamily ResolveFontFamily(string fontFamily, FontFamily.StandardFontFamilies finalFallback, params string[] fallback);
62 }

64 /// <summary>
65 /// Abstract class with a default implementation of font family fallbacks.
66 /// </summary>
67 public abstract class FontLibrary : IFontLibrary
68 {
69 /// <inheritdoc/>
70 public abstract FontFamily ResolveFontFamily(string fontFamily);
71
72 /// <inheritdoc/>
73 public abstract FontFamily ResolveFontFamily(FontFamily.StandardFontFamilies standardFontFamily);
74
75 /// <inheritdoc/>
76 public virtual FontFamily ResolveFontFamily(string fontFamily, params string[] fallback)
77 {
78 bool found = false;
79
80 FontFamily tbr = null;
81
82 try
83 {
84 tbr = ResolveFontFamily(fontFamily);
85 if (tbr.TrueTypeFile != null)
86 {
87 found = true;
88 }
89 }
90 catch
91 {
92 tbr = null;
93 found = false;
94 }
95
96 if (!found)
97 {
98 for (int i = 0; i < fallback.Length; i++)
99 {
100 try
101 {
102 tbr = ResolveFontFamily(fontFamily);
103 if (tbr.TrueTypeFile != null)
104 {
105 found = true;
106 }
107 }
108 catch
109 {
110 tbr = null;
111 found = false;
112 }
113
114 if (found)
115 {
116 return tbr;
117 }
118 }
119 }
120 else
121 {
122 return tbr;
123 }
124
125 throw new FontFamilyCreationException(fontFamily);
126 }
127

129 public virtual FontFamily ResolveFontFamily(string fontFamily, FontFamily.StandardFontFamilies finalFallback, params string[] fallback)
130 {
131 bool found = false;
132
133 FontFamily tbr = null;
134
135 try
136 {
137 tbr = ResolveFontFamily(fontFamily);
138 if (tbr.TrueTypeFile != null)
139 {
140 found = true;
141 }
142 }
143 catch
144 {
145 tbr = null;
146 found = false;
147 }
148
149 if (!found)
150 {
151 for (int i = 0; i < fallback.Length; i++)
152 {
153 try
154 {
155 tbr = ResolveFontFamily(fontFamily);
156 if (tbr.TrueTypeFile != null)
157 {
158 found = true;
159 }
160 }
161 catch
162 {
163 tbr = null;
164 found = false;
165 }
166
167 if (found)
168 {
169 return tbr;
170 }
171 }
172 }
173 else
174 {
175 return tbr;
176 }
177
178 return ResolveFontFamily(finalFallback);

399 /// <inheritdoc/>
400 public override FontFamily ResolveFontFamily(string fontFamily)
401 {
402 if (KnownFonts.TryGetValue(fontFamily, out string knownFontName))
403 {
404 if (LoadedFonts.TryGetValue(knownFontName, out FontFamily tbr))
405 {
406 return tbr;
407 }
408 else
409 {
410 if (NotLoadedFonts.TryGetValue(knownFontName, out string ttfFile))
411 {
412 tbr = defaultLibrary.ResolveFontFamily(ttfFile);
413
414 if (tbr.TrueTypeFile != null)
415 {
416 string familyName = tbr.TrueTypeFile.GetFontFamilyName();
417
418 this.LoadedFonts[familyName] = tbr;
419 this.KnownFonts[familyName] = familyName;
420 this.KnownFonts[tbr.TrueTypeFile.GetFontName()] = familyName;
421 }
422
423 return tbr;
424 }
425 else
426 {
427 return defaultLibrary.ResolveFontFamily(fontFamily);
428 }
429 }
430 }
431 else
432 {
433 return defaultLibrary.ResolveFontFamily(fontFamily);
434 }
435 }

458 /// <summary>
459 /// A default font library that resolves standard families using the embedded fonts.
460 /// </summary>
461 public class DefaultFontLibrary : FontLibrary
462 {
463 /// <inheritdoc/>
464 public override FontFamily ResolveFontFamily(string fontFamily)
465 {
466 lock (FontFamily.fontFamilyLock)
467 {
468 bool isStandardFamily;
469
470 if (FontFamily.StandardFamilies.Contains(fontFamily) || FontFamily.StandardFamilies.Contains(fontFamily.Replace(" ", "-")))
471 {
472 isStandardFamily = true;
473 }
474 else
475 {
476 isStandardFamily = false;
477 }
478
479 if (isStandardFamily)
480 {
481 fontFamily = fontFamily.Replace(" ", "-");
482 Stream ttfStream = FontFamily.GetManifestResourceStream(FontFamily.StandardFontFamilyResources[Array.IndexOf(FontFamily.StandardFamilies, fontFamily)]);
483
484 FontFamily tbr = new FontFamily(ttfStream);
485 tbr.IsStandardFamily = true;
486 tbr.FileName = fontFamily;
487 tbr.FamilyName = tbr.FileName.Replace("-", " ");
488
489 if (fontFamily == "Times-Italic" || fontFamily == "Times-BoldItalic" || fontFamily == "Helvetica-Oblique" || fontFamily == "Helvetica-BoldOblique" || fontFamily == "Courier-Oblique" || fontFamily == "Courier-BoldOblique")
490 {
491 tbr.IsItalic = true;
492 tbr.IsOblique = (fontFamily == "Courier-Oblique" || fontFamily == "Courier-BoldOblique");
493 }
494 else
495 {
496 tbr.IsItalic = false;
497 tbr.IsOblique = false;
498 }
499
500 return tbr;
501 }
502 else
503 {
504 try
505 {
506 FontFamily tbr = new FontFamily(TrueTypeFile.CreateTrueTypeFile(fontFamily));
507 tbr.FileName = fontFamily;
508 tbr.FamilyName = tbr.TrueTypeFile?.GetFullFontFamilyName() ?? tbr.FileName;
509 return tbr;
510 }
511 catch
512 {
513 FontFamily tbr = new FontFamily();
514 tbr.FileName = fontFamily;
515 tbr.FamilyName = fontFamily;
516 return tbr;
517 }
518 }
519 }
520 }
521
522 /// <inheritdoc/>
523 public override FontFamily ResolveFontFamily(FontFamily.StandardFontFamilies standardFontFamily)
524 {
525 lock (FontFamily.fontFamilyLock)
526 {
527 Stream ttfStream = FontFamily.GetManifestResourceStream(FontFamily.StandardFontFamilyResources[(int)standardFontFamily]);
528 FontFamily tbr = new FontFamily(ttfStream);
529
530
531 tbr.IsStandardFamily = true;
532
533 tbr.FileName = FontFamily.StandardFamilies[(int)standardFontFamily];
534 tbr.FamilyName = tbr.FileName.Replace("-", " ");
535
536 if (tbr.FileName == "Times-Italic" || tbr.FileName == "Times-BoldItalic" || tbr.FileName == "Helvetica-Oblique" || tbr.FileName == "Helvetica-BoldOblique" || tbr.FileName == "Courier-Oblique" || tbr.FileName == "Courier-BoldOblique")
537 {
538 tbr.IsItalic = true;
539 tbr.IsOblique = (tbr.FileName == "Courier-Oblique" || tbr.FileName == "Courier-BoldOblique");
540 }
541 else
542 {
543 tbr.IsItalic = false;
544 tbr.IsOblique = false;
545 }
546
547 return tbr;
548 }
549 }

552 /// <summary>
553 /// A font library that resolves fonts from a folder containing TrueType files.
554 /// </summary>
555 public class FolderFontLibrary : FontLibrary
556 {
557 private Dictionary<string, FontFamily> LoadedFonts = new Dictionary<string, FontFamily>();
558 private Dictionary<string, string> KnownFonts = new Dictionary<string, string>();
559 private IFontLibrary StandardFontLibrary;
560
561 /// <summary>
562 /// Creates a new <see cref="FolderFontLibrary"/> using fonts from the specified path, and using the <see cref="FontFamily.DefaultFontLibrary"/> to resolve the standard font families.
563 /// </summary>
564 /// <param name="folderPath">The path to the folder containing the TrueType files.</param>
565 public FolderFontLibrary(string folderPath) : this(folderPath, FontFamily.DefaultFontLibrary) { }
566
567 /// <summary>
568 /// Creates a new <see cref="FolderFontLibrary"/> using fonts from the specified path, and using the specified <see cref="IFontLibrary"/> to resolve the standard font families.
569 /// </summary>
570 /// <param name="folderPath">The path to the folder containing the TrueType files.</param>
571 /// <param name="standardFontLibrary">The <see cref="IFontLibrary"/> to use when resolving standard font families.</param>
572 public FolderFontLibrary(string folderPath, IFontLibrary standardFontLibrary)
573 {
574 foreach (string file in Directory.GetFiles(folderPath))
575 {
576 bool isValid = TrueTypeFile.GetNames(file, out List<TrueTypeFile.TrueTypeName> names);
577
578 if (isValid && names != null)
579 {
580 foreach (TrueTypeFile.TrueTypeName name in names)
581 {
582 switch (name.NameId)
583 {
584 case TrueTypeFile.TrueTypeName.NameIdentifier.FullName:
585 case TrueTypeFile.TrueTypeName.NameIdentifier.PostScriptName:
586 KnownFonts[name.Name] = file;
587 break;
588 case TrueTypeFile.TrueTypeName.NameIdentifier.FontFamily:
589 case TrueTypeFile.TrueTypeName.NameIdentifier.PreferredFamily:
590 if (!KnownFonts.ContainsKey(name.Name))
591 {
592 KnownFonts[name.Name] = file;
593 }
594 break;
595 }
596 }
597 }
598 }
599
600 StandardFontLibrary = standardFontLibrary;
601 }
602
603 /// <inheritdoc/>
604 public override FontFamily ResolveFontFamily(string fontFamily)
605 {
606 if (KnownFonts.TryGetValue(fontFamily, out string knownFontName))
607 {
608 if (LoadedFonts.TryGetValue(knownFontName, out FontFamily tbr))
609 {
610 return tbr;
611 }
612 else
613 {
614 tbr = new FontFamily(TrueTypeFile.CreateTrueTypeFile(knownFontName));
615 tbr.FileName = knownFontName;
616 tbr.FamilyName = tbr.TrueTypeFile?.GetFullFontFamilyName() ?? tbr.FileName;
617
618 if (tbr.TrueTypeFile != null)
619 {
620 this.LoadedFonts[knownFontName] = tbr;
621 }
622
623 return tbr;
