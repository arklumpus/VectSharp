24 /// <summary>
25 /// A plot element that draws a violin plot.

116 /// <inheritdoc/>
117 public void Plot(Graphics target)
118 {
119
120 double min = double.MaxValue;
121 double max = double.MinValue;
122
123 for (int i = 0; i < Data.Count; i++)
124 {
125 min = Math.Min(min, Data[i]);
126 max = Math.Max(max, Data[i]);
127 }
128
129 (double _, double _, double iqr) = Plots.Plot.IQR(Data);
130 double h2 = 2 * iqr / Math.Pow(Data.Count, 1.0 / 3.0);
131 int binCount = Math.Max(1, (int)Math.Ceiling((max - min) / h2));
132
133 binCount = Math.Min(binCount, MaxBins);
134
135 int[] bins = new int[binCount];
136
137 if (max > min)
138 {
139 for (int i = 0; i < Data.Count; i++)
140 {
141 int index = (int)Math.Min(binCount - 1, Math.Floor((Data[i] - min) / (max - min) * binCount));
142
143 bins[index]++;
144 }
145 }
146 else
147 {
148 bins[0] = Data.Count;
149 }
150
151 double binMax = bins.Max();
152
153 double[] perpDirection = new double[] { Direction[1], -Direction[0] };
154
155 List<Point> points = new List<Point>();
156 List<Point> pointsOpposite = new List<Point>();
157
158 Point topPoint = CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * min, Position[1] + Direction[1] * min });
159 Point bottomPoint = CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * max, Position[1] + Direction[1] * max });
160
161 for (int i = 0; i < bins.Length; i++)
162 {
163 double binStart = min + (max - min) / binCount * i;
164 double binEnd = min + (max - min) / binCount * (i + 1);
165
166 double x = (binStart + binEnd) * 0.5;
167
168 if (i == 0)
169 {
170 if (bins.Length > 1 && bins[i + 1] > bins[i])
171 {
172 points.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binStart, Position[1] + Direction[1] * binStart }));
173 pointsOpposite.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binStart, Position[1] + Direction[1] * binStart }));
174 }
175 else
176 {
177 points.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binStart + perpDirection[0] * bins[i] / binMax * Width, Position[1] + Direction[1] * binStart + perpDirection[1] * bins[i] / binMax * Width }));
178 pointsOpposite.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binStart - perpDirection[0] * bins[i] / binMax * Width, Position[1] + Direction[1] * binStart - perpDirection[1] * bins[i] / binMax * Width }));
179 }
180 }
181
182 points.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * x + perpDirection[0] * bins[i] / binMax * Width, Position[1] + Direction[1] * x + perpDirection[1] * bins[i] / binMax * Width }));
183 pointsOpposite.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * x - perpDirection[0] * bins[i] / binMax * Width, Position[1] + Direction[1] * x - perpDirection[1] * bins[i] / binMax * Width }));
184
185 if (i == bins.Length - 1)
186 {
187 if (bins.Length > 1 && bins[i - 1] > bins[i])
188 {
189 points.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binEnd, Position[1] + Direction[1] * binEnd }));
190 pointsOpposite.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binEnd, Position[1] + Direction[1] * binEnd }));
191 }
192 else
193 {
194 points.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binEnd + perpDirection[0] * bins[i] / binMax * Width, Position[1] + Direction[1] * binEnd + perpDirection[1] * bins[i] / binMax * Width }));
195 pointsOpposite.Add(CoordinateSystem.ToPlotCoordinates(new double[] { Position[0] + Direction[0] * binEnd - perpDirection[0] * bins[i] / binMax * Width, Position[1] + Direction[1] * binEnd - perpDirection[1] * bins[i] / binMax * Width }));
196 }
197 }
198 }
199
200 pointsOpposite.Reverse();
201
202 GraphicsPath path = new GraphicsPath();
203
204 if (Sides == ViolinSide.Left || Sides == ViolinSide.Both)
205 {
206 if (Smooth)
207 {
208 path.AddSmoothSpline(pointsOpposite.ToArray());
209 }
210 else
211 {
212 for (int i= 0; i < pointsOpposite.Count; i++)
213 {
214 if (i > 0)
215 {
216 path.LineTo(pointsOpposite[i]);
217 }
218 else
219 {
220 path.MoveTo(pointsOpposite[i]);
221 }
222 }
223 }
224 }
225 else
226 {
227 path.MoveTo(bottomPoint).LineTo(topPoint);
228 }
229
230 if (Sides == ViolinSide.Right || Sides == ViolinSide.Both)
231 {
232 if (Smooth)
233 {
234 path.AddSmoothSpline(points.ToArray());
235 }
236 else
237 {
238 for (int i = 0; i < points.Count; i++)
239 {
240 path.LineTo(points[i]);
241 }
242 }
243 }
244 else
245 {
246 path.LineTo(topPoint).LineTo(bottomPoint);
247 }
248
249 path.Close();
250
251 string tag = Tag;
252
253 string strokeTag = Tag;
254
255 if (!string.IsNullOrEmpty(Tag) && target.UseUniqueTags)
256 {
257 strokeTag += "@stroke";
258 }
259
260
261 if (PresentationAttributes.Fill != null)
262 {
263 target.FillPath(path, PresentationAttributes.Fill, tag);
264 }
265
266 if (PresentationAttributes.Stroke != null)
267 {
268 target.StrokePath(path, PresentationAttributes.Stroke, PresentationAttributes.LineWidth, PresentationAttributes.LineCap, PresentationAttributes.LineJoin, PresentationAttributes.LineDash, strokeTag);
269 }
270 }
271 }
