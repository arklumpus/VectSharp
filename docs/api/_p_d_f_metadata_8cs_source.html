96 internal PDFDocumentInfo ToPDFDocumentInfo()
97 {
98 if (this.ExcludeMetadata)
99 {
100 return null;
101 }
102 else
103 {
104 PDFDocumentInfo tbr = new PDFDocumentInfo();
105
106 if (!string.IsNullOrEmpty(Title))
107 {
108 tbr.Title = new PDFString(this.Title, PDFString.StringDelimiter.Brackets);
109 }
110
111 if (!string.IsNullOrEmpty(Author))
112 {
113 tbr.Author = new PDFString(this.Author, PDFString.StringDelimiter.Brackets);
114 }
115
116 if (!string.IsNullOrEmpty(Subject))
117 {
118 tbr.Subject = new PDFString(this.Subject, PDFString.StringDelimiter.Brackets);
119 }
120
121 if (!string.IsNullOrEmpty(Keywords))
122 {
123 tbr.Keywords = new PDFString(this.Keywords, PDFString.StringDelimiter.Brackets);
124 }
125
126 if (!string.IsNullOrEmpty(Creator))
127 {
128 tbr.Creator = new PDFString(this.Creator, PDFString.StringDelimiter.Brackets);
129 }
130
131 if (!string.IsNullOrEmpty(Producer))
132 {
133 tbr.Producer = new PDFString(this.Producer, PDFString.StringDelimiter.Brackets);
134 }
135
136 if (this.CreationDate != null)
137 {
138 tbr.CreationDate = new PDFDate(this.CreationDate.Value, this.CreationDateTimeZone ?? TimeZoneInfo.Utc);
139 }
140
141 if (this.ModificationDate != null)
142 {
143 tbr.ModDate = new PDFDate(this.ModificationDate.Value, this.ModificationDateTimeZone ?? TimeZoneInfo.Utc);
144 }
145
146 if (this.CustomProperties != null)
147 {
148 foreach (KeyValuePair<string, object> kvp in this.CustomProperties)
149 {
150 if (kvp.Value != null)
151 {
152 if (kvp.Value is string s)
153 {
154 tbr.CustomProperties[kvp.Key] = new PDFString(s, PDFString.StringDelimiter.Brackets);
155 }
156 else if (kvp.Value is double d)
157 {
158 tbr.CustomProperties[kvp.Key] = new PDFString(d.ToString(System.Globalization.CultureInfo.InvariantCulture), PDFString.StringDelimiter.Brackets);
159 }
160 else if (kvp.Value is DateTime t)
161 {
162 tbr.CustomProperties[kvp.Key] = new PDFDate(t, TimeZoneInfo.Utc);
163 }
164 else
165 {
166 tbr.CustomProperties[kvp.Key] = new PDFString(kvp.Value.ToString(), PDFString.StringDelimiter.Brackets);
167 }
168 }
169 }
170 }
171
172 return tbr;
173 }
