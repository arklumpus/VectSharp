40 public static double[] Multiply(double[,] matrix, double[] vector)
41 {
42 double[] tbr = new double[2];
43
44 tbr[0] = matrix[0, 0] * vector[0] + matrix[0, 1] * vector[1] + matrix[0, 2];
45 tbr[1] = matrix[1, 0] * vector[0] + matrix[1, 1] * vector[1] + matrix[1, 2];
46
47 return tbr;

55 public static double[,] Multiply(double[,] matrix1, double[,] matrix2)
56 {
57 double[,] tbr = new double[3, 3];
58
59 for (int i = 0; i < 3; i++)
60 {
61 for (int j = 0; j < 3; j++)
62 {
63 for (int k = 0; k < 3; k++)
64 {
65 tbr[i, j] += matrix1[i, k] * matrix2[k, j];
66 }
67 }
68 }
69
70 return tbr;

109 {
110 double[,] tbr = new double[3, 3];
111
112 tbr[0, 0] = (m[1, 1] * m[2, 2] - m[1, 2] * m[2, 1]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
113 tbr[0, 1] = -(m[0, 1] * m[2, 2] - m[0, 2] * m[2, 1]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);

267 internal void DrawImage(int sourceX, int sourceY, int sourceWidth, int sourceHeight, double destinationX, double destinationY, double destinationWidth, double destinationHeight, bool interpolate, Image sourceImage)
268 {
269 sourceImage.Mutate(x => x.Crop(new SixLabors.ImageSharp.Rectangle(sourceX, sourceY, sourceWidth, sourceHeight)));
270
271 Point targetPoint = _transform.Multiply(new Point(destinationX, destinationY));
272 Point targetPointX = _transform.Multiply(new Point(destinationX + destinationWidth, destinationY));
273 Point targetPointY = _transform.Multiply(new Point(destinationX, destinationY + destinationHeight));
274 Point targetPointXY = _transform.Multiply(new Point(destinationX + destinationWidth, destinationY + destinationHeight));
275
276 double minX = Math.Min(Math.Min(targetPoint.X, targetPointX.X), Math.Min(targetPointY.X, targetPointXY.X));
277 double minY = Math.Min(Math.Min(targetPoint.Y, targetPointX.Y), Math.Min(targetPointY.Y, targetPointXY.Y));
278
279 double maxX = Math.Max(Math.Max(targetPoint.X, targetPointX.X), Math.Max(targetPointY.X, targetPointXY.X));
280 double maxY = Math.Max(Math.Max(targetPoint.Y, targetPointX.Y), Math.Max(targetPointY.Y, targetPointXY.Y));
281
282 double[,] currTransform = MatrixUtils.Multiply(MatrixUtils.Translate(_transform, destinationX, destinationY), MatrixUtils.Scale(MatrixUtils.Identity, destinationWidth / sourceImage.Width, destinationHeight / sourceImage.Height));
283
284 Point origin = currTransform.Multiply(new Point(0, 0));
285
286 double[,] translation = MatrixUtils.Translate(MatrixUtils.Identity, -minX, -minY);
287
288 double[,] centeredTransform = MatrixUtils.Multiply(translation, currTransform);
289
290 SixLabors.ImageSharp.Processing.Processors.Transforms.IResampler resampler;
291
292 if (interpolate)
293 {
294 resampler = KnownResamplers.Bicubic;
295 }
296 else
297 {
298 resampler = KnownResamplers.NearestNeighbor;
299 }
300
301 sourceImage.Mutate(x => x.Transform(new SixLabors.ImageSharp.Rectangle(sourceX, sourceY, sourceWidth, sourceHeight), centeredTransform.ToMatrix(), new SixLabors.ImageSharp.Size((int)Math.Round(maxX - minX), (int)Math.Round(maxY - minY)), resampler));
302
303 if (this.CurrentClip == null)
304 {
305 this.Image.Mutate(x => x.CheckedDrawImage(sourceImage, new SixLabors.ImageSharp.Point((int)Math.Round(minX), (int)Math.Round(minY)), 1));
306 }
307 else
308 {
309 this.ClipBuffer.Mutate(x => x.Clear(Color.FromRgba(0, 0, 0, 0)));
310 this.ClipBuffer.Mutate(x => x.CheckedDrawImage(sourceImage, new SixLabors.ImageSharp.Point((int)Math.Round(minX), (int)Math.Round(minY)), 1));
311
312 GraphicsOptions opt = new GraphicsOptions() { AlphaCompositionMode = SixLabors.ImageSharp.PixelFormats.PixelAlphaCompositionMode.SrcIn };
313
314 this.Buffer.Dispose();
315
316 this.Buffer = this.CurrentClip.Clone(x => x.CheckedDrawImage(this.ClipBuffer, opt));
317
318 this.Image.Mutate(x => x.CheckedDrawImage(this.Buffer, 1));
319 }
320 }

368 {
369 Utils.CoerceNaNAndInfinityToZero(ref x, ref y);
370
371 GraphicsPath textPath = new GraphicsPath().AddText(x, y, text, Font, TextBaseline);
372
373 for (int j = 0; j < textPath.Segments.Count; j++)
374 {

385 break;
386 case VectSharp.SegmentType.Close:
387 this.Close();
388 break;
389 }
