54 PlotElementPresentationAttributes axisPresentationAttributes = null,
55 double axisArrowSize = 3,
56 PlotElementPresentationAttributes axisLabelPresentationAttributes = null,
57 PlotElementPresentationAttributes axisTitlePresentationAttributes = null,
58 string xAxisTitle = null,
59 string yAxisTitle = null,
60 string title = null,
61 PlotElementPresentationAttributes titlePresentationAttributes = null,
62 PlotElementPresentationAttributes gridPresentationAttributes = null,
63 IReadOnlyList<PlotElementPresentationAttributes> linePresentationAttributes = null,
64 IReadOnlyList<PlotElementPresentationAttributes> pointPresentationAttributes = null,
65 IReadOnlyList<double> pointSize = null,
66 IReadOnlyList<IDataPointElement> dataPointElements = null,
67 IContinuousInvertibleCoordinateSystem coordinateSystem = null)
68 {
69 if (double.IsNaN(resolution))
70 {
71 if (coordinateSystem == null)
72 {
73 resolution = (max - min) / 500;
74 }
75 else
76 {
77 resolution = coordinateSystem.Resolution[0];
78 }
79 }
80
81 int steps = (int)Math.Ceiling((max - min) / resolution);
82
83 List<double[]>[] data = new List<double[]>[functions.Count];
84
85 for (int i = 0; i < functions.Count; i++)
86 {
87 data[i] = new List<double[]>(steps + 1);
88
89 for (int j = 0; j <= steps; j++)
90 {
91 double x = min + (max - min) / steps * j;
92 double y = functions[i](x);
93
94 if (!double.IsNaN(y))
95 {
96 if (vertical)
97 {
98 data[i].Add(new double[] { x, y });
99 }
100 else
101 {
102 data[i].Add(new double[] { y, x });
103 }
104 }
105 }
106 }
107
108 return Create.LineCharts(data, smooth, width, height, axisPresentationAttributes, axisArrowSize, axisLabelPresentationAttributes, axisTitlePresentationAttributes, xAxisTitle, yAxisTitle, title, titlePresentationAttributes, gridPresentationAttributes, linePresentationAttributes, pointPresentationAttributes, pointSize, dataPointElements, coordinateSystem);
109 }
