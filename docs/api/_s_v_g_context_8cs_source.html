31 public static double[] Multiply(double[,] matrix, double[] vector)
32 {
33 double[] tbr = new double[2];
34
35 tbr[0] = matrix[0, 0] * vector[0] + matrix[0, 1] * vector[1] + matrix[0, 2];
36 tbr[1] = matrix[1, 0] * vector[0] + matrix[1, 1] * vector[1] + matrix[1, 2];
37
38 return tbr;
39 }

41 public static Point Multiply(double[,] matrix, Point vector)
42 {
43 double[] tbr = new double[2];
44
45 tbr[0] = matrix[0, 0] * vector.X + matrix[0, 1] * vector.Y + matrix[0, 2];
46 tbr[1] = matrix[1, 0] * vector.X + matrix[1, 1] * vector.Y + matrix[1, 2];
47
48 return new Point(tbr[0], tbr[1]);
49 }

51 public static double[,] Multiply(double[,] matrix1, double[,] matrix2)
52 {
53 double[,] tbr = new double[3, 3];
54
55 for (int i = 0; i < 3; i++)
56 {
57 for (int j = 0; j < 3; j++)
58 {
59 for (int k = 0; k < 3; k++)
60 {
61 tbr[i, j] += matrix1[i, k] * matrix2[k, j];
62 }
63 }
64 }
65
66 return tbr;
67 }

69 public static double[,] Rotate(double[,] matrix, double angle)
70 {
71 double[,] rotationMatrix = new double[3, 3];
72 rotationMatrix[0, 0] = Math.Cos(angle);
73 rotationMatrix[0, 1] = -Math.Sin(angle);
74 rotationMatrix[1, 0] = Math.Sin(angle);
75 rotationMatrix[1, 1] = Math.Cos(angle);
76 rotationMatrix[2, 2] = 1;
77
78 return Multiply(matrix, rotationMatrix);
79 }

81 public static double[,] Translate(double[,] matrix, double x, double y)
82 {
83 double[,] translationMatrix = new double[3, 3];
84 translationMatrix[0, 0] = 1;
85 translationMatrix[0, 2] = x;
86 translationMatrix[1, 1] = 1;
87 translationMatrix[1, 2] = y;
88 translationMatrix[2, 2] = 1;
89
90 return Multiply(matrix, translationMatrix);

105 public static double[,] Invert(double[,] m)
106 {
107 double[,] tbr = new double[3, 3];
108
109 tbr[0, 0] = (m[1, 1] * m[2, 2] - m[1, 2] * m[2, 1]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
110 tbr[0, 1] = -(m[0, 1] * m[2, 2] - m[0, 2] * m[2, 1]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
111 tbr[0, 2] = (m[0, 1] * m[1, 2] - m[0, 2] * m[1, 1]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
112 tbr[1, 0] = -(m[1, 0] * m[2, 2] - m[1, 2] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
113 tbr[1, 1] = (m[0, 0] * m[2, 2] - m[0, 2] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
114 tbr[1, 2] = -(m[0, 0] * m[1, 2] - m[0, 2] * m[1, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
115 tbr[2, 0] = (m[1, 0] * m[2, 1] - m[1, 1] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
116 tbr[2, 1] = -(m[0, 0] * m[2, 1] - m[0, 1] * m[2, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);
117 tbr[2, 2] = (m[0, 0] * m[1, 1] - m[0, 1] * m[1, 0]) / (m[0, 0] * m[1, 1] * m[2, 2] - m[0, 0] * m[1, 2] * m[2, 1] - m[1, 0] * m[0, 1] * m[2, 2] + m[2, 0] * m[0, 1] * m[1, 2] + m[1, 0] * m[0, 2] * m[2, 1] - m[2, 0] * m[0, 2] * m[1, 1]);

191 private string GetClass(Dictionary<string, string> style)
192 {
193 foreach (KeyValuePair<string, Dictionary<string, string>> styleClass in Styles)
194 {
195 if (styleClass.Value.Count == style.Count)
196 {
197 bool mismatch = false;
198
199 foreach (KeyValuePair<string, string> styleItem in style)
200 {
201 if (!styleClass.Value.TryGetValue(styleItem.Key, out string value))
202 {
203 mismatch = true;
204 break;
205 }
206 else if (value != styleItem.Value)
207 {
208 mismatch = true;
209 break;
210 }
211 }
212
213 if (!mismatch)
214 {
215 return styleClass.Key;
216 }
217 }
218 }
219
220 string tbr = "class" + (Styles.Count + 1).ToString();
221
222 Styles.Add(tbr, style);
223
224 return tbr;

290 private void PathText(string text, double x, double y)
291 {
292 GraphicsPath textPath = new GraphicsPath().AddText(x, y, text, Font, TextBaseline);
293
294 for (int j = 0; j < textPath.Segments.Count; j++)
295 {
296 switch (textPath.Segments[j].Type)
297 {
298 case VectSharp.SegmentType.Move:
299 this.MoveTo(textPath.Segments[j].Point.X, textPath.Segments[j].Point.Y);
300 break;
301 case VectSharp.SegmentType.Line:
302 this.LineTo(textPath.Segments[j].Point.X, textPath.Segments[j].Point.Y);
303 break;
304 case VectSharp.SegmentType.CubicBezier:
305 this.CubicBezierTo(textPath.Segments[j].Points[0].X, textPath.Segments[j].Points[0].Y, textPath.Segments[j].Points[1].X, textPath.Segments[j].Points[1].Y, textPath.Segments[j].Points[2].X, textPath.Segments[j].Points[2].Y);
306 break;
307 case VectSharp.SegmentType.Close:
308 this.Close();
309 break;
310 }
311 }

460 public void FillText(string text, double x, double y)
461 {
462 if (!TextToPaths)
463 {
464 if (!UsedFontFamilies.ContainsKey(Font.FontFamily.FileName))
465 {
466 UsedFontFamilies.Add(Font.FontFamily.FileName, Font.FontFamily);
467 UsedChars.Add(Font.FontFamily.FileName, new HashSet<char>());
468 }
469
470 UsedChars[Font.FontFamily.FileName].UnionWith(text);
471
472 Font.DetailedFontMetrics metrics = Font.MeasureTextAdvanced(text);
473
474 double[,] currTransform = null;
475 double[,] deltaTransform = MatrixUtils.Identity;
476
477 switch (TextBaseline)
478 {
479 case TextBaselines.Baseline:
480 currTransform = MatrixUtils.Translate(_transform, x - metrics.LeftSideBearing, y);
481 deltaTransform = MatrixUtils.Translate(deltaTransform, x - metrics.LeftSideBearing, y);
482 break;
483 case TextBaselines.Top:
484 currTransform = MatrixUtils.Translate(_transform, x - metrics.LeftSideBearing, y + metrics.Top);
485 deltaTransform = MatrixUtils.Translate(deltaTransform, x - metrics.LeftSideBearing, y + metrics.Top);
486 break;
487 case TextBaselines.Bottom:
488 currTransform = MatrixUtils.Translate(_transform, x - metrics.LeftSideBearing, y + metrics.Bottom);
489 deltaTransform = MatrixUtils.Translate(deltaTransform, x - metrics.LeftSideBearing, y + metrics.Bottom);
490 break;
491 case TextBaselines.Middle:
492 currTransform = MatrixUtils.Translate(_transform, x - metrics.LeftSideBearing, y + (metrics.Top + metrics.Bottom) * 0.5);
493 deltaTransform = MatrixUtils.Translate(deltaTransform, x - metrics.LeftSideBearing, y + (metrics.Top + metrics.Bottom) * 0.5);
494 break;
495 default:
496 currTransform = MatrixUtils.Translate(_transform, x - metrics.LeftSideBearing, y);

498 break;
